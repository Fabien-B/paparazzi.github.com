73 void ahrs_update_gps(void) {
74 
75  float hspeed_mod_f = gps.gspeed / 100.;
76  float course_f = gps.course / 1e7;
77 
78  // Heading estimator from wind-information, usually computed with -DWIND_INFO
79  // wind_north and wind_east initialized to 0, so still correct if not updated
80  float w_vn = cosf(course_f) * hspeed_mod_f - wind_north;
81  float w_ve = sinf(course_f) * hspeed_mod_f - wind_east;
82  ahrs_float.ltp_to_body_euler.psi = atan2f(w_ve, w_vn);
83  if (ahrs_float.ltp_to_body_euler.psi < 0.)
84  ahrs_float.ltp_to_body_euler.psi += 2 * M_PI;
85 
86  ahrs_update_fw_estimator();
87 }

89 void ahrs_update_infrared(void) {
90  ahrs_float.ltp_to_body_euler.phi = atan2(infrared.roll, infrared.top) - infrared.roll_neutral;
91 
92  ahrs_float.ltp_to_body_euler.theta = atan2(infrared.pitch, infrared.top) - infrared.pitch_neutral;
93 
94  if (ahrs_float.ltp_to_body_euler.theta < -M_PI_2)
95  ahrs_float.ltp_to_body_euler.theta += M_PI;
96  else if (ahrs_float.ltp_to_body_euler.theta > M_PI_2)
97  ahrs_float.ltp_to_body_euler.theta -= M_PI;
98 
99  if (ahrs_float.ltp_to_body_euler.phi >= 0)
100  ahrs_float.ltp_to_body_euler.phi *= infrared.correction_right;
101  else
102  ahrs_float.ltp_to_body_euler.phi *= infrared.correction_left;
103 
104  if (ahrs_float.ltp_to_body_euler.theta >= 0)
105  ahrs_float.ltp_to_body_euler.theta *= infrared.correction_up;
106  else
107  ahrs_float.ltp_to_body_euler.theta *= infrared.correction_down;
108 
109  ahrs_update_fw_estimator();
110 }
