142 STATIC_INLINE void handle_periodic_tasks( void ) {
143  if (sys_time_check_and_ack_timer(main_periodic_tid))
144  main_periodic();
145  if (sys_time_check_and_ack_timer(radio_control_tid))
146  radio_control_periodic_task();
147  if (sys_time_check_and_ack_timer(failsafe_tid))
148  failsafe_check();
149  if (sys_time_check_and_ack_timer(electrical_tid))
150  electrical_periodic();
151  if (sys_time_check_and_ack_timer(baro_tid))
152  baro_periodic();
153  if (sys_time_check_and_ack_timer(telemetry_tid))
154  telemetry_periodic();
155 }

179 STATIC_INLINE void failsafe_check( void ) {
180  if (radio_control.status != RC_OK &&
181  autopilot_mode != AP_MODE_KILL &&
182  autopilot_mode != AP_MODE_NAV)
183  {
184  autopilot_set_mode(AP_MODE_FAILSAFE);
185  }
186 
187 #if USE_GPS
188  if (radio_control.status != RC_OK &&
189  autopilot_mode == AP_MODE_NAV &&
190  GpsIsLost())
191  {
192  autopilot_set_mode(AP_MODE_FAILSAFE);
193  }
194 #endif
195 }
