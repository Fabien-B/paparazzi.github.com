102 void stabilization_attitude_run(bool_t in_flight) {
103 
104 
105  /* update reference */
106  stabilization_attitude_ref_update();
107 
108  /* compute feedforward command */
109  stabilization_att_ff_cmd[COMMAND_ROLL] =
110  OFFSET_AND_ROUND(stabilization_gains.dd.x * stab_att_ref_accel.p, 5);
111  stabilization_att_ff_cmd[COMMAND_PITCH] =
112  OFFSET_AND_ROUND(stabilization_gains.dd.y * stab_att_ref_accel.q, 5);
113  stabilization_att_ff_cmd[COMMAND_YAW] =
114  OFFSET_AND_ROUND(stabilization_gains.dd.z * stab_att_ref_accel.r, 5);
115 
116  /* compute feedback command */
117  /* attitude error */
118  const struct Int32Eulers att_ref_scaled = {
119  OFFSET_AND_ROUND(stab_att_ref_euler.phi, (REF_ANGLE_FRAC - INT32_ANGLE_FRAC)),
120  OFFSET_AND_ROUND(stab_att_ref_euler.theta, (REF_ANGLE_FRAC - INT32_ANGLE_FRAC)),
121  OFFSET_AND_ROUND(stab_att_ref_euler.psi, (REF_ANGLE_FRAC - INT32_ANGLE_FRAC)) };
122  struct Int32Eulers att_err;
123  EULERS_DIFF(att_err, att_ref_scaled, ahrs.ltp_to_body_euler);
124  INT32_ANGLE_NORMALIZE(att_err.psi);
125 
126  if (in_flight) {
127  /* update integrator */
128  EULERS_ADD(stabilization_att_sum_err, att_err);
129  EULERS_BOUND_CUBE(stabilization_att_sum_err, -MAX_SUM_ERR, MAX_SUM_ERR);
130  }
131  else {
132  INT_EULERS_ZERO(stabilization_att_sum_err);
133  }
134 
135  /* rate error */
136  const struct Int32Rates rate_ref_scaled = {
137  OFFSET_AND_ROUND(stab_att_ref_rate.p, (REF_RATE_FRAC - INT32_RATE_FRAC)),
138  OFFSET_AND_ROUND(stab_att_ref_rate.q, (REF_RATE_FRAC - INT32_RATE_FRAC)),
139  OFFSET_AND_ROUND(stab_att_ref_rate.r, (REF_RATE_FRAC - INT32_RATE_FRAC)) };
140  struct Int32Rates rate_err;
141  RATES_DIFF(rate_err, rate_ref_scaled, ahrs.body_rate);
142 
143  /* PID */
144  stabilization_att_fb_cmd[COMMAND_ROLL] =
145  stabilization_gains.p.x * att_err.phi +
146  stabilization_gains.d.x * rate_err.p +
147  OFFSET_AND_ROUND2((stabilization_gains.i.x * stabilization_att_sum_err.phi), 10);
148 
149  stabilization_att_fb_cmd[COMMAND_PITCH] =
150  stabilization_gains.p.y * att_err.theta +
151  stabilization_gains.d.y * rate_err.q +
152  OFFSET_AND_ROUND2((stabilization_gains.i.y * stabilization_att_sum_err.theta), 10);
153 
154  stabilization_att_fb_cmd[COMMAND_YAW] =
155  stabilization_gains.p.z * att_err.psi +
156  stabilization_gains.d.z * rate_err.r +
157  OFFSET_AND_ROUND2((stabilization_gains.i.z * stabilization_att_sum_err.psi), 10);
158 
159 
160  /* with P gain of 100, att_err of 180deg (3.14 rad)
161  * fb cmd: 100 * 3.14 * 2^12 / 2^CMD_SHIFT = 628
162  * max possible command is 9600
163  */
164 #define CMD_SHIFT 11
165 
166  /* sum feedforward and feedback */
167  stabilization_cmd[COMMAND_ROLL] =
168  OFFSET_AND_ROUND((stabilization_att_fb_cmd[COMMAND_ROLL]+stabilization_att_ff_cmd[COMMAND_ROLL]), CMD_SHIFT);
169 
170  stabilization_cmd[COMMAND_PITCH] =
171  OFFSET_AND_ROUND((stabilization_att_fb_cmd[COMMAND_PITCH]+stabilization_att_ff_cmd[COMMAND_PITCH]), CMD_SHIFT);
172 
173  stabilization_cmd[COMMAND_YAW] =
174  OFFSET_AND_ROUND((stabilization_att_fb_cmd[COMMAND_YAW]+stabilization_att_ff_cmd[COMMAND_YAW]), CMD_SHIFT);
175 
176  /* bound the result */
177  BoundAbs(stabilization_cmd[COMMAND_ROLL], MAX_PPRZ);
178  BoundAbs(stabilization_cmd[COMMAND_PITCH], MAX_PPRZ);
179  BoundAbs(stabilization_cmd[COMMAND_YAW], MAX_PPRZ);
180 
181 }
