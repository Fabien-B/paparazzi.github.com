142 void ins_propagate() {
143  /* untilt accels */
144  struct Int32Vect3 accel_meas_body;
145  INT32_RMAT_TRANSP_VMULT(accel_meas_body, imu.body_to_imu_rmat, imu.accel);
146  struct Int32Vect3 accel_meas_ltp;
147  INT32_RMAT_TRANSP_VMULT(accel_meas_ltp, ahrs.ltp_to_body_rmat, accel_meas_body);
148 
149 #if USE_VFF
150  float z_accel_meas_float = ACCEL_FLOAT_OF_BFP(accel_meas_ltp.z);
151  if (baro.status == BS_RUNNING && ins_baro_initialised) {
152  vff_propagate(z_accel_meas_float);
153  ins_ltp_accel.z = ACCEL_BFP_OF_REAL(vff_zdotdot);
154  ins_ltp_speed.z = SPEED_BFP_OF_REAL(vff_zdot);
155  ins_ltp_pos.z = POS_BFP_OF_REAL(vff_z);
156  }
157  else { // feed accel from the sensors
158  // subtract -9.81m/s2 (acceleration measured due to gravity, but vehivle not accelerating in ltp)
159  ins_ltp_accel.z = accel_meas_ltp.z + ACCEL_BFP_OF_REAL(9.81);
160  }
161 #else
162  ins_ltp_accel.z = accel_meas_ltp.z + ACCEL_BFP_OF_REAL(9.81);
163 #endif /* USE_VFF */
164 
165 #if USE_HFF
166  /* propagate horizontal filter */
167  b2_hff_propagate();
168 #else
169  ins_ltp_accel.x = accel_meas_ltp.x;
170  ins_ltp_accel.y = accel_meas_ltp.y;
171 #endif /* USE_HFF */
172 
173  INT32_VECT3_ENU_OF_NED(ins_enu_pos, ins_ltp_pos);
174  INT32_VECT3_ENU_OF_NED(ins_enu_speed, ins_ltp_speed);
175  INT32_VECT3_ENU_OF_NED(ins_enu_accel, ins_ltp_accel);
176 }

178 void ins_update_baro() {
179 #if USE_VFF
180  if (baro.status == BS_RUNNING) {
181  if (!ins_baro_initialised) {
182  ins_qfe = baro.absolute;
183  ins_baro_initialised = TRUE;
184  }
185  if (ins_vf_realign) {
186  ins_vf_realign = FALSE;
187  ins_qfe = baro.absolute;
188 #if USE_SONAR
189  ins_sonar_offset = sonar_meas;
190 #endif
191  vff_realign(0.);
192  ins_ltp_accel.z = ACCEL_BFP_OF_REAL(vff_zdotdot);
193  ins_ltp_speed.z = SPEED_BFP_OF_REAL(vff_zdot);
194  ins_ltp_pos.z = POS_BFP_OF_REAL(vff_z);
195  ins_enu_pos.z = -ins_ltp_pos.z;
196  ins_enu_speed.z = -ins_ltp_speed.z;
197  ins_enu_accel.z = -ins_ltp_accel.z;
198  }
199  else { /* not realigning, so normal update with baro measurement */
200  ins_baro_alt = ((baro.absolute - ins_qfe) * INS_BARO_SENS_NUM)/INS_BARO_SENS_DEN;
201  float alt_float = POS_FLOAT_OF_BFP(ins_baro_alt);
202  vff_update(alt_float);
203  }
204  }
205 #endif
206 }

209 void ins_update_gps(void) {
210 #if USE_GPS
211  if (gps.fix == GPS_FIX_3D) {
212  if (!ins_ltp_initialised) {
213  ltp_def_from_ecef_i(&ins_ltp_def, &gps.ecef_pos);
214  ins_ltp_def.lla.alt = gps.lla_pos.alt;
215  ins_ltp_def.hmsl = gps.hmsl;
216  ins_ltp_initialised = TRUE;
217  }
218  ned_of_ecef_point_i(&ins_gps_pos_cm_ned, &ins_ltp_def, &gps.ecef_pos);
219  ned_of_ecef_vect_i(&ins_gps_speed_cm_s_ned, &ins_ltp_def, &gps.ecef_vel);
220 #if USE_HFF
221  VECT2_ASSIGN(ins_gps_pos_m_ned, ins_gps_pos_cm_ned.x, ins_gps_pos_cm_ned.y);
222  VECT2_SDIV(ins_gps_pos_m_ned, ins_gps_pos_m_ned, 100.);
223  VECT2_ASSIGN(ins_gps_speed_m_s_ned, ins_gps_speed_cm_s_ned.x, ins_gps_speed_cm_s_ned.y);
224  VECT2_SDIV(ins_gps_speed_m_s_ned, ins_gps_speed_m_s_ned, 100.);
225  if (ins_hf_realign) {
226  ins_hf_realign = FALSE;
227 #ifdef SITL
228  struct FloatVect2 true_pos, true_speed;
229  VECT2_COPY(true_pos, fdm.ltpprz_pos);
230  VECT2_COPY(true_speed, fdm.ltpprz_ecef_vel);
231  b2_hff_realign(true_pos, true_speed);
232 #else
233  const struct FloatVect2 zero = {0.0, 0.0};
234  b2_hff_realign(ins_gps_pos_m_ned, zero);
235 #endif
236  }
237  b2_hff_update_gps();
238 #if !USE_VFF /* vff not used */
239  ins_ltp_pos.z = (ins_gps_pos_cm_ned.z * INT32_POS_OF_CM_NUM) / INT32_POS_OF_CM_DEN;
240  ins_ltp_speed.z = (ins_gps_speed_cm_s_ned.z * INT32_SPEED_OF_CM_S_NUM) INT32_SPEED_OF_CM_S_DEN;
241 #endif /* vff not used */
242 #endif /* hff used */
243 
244 
245 #if !USE_HFF /* hff not used */
246 #if !USE_VFF /* neither hf nor vf used */
247  INT32_VECT3_SCALE_3(ins_ltp_pos, ins_gps_pos_cm_ned, INT32_POS_OF_CM_NUM, INT32_POS_OF_CM_DEN);
248  INT32_VECT3_SCALE_3(ins_ltp_speed, ins_gps_speed_cm_s_ned, INT32_SPEED_OF_CM_S_NUM, INT32_SPEED_OF_CM_S_DEN);
249 #else /* only vff used */
250  INT32_VECT2_SCALE_2(ins_ltp_pos, ins_gps_pos_cm_ned, INT32_POS_OF_CM_NUM, INT32_POS_OF_CM_DEN);
251  INT32_VECT2_SCALE_2(ins_ltp_speed, ins_gps_speed_cm_s_ned, INT32_SPEED_OF_CM_S_NUM, INT32_SPEED_OF_CM_S_DEN);
252 #endif
253 
254 #if USE_GPS_LAG_HACK
255  VECT2_COPY(d_pos, ins_ltp_speed);
256  INT32_VECT2_RSHIFT(d_pos, d_pos, 11);
257  VECT2_ADD(ins_ltp_pos, d_pos);
258 #endif
259 #endif /* hff not used */
260 
261  INT32_VECT3_ENU_OF_NED(ins_enu_pos, ins_ltp_pos);
262  INT32_VECT3_ENU_OF_NED(ins_enu_speed, ins_ltp_speed);
263  INT32_VECT3_ENU_OF_NED(ins_enu_accel, ins_ltp_accel);
264  }
265 #endif /* USE_GPS */
266 }

268 void ins_update_sonar() {
269 #if USE_SONAR && USE_VFF
270  static int32_t sonar_filtered = 0;
271  sonar_filtered = (sonar_meas + 2*sonar_filtered) / 3;
272  /* update baro_qfe assuming a flat ground */
273  if (ins_update_on_agl && baro.status == BS_RUNNING) {
274  int32_t d_sonar = (((int32_t)sonar_filtered - ins_sonar_offset) * INS_SONAR_SENS_NUM) / INS_SONAR_SENS_DEN;
275  ins_qfe = baro.absolute + (d_sonar * (INS_BARO_SENS_DEN))/INS_BARO_SENS_NUM;
276  }
277 #endif
