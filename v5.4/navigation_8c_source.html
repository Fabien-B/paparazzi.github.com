186 static inline void UNUSED nav_advance_carrot(void)
187 {
188  struct EnuCoor_i *pos = stateGetPositionEnu_i();
189  /* compute a vector to the waypoint */
190  struct Int32Vect2 path_to_waypoint;
191  VECT2_DIFF(path_to_waypoint, navigation_target, *pos);
192 
193  /* saturate it */
194  VECT2_STRIM(path_to_waypoint, -(1 << 15), (1 << 15));
195 
196  int32_t dist_to_waypoint = int32_vect2_norm(&path_to_waypoint);
197 
198  if (dist_to_waypoint < CLOSE_TO_WAYPOINT) {
199  VECT2_COPY(navigation_carrot, navigation_target);
200  } else {
201  struct Int32Vect2 path_to_carrot;
202  VECT2_SMUL(path_to_carrot, path_to_waypoint, CARROT_DIST);
203  VECT2_SDIV(path_to_carrot, path_to_carrot, dist_to_waypoint);
204  VECT2_SUM(navigation_carrot, path_to_carrot, *pos);
205  }
206 }

221 void nav_circle(struct EnuCoor_i *wp_center, int32_t radius)
222 {
223  if (radius == 0) {
224  VECT2_COPY(navigation_target, *wp_center);
225  dist2_to_wp = get_dist2_to_point(wp_center);
226  } else {
227  struct Int32Vect2 pos_diff;
228  VECT2_DIFF(pos_diff, *stateGetPositionEnu_i(), *wp_center);
229  // go back to half metric precision or values are too large
230  //INT32_VECT2_RSHIFT(pos_diff,pos_diff,INT32_POS_FRAC/2);
231  // store last qdr
232  int32_t last_qdr = nav_circle_qdr;
233  // compute qdr
234  nav_circle_qdr = int32_atan2(pos_diff.y, pos_diff.x);
235  // increment circle radians
236  if (nav_circle_radians != 0) {
237  int32_t angle_diff = nav_circle_qdr - last_qdr;
238  INT32_ANGLE_NORMALIZE(angle_diff);
239  nav_circle_radians += angle_diff;
240  } else {
241  // Smallest angle to increment at next step
242  nav_circle_radians = 1;
243  }
244 
245  // direction of rotation
246  int8_t sign_radius = radius > 0 ? 1 : -1;
247  // absolute radius
248  int32_t abs_radius = abs(radius);
249  // carrot_angle
250  int32_t carrot_angle = ((CARROT_DIST << INT32_ANGLE_FRAC) / abs_radius);
251  Bound(carrot_angle, (INT32_ANGLE_PI / 16), INT32_ANGLE_PI_4);
252  carrot_angle = nav_circle_qdr - sign_radius * carrot_angle;
253  int32_t s_carrot, c_carrot;
254  PPRZ_ITRIG_SIN(s_carrot, carrot_angle);
255  PPRZ_ITRIG_COS(c_carrot, carrot_angle);
256  // compute setpoint
257  VECT2_ASSIGN(pos_diff, abs_radius * c_carrot, abs_radius * s_carrot);
258  INT32_VECT2_RSHIFT(pos_diff, pos_diff, INT32_TRIG_FRAC);
259  VECT2_SUM(navigation_target, *wp_center, pos_diff);
260  }
261  nav_circle_center = *wp_center;
262  nav_circle_radius = radius;
263  horizontal_mode = HORIZONTAL_MODE_CIRCLE;
264 }

267 void nav_route(struct EnuCoor_i *wp_start, struct EnuCoor_i *wp_end)
268 {
269  struct Int32Vect2 wp_diff, pos_diff, wp_diff_prec;
270  VECT2_DIFF(wp_diff, *wp_end, *wp_start);
271  VECT2_DIFF(pos_diff, *stateGetPositionEnu_i(), *wp_start);
272  // go back to metric precision or values are too large
273  VECT2_COPY(wp_diff_prec, wp_diff);
274  INT32_VECT2_RSHIFT(wp_diff, wp_diff, INT32_POS_FRAC);
275  INT32_VECT2_RSHIFT(pos_diff, pos_diff, INT32_POS_FRAC);
276  uint32_t leg_length2 = Max((wp_diff.x * wp_diff.x + wp_diff.y * wp_diff.y), 1);
277  nav_leg_length = int32_sqrt(leg_length2);
278  nav_leg_progress = (pos_diff.x * wp_diff.x + pos_diff.y * wp_diff.y) / nav_leg_length;
279  int32_t progress = Max((CARROT_DIST >> INT32_POS_FRAC), 0);
280  nav_leg_progress += progress;
281  int32_t prog_2 = nav_leg_length;
282  Bound(nav_leg_progress, 0, prog_2);
283  struct Int32Vect2 progress_pos;
284  VECT2_SMUL(progress_pos, wp_diff_prec, ((float)nav_leg_progress) / nav_leg_length);
285  VECT2_SUM(navigation_target, *wp_start, progress_pos);
286 
287  nav_segment_start = *wp_start;
288  nav_segment_end = *wp_end;
289  horizontal_mode = HORIZONTAL_MODE_ROUTE;
290 
291  dist2_to_wp = get_dist2_to_point(wp_end);
292 }

294 bool_t nav_approaching_from(struct EnuCoor_i *wp, struct EnuCoor_i *from, int16_t approaching_time)
295 {
296  int32_t dist_to_point;
297  struct Int32Vect2 diff;
298  struct EnuCoor_i *pos = stateGetPositionEnu_i();
299 
300  /* if an approaching_time is given, estimate diff after approching_time secs */
301  if (approaching_time > 0) {
302  struct Int32Vect2 estimated_pos;
303  struct Int32Vect2 estimated_progress;
304  struct EnuCoor_i *speed = stateGetSpeedEnu_i();
305  VECT2_SMUL(estimated_progress, *speed, approaching_time);
306  INT32_VECT2_RSHIFT(estimated_progress, estimated_progress, (INT32_SPEED_FRAC - INT32_POS_FRAC));
307  VECT2_SUM(estimated_pos, *pos, estimated_progress);
308  VECT2_DIFF(diff, *wp, estimated_pos);
309  }
310  /* else use current position */
311  else {
312  VECT2_DIFF(diff, *wp, *pos);
313  }
314  /* compute distance of estimated/current pos to target wp
315  * distance with half metric precision (6.25 cm)
316  */
317  INT32_VECT2_RSHIFT(diff, diff, INT32_POS_FRAC / 2);
318  dist_to_point = int32_vect2_norm(&diff);
319 
320  /* return TRUE if we have arrived */
321  if (dist_to_point < BFP_OF_REAL(ARRIVED_AT_WAYPOINT, INT32_POS_FRAC / 2)) {
322  return TRUE;
323  }
324 
325  /* if coming from a valid waypoint */
326  if (from != NULL) {
327  /* return TRUE if normal line at the end of the segment is crossed */
328  struct Int32Vect2 from_diff;
329  VECT2_DIFF(from_diff, *wp, *from);
330  INT32_VECT2_RSHIFT(from_diff, from_diff, INT32_POS_FRAC / 2);
331  return (diff.x * from_diff.x + diff.y * from_diff.y < 0);
332  }
333 
334  return FALSE;
335 }

337 bool_t nav_check_wp_time(struct EnuCoor_i *wp, uint16_t stay_time)
338 {
339  uint16_t time_at_wp;
340  uint32_t dist_to_point;
341  static uint16_t wp_entry_time = 0;
342  static bool_t wp_reached = FALSE;
343  static struct EnuCoor_i wp_last = { 0, 0, 0 };
344  struct Int32Vect2 diff;
345 
346  if ((wp_last.x != wp->x) || (wp_last.y != wp->y)) {
347  wp_reached = FALSE;
348  wp_last = *wp;
349  }
350  VECT2_DIFF(diff, *wp, *stateGetPositionEnu_i());
351  INT32_VECT2_RSHIFT(diff, diff, INT32_POS_FRAC / 2);
352  dist_to_point = int32_vect2_norm(&diff);
353  if (dist_to_point < BFP_OF_REAL(ARRIVED_AT_WAYPOINT, INT32_POS_FRAC / 2)) {
354  if (!wp_reached) {
355  wp_reached = TRUE;
356  wp_entry_time = autopilot_flight_time;
357  time_at_wp = 0;
358  } else {
359  time_at_wp = autopilot_flight_time - wp_entry_time;
360  }
361  } else {
362  time_at_wp = 0;
363  wp_reached = FALSE;
364  }
365  if (time_at_wp > stay_time) {
366  INT_VECT3_ZERO(wp_last);
367  return TRUE;
368  }
369  return FALSE;
370 }
