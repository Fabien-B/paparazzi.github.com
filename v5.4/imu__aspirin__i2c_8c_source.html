83 void imu_impl_init(void)
84 {
85  imu_aspirin.accel_valid = FALSE;
86  imu_aspirin.gyro_valid = FALSE;
87  imu_aspirin.mag_valid = FALSE;
88 
89  /* Set accel configuration */
90  adxl345_i2c_init(&imu_aspirin.acc_adxl, &(ASPIRIN_I2C_DEV), ADXL345_ADDR);
91  // set the data rate
92  imu_aspirin.acc_adxl.config.rate = ASPIRIN_ACCEL_RATE;
94  //imu_aspirin.acc_adxl.config.drdy_int_enable = TRUE;
95 
96  // With CS tied high to VDD I/O, the ADXL345 is in I2C mode
97 #ifdef ASPIRIN_I2C_CS_PORT
98  gpio_setup_output(ASPIRIN_I2C_CS_PORT, ASPIRIN_I2C_CS_PIN);
99  gpio_set(ASPIRIN_I2C_CS_PORT, ASPIRIN_I2C_CS_PIN);
100 #endif
101 
102  /* Gyro configuration and initalization */
103  itg3200_init(&imu_aspirin.gyro_itg, &(ASPIRIN_I2C_DEV), ITG3200_ADDR);
104  /* change the default config */
105  // Aspirin sample rate divider
106  imu_aspirin.gyro_itg.config.smplrt_div = ASPIRIN_GYRO_SMPLRT_DIV;
107  // digital low pass filter
108  imu_aspirin.gyro_itg.config.dlpf_cfg = ASPIRIN_GYRO_LOWPASS;
109 
111  /* interrupt on data ready, idle high, latch until read any register */
112  //itg_conf.int_cfg = (0x01 | (0x1<<4) | (0x1<<5) | 0x01<<7);
113 
114  /* initialize mag and set default options */
115  hmc58xx_init(&imu_aspirin.mag_hmc, &(ASPIRIN_I2C_DEV), HMC58XX_ADDR);
116 #ifdef IMU_ASPIRIN_VERSION_1_0
117  imu_aspirin.mag_hmc.type = HMC_TYPE_5843;
118 #endif
119 }

122 void imu_periodic(void)
123 {
124  adxl345_i2c_periodic(&imu_aspirin.acc_adxl);
125 
126  // Start reading the latest gyroscope data
127  itg3200_periodic(&imu_aspirin.gyro_itg);
128

133 void imu_aspirin_i2c_event(void)
134 {
135  adxl345_i2c_event(&imu_aspirin.acc_adxl);
136  if (imu_aspirin.acc_adxl.data_available) {
137  VECT3_COPY(imu.accel_unscaled, imu_aspirin.acc_adxl.data.vect);
138  imu_aspirin.acc_adxl.data_available = FALSE;
139  imu_aspirin.accel_valid = TRUE;
140  }
141 
142  /* If the itg3200 I2C transaction has succeeded: convert the data */
143  itg3200_event(&imu_aspirin.gyro_itg);
144  if (imu_aspirin.gyro_itg.data_available) {
145  RATES_COPY(imu.gyro_unscaled, imu_aspirin.gyro_itg.data.rates);
146  imu_aspirin.gyro_itg.data_available = FALSE;
147  imu_aspirin.gyro_valid = TRUE;
148  }
149 
150  /* HMC58XX event task */
151  hmc58xx_event(&imu_aspirin.mag_hmc);
152  if (imu_aspirin.mag_hmc.data_available) {
153 #ifdef IMU_ASPIRIN_VERSION_1_0
154  VECT3_COPY(imu.mag_unscaled, imu_aspirin.mag_hmc.data.vect);
155 #else // aspirin 1.5 with hmc5883
156  imu.mag_unscaled.x = imu_aspirin.mag_hmc.data.vect.y;
157  imu.mag_unscaled.y = -imu_aspirin.mag_hmc.data.vect.x;
158  imu.mag_unscaled.z = imu_aspirin.mag_hmc.data.vect.z;
159 #endif
160  imu_aspirin.mag_hmc.data_available = FALSE;
161  imu_aspirin.mag_valid = TRUE;
162  }
163 }
