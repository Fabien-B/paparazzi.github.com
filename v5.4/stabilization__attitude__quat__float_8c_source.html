141 void stabilization_attitude_init(void)
142 {
143 
144  stabilization_attitude_ref_init();
145 
146  for (int i = 0; i < STABILIZATION_ATTITUDE_GAIN_NB; i++) {
147  VECT3_ASSIGN(stabilization_gains[i].p, phi_pgain[i], theta_pgain[i], psi_pgain[i]);
148  VECT3_ASSIGN(stabilization_gains[i].d, phi_dgain[i], theta_dgain[i], psi_dgain[i]);
149  VECT3_ASSIGN(stabilization_gains[i].i, phi_igain[i], theta_igain[i], psi_igain[i]);
150  VECT3_ASSIGN(stabilization_gains[i].dd, phi_ddgain[i], theta_ddgain[i], psi_ddgain[i]);
151  VECT3_ASSIGN(stabilization_gains[i].rates_d, phi_dgain_d[i], theta_dgain_d[i], psi_dgain_d[i]);
152 #ifdef HAS_SURFACE_COMMANDS
153  VECT3_ASSIGN(stabilization_gains[i].surface_p, phi_pgain_surface[i], theta_pgain_surface[i], psi_pgain_surface[i]);
154  VECT3_ASSIGN(stabilization_gains[i].surface_d, phi_dgain_surface[i], theta_dgain_surface[i], psi_dgain_surface[i]);
155  VECT3_ASSIGN(stabilization_gains[i].surface_i, phi_igain_surface[i], theta_igain_surface[i], psi_igain_surface[i]);
156  VECT3_ASSIGN(stabilization_gains[i].surface_dd, phi_ddgain_surface[i], theta_ddgain_surface[i], psi_ddgain_surface[i]);

286 void stabilization_attitude_run(bool_t enable_integrator)
287 {
288 
289  /*
290  * Update reference
291  */
292  stabilization_attitude_ref_update();
293 
294  /*
295  * Compute errors for feedback
296  */
297 
298  /* attitude error */
299  struct FloatQuat att_err;
300  struct FloatQuat *att_quat = stateGetNedToBodyQuat_f();
301  float_quat_inv_comp(&att_err, att_quat, &stab_att_ref_quat);
302  /* wrap it in the shortest direction */
303  float_quat_wrap_shortest(&att_err);
304 
305  /* rate error */
306  struct FloatRates rate_err;
307  struct FloatRates *body_rate = stateGetBodyRates_f();
308  RATES_DIFF(rate_err, stab_att_ref_rate, *body_rate);
309  /* rate_d error */
310  RATES_DIFF(body_rate_d, *body_rate, last_body_rate);
311  RATES_COPY(last_body_rate, *body_rate);
312 
313 #define INTEGRATOR_BOUND 1.0
314  /* integrated error */
315  if (enable_integrator) {
316  stabilization_att_sum_err_quat.qx += att_err.qx / IERROR_SCALE;
317  stabilization_att_sum_err_quat.qy += att_err.qy / IERROR_SCALE;
318  stabilization_att_sum_err_quat.qz += att_err.qz / IERROR_SCALE;
319  Bound(stabilization_att_sum_err_quat.qx, -INTEGRATOR_BOUND, INTEGRATOR_BOUND);
320  Bound(stabilization_att_sum_err_quat.qy, -INTEGRATOR_BOUND, INTEGRATOR_BOUND);
321  Bound(stabilization_att_sum_err_quat.qz, -INTEGRATOR_BOUND, INTEGRATOR_BOUND);
322  } else {
323  /* reset accumulator */
324  float_quat_identity(&stabilization_att_sum_err_quat);
325  }
326 
327  attitude_run_ff(stabilization_att_ff_cmd, &stabilization_gains[gain_idx], &stab_att_ref_accel);
328 
329  attitude_run_fb(stabilization_att_fb_cmd, &stabilization_gains[gain_idx], &att_err, &rate_err, &body_rate_d,
330  &stabilization_att_sum_err_quat);
331 
332  stabilization_cmd[COMMAND_ROLL] = stabilization_att_fb_cmd[COMMAND_ROLL] + stabilization_att_ff_cmd[COMMAND_ROLL];
333  stabilization_cmd[COMMAND_PITCH] = stabilization_att_fb_cmd[COMMAND_PITCH] + stabilization_att_ff_cmd[COMMAND_PITCH];
334  stabilization_cmd[COMMAND_YAW] = stabilization_att_fb_cmd[COMMAND_YAW] + stabilization_att_ff_cmd[COMMAND_YAW];
335 
336 #ifdef HAS_SURFACE_COMMANDS
337  stabilization_cmd[COMMAND_ROLL_SURFACE] = stabilization_att_fb_cmd[COMMAND_ROLL_SURFACE] +
338  stabilization_att_ff_cmd[COMMAND_ROLL_SURFACE];
339  stabilization_cmd[COMMAND_PITCH_SURFACE] = stabilization_att_fb_cmd[COMMAND_PITCH_SURFACE] +
340  stabilization_att_ff_cmd[COMMAND_PITCH_SURFACE];
341  stabilization_cmd[COMMAND_YAW_SURFACE] = stabilization_att_fb_cmd[COMMAND_YAW_SURFACE] +
342  stabilization_att_ff_cmd[COMMAND_YAW_SURFACE];
343 #endif
344 
345  /* bound the result */
346  BoundAbs(stabilization_cmd[COMMAND_ROLL], MAX_PPRZ);
347  BoundAbs(stabilization_cmd[COMMAND_PITCH], MAX_PPRZ);
348  BoundAbs(stabilization_cmd[COMMAND_YAW], MAX_PPRZ);
349 }
