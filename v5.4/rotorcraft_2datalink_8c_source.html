61 void dl_parse_msg(void)
62 {
63 
64  datalink_time = 0;
65 
66  uint8_t msg_id = IdOfMsg(dl_buffer);
67  switch (msg_id) {
68 
69  case DL_PING: {
70  DOWNLINK_SEND_PONG(DefaultChannel, DefaultDevice);
71  }
72  break;
73 
74  case DL_SETTING : {
75  if (DL_SETTING_ac_id(dl_buffer) != AC_ID) { break; }
76  uint8_t i = DL_SETTING_index(dl_buffer);
77  float var = DL_SETTING_value(dl_buffer);
78  DlSetting(i, var);
79  DOWNLINK_SEND_DL_VALUE(DefaultChannel, DefaultDevice, &i, &var);
80  }
81  break;
82 
83  case DL_GET_SETTING : {
84  if (DL_GET_SETTING_ac_id(dl_buffer) != AC_ID) { break; }
85  uint8_t i = DL_GET_SETTING_index(dl_buffer);
86  float val = settings_get_value(i);
87  DOWNLINK_SEND_DL_VALUE(DefaultChannel, DefaultDevice, &i, &val);
88  }
89  break;
90 
91 #if defined USE_NAVIGATION
92  case DL_BLOCK : {
93  if (DL_BLOCK_ac_id(dl_buffer) != AC_ID) { break; }
94  nav_goto_block(DL_BLOCK_block_id(dl_buffer));
95  }
96  break;
97 
98  case DL_MOVE_WP : {
99  uint8_t ac_id = DL_MOVE_WP_ac_id(dl_buffer);
100  if (ac_id != AC_ID) { break; }
101  if (stateIsLocalCoordinateValid()) {
102  uint8_t wp_id = DL_MOVE_WP_wp_id(dl_buffer);
103  struct LlaCoor_i lla;
104  lla.lat = DL_MOVE_WP_lat(dl_buffer);
105  lla.lon = DL_MOVE_WP_lon(dl_buffer);
106  /* WP_alt from message is alt above MSL in mm
107  * lla.alt is above ellipsoid in mm
108  */
109  lla.alt = DL_MOVE_WP_alt(dl_buffer) - state.ned_origin_i.hmsl +
110  state.ned_origin_i.lla.alt;
111  nav_move_waypoint_lla(wp_id, &lla);
112  }
113  }
114  break;
115 #endif /* USE_NAVIGATION */
116 #ifdef RADIO_CONTROL_TYPE_DATALINK
117  case DL_RC_3CH :
118 #ifdef RADIO_CONTROL_DATALINK_LED
119  LED_TOGGLE(RADIO_CONTROL_DATALINK_LED);
120 #endif
121  parse_rc_3ch_datalink(
122  DL_RC_3CH_throttle_mode(dl_buffer),
123  DL_RC_3CH_roll(dl_buffer),
124  DL_RC_3CH_pitch(dl_buffer));
125  break;
126  case DL_RC_4CH :
127  if (DL_RC_4CH_ac_id(dl_buffer) == AC_ID) {
128 #ifdef RADIO_CONTROL_DATALINK_LED
129  LED_TOGGLE(RADIO_CONTROL_DATALINK_LED);
130 #endif
131  parse_rc_4ch_datalink(DL_RC_4CH_mode(dl_buffer),
132  DL_RC_4CH_throttle(dl_buffer),
133  DL_RC_4CH_roll(dl_buffer),
134  DL_RC_4CH_pitch(dl_buffer),
135  DL_RC_4CH_yaw(dl_buffer));
136  }
137  break;
138 #endif // RADIO_CONTROL_TYPE_DATALINK
139 #if defined GPS_DATALINK
140  case DL_REMOTE_GPS :
141  // Check if the GPS is for this AC
142  if (DL_REMOTE_GPS_ac_id(dl_buffer) != AC_ID) { break; }
143 
144  // Parse the GPS
145  parse_gps_datalink(
146  DL_REMOTE_GPS_numsv(dl_buffer),
147  DL_REMOTE_GPS_ecef_x(dl_buffer),
148  DL_REMOTE_GPS_ecef_y(dl_buffer),
149  DL_REMOTE_GPS_ecef_z(dl_buffer),
150  DL_REMOTE_GPS_lat(dl_buffer),
151  DL_REMOTE_GPS_lon(dl_buffer),
152  DL_REMOTE_GPS_alt(dl_buffer),
153  DL_REMOTE_GPS_hmsl(dl_buffer),
154  DL_REMOTE_GPS_ecef_xd(dl_buffer),
155  DL_REMOTE_GPS_ecef_yd(dl_buffer),
156  DL_REMOTE_GPS_ecef_zd(dl_buffer),
157  DL_REMOTE_GPS_tow(dl_buffer),
158  DL_REMOTE_GPS_course(dl_buffer));
159  break;
160 #endif
161  default:
162  break;
163  }
164  /* Parse modules datalink */
165  modules_parse_datalink(msg_id);
166 }
