86 static void start_message(struct xbee_transport *trans, struct link_device *dev, uint8_t payload_len)
87 {
88  downlink.nb_msgs++;
89  dev->transmit(dev->periph, XBEE_START);
90  const uint16_t len = payload_len + XBEE_API_OVERHEAD;
91  dev->transmit(dev->periph, (len >> 8));
92  dev->transmit(dev->periph, (len & 0xff));
93  trans->cs_tx = 0;
94  const uint8_t header[] = XBEE_TX_HEADER;
95  put_bytes(trans, dev, DL_TYPE_UINT8, DL_FORMAT_SCALAR, XBEE_TX_OVERHEAD + 1, header);
96 }

98 static void end_message(struct xbee_transport *trans, struct link_device *dev)
99 {
100  trans->cs_tx = 0xff - trans->cs_tx;

123 static uint8_t xbee_text_reply_is_ok(void)
124 {
125  char c[2];
126  int count = 0;
127 
128  while (TransportLink(XBEE_UART, ChAvailable())) {
129  char cc = TransportLink(XBEE_UART, Getch());
130  if (count < 2) {
131  c[count] = cc;
132  }
133  count++;
134  }
135 
136  if ((count > 2) && (c[0] == 'O') && (c[1] == 'K')) {
137  return TRUE;
138  }
139 
140  return FALSE;
141 }
