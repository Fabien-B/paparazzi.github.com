84 /*------------- =*= Invariant Observers =*= -------------*
85  *
86  * State vector :
87  *
88  * x = [q0 q1 q2 q3 vx vy vz px py pz wb1 wb2 wb3 hb as]'
89  *
90  * Dynamic model (dim = 15) :
91  *
92  * x_qdot = 0.5 * x_quat * ( x_rates - x_bias );
93  * x_Vdot = A + 1/as (q * am * (q)-1);
94  * x_Xdot = V;
95  * x_bias_dot = 0;
96  * x_asdot = 0;
97  * x_hbdot = 0;
98  *
99  * Observation model (dim = 10):
100  * yv = V;
101  * yx = X;
102  * yh = <X,e3> - hb;
103  * yb = (q)-1 *B * q; (B : magnetometers)
104  *
105  *------------------------------------------------------*/

349 void ahrs_propagate(float dt)
350 {
351  struct FloatVect3 accel;
352  struct FloatRates body_rates;
353 
354  // realign all the filter if needed
355  // a complete init cycle is required
356  if (ins_impl.reset) {
357  ins_impl.reset = FALSE;
358  ins.status = INS_UNINIT;
359  ahrs.status = AHRS_UNINIT;
360  init_invariant_state();
361  }
362 
363  // fill command vector
364  struct Int32Rates gyro_meas_body;
365  struct Int32RMat *body_to_imu_rmat = orientationGetRMat_i(&imu.body_to_imu);
366  int32_rmat_transp_ratemult(&gyro_meas_body, body_to_imu_rmat, &imu.gyro);
367  RATES_FLOAT_OF_BFP(ins_impl.cmd.rates, gyro_meas_body);
368  struct Int32Vect3 accel_meas_body;
369  int32_rmat_transp_vmult(&accel_meas_body, body_to_imu_rmat, &imu.accel);
370  ACCELS_FLOAT_OF_BFP(ins_impl.cmd.accel, accel_meas_body);
371 
372  // update correction gains
373  error_output(&ins_impl);
374 
375  // propagate model
376  struct inv_state new_state;
377  runge_kutta_4_float((float *)&new_state,
378  (float *)&ins_impl.state, INV_STATE_DIM,
379  (float *)&ins_impl.cmd, INV_COMMAND_DIM,
380  invariant_model, dt);
381  ins_impl.state = new_state;
382 
383  // normalize quaternion
384  FLOAT_QUAT_NORMALIZE(ins_impl.state.quat);
385 
386  // set global state
387  stateSetNedToBodyQuat_f(&ins_impl.state.quat);
388  RATES_DIFF(body_rates, ins_impl.cmd.rates, ins_impl.state.bias);
389  stateSetBodyRates_f(&body_rates);
390  stateSetPositionNed_f(&ins_impl.state.pos);
391  stateSetSpeedNed_f(&ins_impl.state.speed);
392  // untilt accel and remove gravity
393  struct FloatQuat q_b2n;
394  float_quat_invert(&q_b2n, &ins_impl.state.quat);
395  float_quat_vmult(&accel, &q_b2n, &ins_impl.cmd.accel);
396  VECT3_SMUL(accel, accel, 1. / (ins_impl.state.as));
397  VECT3_ADD(accel, A);
398  stateSetAccelNed_f((struct NedCoor_f *)&accel);
399

425 #endif
426

504 static void baro_cb(uint8_t __attribute__((unused)) sender_id, const float *pressure)
505 {
506  static float ins_qfe = 101325.0f;
507  static float alpha = 10.0f;
508  static int32_t i = 1;
509  static float baro_moy = 0.0f;
510  static float baro_prev = 0.0f;
511 
512  if (!ins_baro_initialized) {
513  // try to find a stable qfe
514  // TODO generic function in pprz_isa ?
515  if (i == 1) {
516  baro_moy = *pressure;
517  baro_prev = *pressure;
518  }
519  baro_moy = (baro_moy * (i - 1) + *pressure) / i;
520  alpha = (10.*alpha + (baro_moy - baro_prev)) / (11.0f);
521  baro_prev = baro_moy;
522  // test stop condition
523  if (fabs(alpha) < 0.005f) {
524  ins_qfe = baro_moy;
525  ins_baro_initialized = TRUE;
526  }
527  if (i == 250) {
528  ins_qfe = *pressure;
529  ins_baro_initialized = TRUE;
530  }
531  i++;
532  } else { /* normal update with baro measurement */
533  ins_impl.meas.baro_alt = -pprz_isa_height_of_pressure(*pressure, ins_qfe); // Z down
534  }
535 }

584  struct FloatRates rates_unbiased;
585  struct FloatVect3 tmp_vect;
586  struct FloatQuat tmp_quat;
587 
588  // test accel sensitivity
589  if (fabs(s->as) < 0.1) {
590  // too small, return x_dot = 0 to avoid division by 0
591  float_vect_zero(o, n);
592  // TODO set ins state to error
593  return;
594  }
595 
596  /* dot_q = 0.5 * q * (x_rates - x_bias) + LE * q + (1 - ||q||^2) * q */
597  RATES_DIFF(rates_unbiased, c->rates, s->bias);
598  /* qd = 0.5 * q * rates_unbiased = -0.5 * rates_unbiased * q */
599  float_quat_derivative(&s_dot.quat, &rates_unbiased, &(s->quat));
600 
601  float_quat_vmul_right(&tmp_quat, &(s->quat), &ins_impl.corr.LE);
602  QUAT_ADD(s_dot.quat, tmp_quat);
603 
604  float norm2_r = 1. - FLOAT_QUAT_NORM2(s->quat);
605  QUAT_SMUL(tmp_quat, s->quat, norm2_r);
606  QUAT_ADD(s_dot.quat, tmp_quat);
607 
608  /* dot_V = A + (1/as) * (q * am * q-1) + ME */
609  struct FloatQuat q_b2n;
610  float_quat_invert(&q_b2n, &(s->quat));
611  float_quat_vmult((struct FloatVect3 *)&s_dot.speed, &q_b2n, &(c->accel));
612  VECT3_SMUL(s_dot.speed, s_dot.speed, 1. / (s->as));
613  VECT3_ADD(s_dot.speed, A);
614  VECT3_ADD(s_dot.speed, ins_impl.corr.ME);
615 
616  /* dot_X = V + NE */
617  VECT3_SUM(s_dot.pos, s->speed, ins_impl.corr.NE);
618 
619  /* bias_dot = q-1 * (OE) * q */
620  float_quat_vmult(&tmp_vect, &(s->quat), &ins_impl.corr.OE);
621  RATES_ASSIGN(s_dot.bias, tmp_vect.x, tmp_vect.y, tmp_vect.z);
622 
623  /* as_dot = as * RE */
624  s_dot.as = (s->as) * (ins_impl.corr.RE);
625 
626  /* hb_dot = SE */
627  s_dot.hb = ins_impl.corr.SE;
628 
629  // set output
630  memcpy(o, &s_dot, n * sizeof(float));
631 }

637 static inline void error_output(struct InsFloatInv *_ins)
638 {
639 
640  struct FloatVect3 YBt, I, Ev, Eb, Ex, Itemp, Ebtemp, Evtemp;
641  float Eh;
642  float temp;
643 
644  // test accel sensitivity
645  if (fabs(_ins->state.as) < 0.1) {
646  // too small, don't do anything to avoid division by 0
647  return;
648  }
649 
650  /* YBt = q * yB * q-1 */
651  struct FloatQuat q_b2n;
652  float_quat_invert(&q_b2n, &(_ins->state.quat));
653  float_quat_vmult(&YBt, &q_b2n, &(_ins->meas.mag));
654 
655  float_quat_vmult(&I, &q_b2n, &(_ins->cmd.accel));
656  VECT3_SMUL(I, I, 1. / (_ins->state.as));
657 
658  /*--------- E = ( ŷ - y ) ----------*/
659  /* Eb = ( B - YBt ) */
660  VECT3_DIFF(Eb, B, YBt);
661 
662  // pos and speed error only if GPS data are valid
663  // or while waiting first GPS data to prevent diverging
664  if ((gps.fix == GPS_FIX_3D && ins.status == INS_RUNNING
665 #if INS_UPDATE_FW_ESTIMATOR
666  && state.utm_initialized_f
667 #else
668  && state.ned_initialized_f
669 #endif
670  ) || !ins_gps_fix_once) {
671  /* Ev = (V - YV) */
672  VECT3_DIFF(Ev, _ins->state.speed, _ins->meas.speed_gps);
673  /* Ex = (X - YX) */
674  VECT3_DIFF(Ex, _ins->state.pos, _ins->meas.pos_gps);
675  } else {
676  FLOAT_VECT3_ZERO(Ev);
677  FLOAT_VECT3_ZERO(Ex);
678  }
679  /* Eh = < X,e3 > - hb - YH */
680  Eh = _ins->state.pos.z - _ins->state.hb - _ins->meas.baro_alt;
681 
682  /*--------------Gains--------------*/
683 
684  /**** LvEv + LbEb = -lvIa x Ev + lb < B x Eb, Ia > Ia *****/
685  VECT3_SMUL(Itemp, I, -_ins->gains.lv / 100.);
686  VECT3_CROSS_PRODUCT(Evtemp, Itemp, Ev);
687 
688  VECT3_CROSS_PRODUCT(Ebtemp, B, Eb);
689  temp = VECT3_DOT_PRODUCT(Ebtemp, I);
690  temp = (_ins->gains.lb / 100.) * temp;
691 
692  VECT3_SMUL(Ebtemp, I, temp);
693  VECT3_ADD(Evtemp, Ebtemp);
694  VECT3_COPY(_ins->corr.LE, Evtemp);
695 
696  /***** MvEv + MhEh = -mv * Ev + (-mh * <Eh,e3>)********/
697  _ins->corr.ME.x = (-_ins->gains.mv) * Ev.x + 0.;
698  _ins->corr.ME.y = (-_ins->gains.mv) * Ev.y + 0.;
699  _ins->corr.ME.z = ((-_ins->gains.mvz) * Ev.z) + ((-_ins->gains.mh) * Eh);
700 
701  /****** NxEx + NhEh = -nx * Ex + (-nh * <Eh, e3>) ********/
702  _ins->corr.NE.x = (-_ins->gains.nx) * Ex.x + 0.;
703  _ins->corr.NE.y = (-_ins->gains.nx) * Ex.y + 0.;
704  _ins->corr.NE.z = ((-_ins->gains.nxz) * Ex.z) + ((-_ins->gains.nh) * Eh);
705 
706  /****** OvEv + ObEb = ovIa x Ev - ob < B x Eb, Ia > Ia ********/
707  VECT3_SMUL(Itemp, I, _ins->gains.ov / 1000.);
708  VECT3_CROSS_PRODUCT(Evtemp, Itemp, Ev);
709 
710  VECT3_CROSS_PRODUCT(Ebtemp, B, Eb);
711  temp = VECT3_DOT_PRODUCT(Ebtemp, I);
712  temp = (-_ins->gains.ob / 1000.) * temp;
713 
714  VECT3_SMUL(Ebtemp, I, temp);
715  VECT3_ADD(Evtemp, Ebtemp);
716  VECT3_COPY(_ins->corr.OE, Evtemp);
717 
718  /* a scalar */
719  /****** RvEv + RhEh = rv < Ia, Ev > + (-rhEh) **************/
720  _ins->corr.RE = ((_ins->gains.rv / 100.) * VECT3_DOT_PRODUCT(Ev, I)) + ((-_ins->gains.rh / 10000.) * Eh);
721 
722  /****** ShEh ******/
723  _ins->corr.SE = (_ins->gains.sh) * Eh;
724 
725 }
