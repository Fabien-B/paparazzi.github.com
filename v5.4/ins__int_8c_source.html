242 void ins_propagate(float dt)
243 {
244  /* untilt accels */
245  struct Int32Vect3 accel_meas_body;
246  struct Int32RMat *body_to_imu_rmat = orientationGetRMat_i(&imu.body_to_imu);
247  int32_rmat_transp_vmult(&accel_meas_body, body_to_imu_rmat, &imu.accel);
248  struct Int32Vect3 accel_meas_ltp;
249  int32_rmat_transp_vmult(&accel_meas_ltp, stateGetNedToBodyRMat_i(), &accel_meas_body);
250 
251  float z_accel_meas_float = ACCEL_FLOAT_OF_BFP(accel_meas_ltp.z);
252  if (ins_impl.baro_initialized) {
253  vff_propagate(z_accel_meas_float, dt);
254  ins_update_from_vff();
255  } else { // feed accel from the sensors
256  // subtract -9.81m/s2 (acceleration measured due to gravity,
257  // but vehicle not accelerating in ltp)
258  ins_impl.ltp_accel.z = accel_meas_ltp.z + ACCEL_BFP_OF_REAL(9.81);
259  }
260 
261 #if USE_HFF
262  /* propagate horizontal filter */
263  b2_hff_propagate();
264  /* convert and copy result to ins_impl */
265  ins_update_from_hff();
266 #else
267  ins_impl.ltp_accel.x = accel_meas_ltp.x;
268  ins_impl.ltp_accel.y = accel_meas_ltp.y;
269 #endif /* USE_HFF */
270 
271  ins_ned_to_state();
272 }

274 static void baro_cb(uint8_t __attribute__((unused)) sender_id, const float *pressure)
275 {
276  if (!ins_impl.baro_initialized && *pressure > 1e-7) {
277  // wait for a first positive value
278  ins_impl.qfe = *pressure;
279  ins_impl.baro_initialized = TRUE;
280  }
281 
282  if (ins_impl.baro_initialized) {
283  if (ins_impl.vf_reset) {
284  ins_impl.vf_reset = FALSE;
285  ins_impl.qfe = *pressure;
286  vff_realign(0.);
287  ins_update_from_vff();
288  } else {
289  ins_impl.baro_z = -pprz_isa_height_of_pressure(*pressure, ins_impl.qfe);
290 #if USE_VFF_EXTENDED
291  vff_update_baro(ins_impl.baro_z);
292 #else
293  vff_update(ins_impl.baro_z);
294 #endif
295  }
296  ins_ned_to_state();
297  }
298 }

300 #if USE_GPS
301 void ins_update_gps(void)
302 {
303  if (gps.fix == GPS_FIX_3D) {
304  if (!ins_impl.ltp_initialized) {
305  ltp_def_from_ecef_i(&ins_impl.ltp_def, &gps.ecef_pos);
306  ins_impl.ltp_def.lla.alt = gps.lla_pos.alt;
307  ins_impl.ltp_def.hmsl = gps.hmsl;
308  ins_impl.ltp_initialized = TRUE;
309  stateSetLocalOrigin_i(&ins_impl.ltp_def);
310  }
311 
312  struct NedCoor_i gps_pos_cm_ned;
313  ned_of_ecef_point_i(&gps_pos_cm_ned, &ins_impl.ltp_def, &gps.ecef_pos);
315  struct NedCoor_i gps_speed_cm_s_ned;
316  ned_of_ecef_vect_i(&gps_speed_cm_s_ned, &ins_impl.ltp_def, &gps.ecef_vel);
317 
318 #if INS_USE_GPS_ALT
319  vff_update_z_conf((float)gps_pos_cm_ned.z / 100.0, INS_VFF_R_GPS);
320 #endif
321 
322 #if USE_HFF
323  /* horizontal gps transformed to NED in meters as float */
324  struct FloatVect2 gps_pos_m_ned;
325  VECT2_ASSIGN(gps_pos_m_ned, gps_pos_cm_ned.x, gps_pos_cm_ned.y);
326  VECT2_SDIV(gps_pos_m_ned, gps_pos_m_ned, 100.0f);
327 
328  struct FloatVect2 gps_speed_m_s_ned;
329  VECT2_ASSIGN(gps_speed_m_s_ned, gps_speed_cm_s_ned.x, gps_speed_cm_s_ned.y);
330  VECT2_SDIV(gps_speed_m_s_ned, gps_speed_m_s_ned, 100.);
331 
332  if (ins_impl.hf_realign) {
333  ins_impl.hf_realign = FALSE;
334  const struct FloatVect2 zero = {0.0f, 0.0f};
335  b2_hff_realign(gps_pos_m_ned, zero);
336  }
337  // run horizontal filter
338  b2_hff_update_gps(&gps_pos_m_ned, &gps_speed_m_s_ned);
339  // convert and copy result to ins_impl
340  ins_update_from_hff();
341 
342 #else /* hff not used */
343  /* simply copy horizontal pos/speed from gps */
344  INT32_VECT2_SCALE_2(ins_impl.ltp_pos, gps_pos_cm_ned,
345  INT32_POS_OF_CM_NUM, INT32_POS_OF_CM_DEN);
346  INT32_VECT2_SCALE_2(ins_impl.ltp_speed, gps_speed_cm_s_ned,
347  INT32_SPEED_OF_CM_S_NUM, INT32_SPEED_OF_CM_S_DEN);
348 #endif /* USE_HFF */
349 
350  ins_ned_to_state();
351  }
352 }

356 #if USE_SONAR
357 static void sonar_cb(uint8_t __attribute__((unused)) sender_id, const float *distance)
358 {
359  static float last_offset = 0.;
360 
361  /* update filter assuming a flat ground */
362  if (*distance < INS_SONAR_MAX_RANGE && *distance > INS_SONAR_MIN_RANGE
363 #ifdef INS_SONAR_THROTTLE_THRESHOLD
364  && stabilization_cmd[COMMAND_THRUST] < INS_SONAR_THROTTLE_THRESHOLD
365 #endif
366 #ifdef INS_SONAR_BARO_THRESHOLD
367  && ins_impl.baro_z > -INS_SONAR_BARO_THRESHOLD /* z down */
368 #endif
369  && ins_impl.update_on_agl
370  && ins_impl.baro_initialized) {
371  vff_update_z_conf(-(*distance), VFF_R_SONAR_0 + VFF_R_SONAR_OF_M * fabsf(*distance));
372  last_offset = vff.offset;
373  } else {
374  /* update offset with last value to avoid divergence */
375  vff_update_offset(last_offset);
376  }
377 }
378 #endif // USE_SONAR
