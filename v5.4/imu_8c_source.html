209 void imu_SetBodyToImuCurrent(float set)
210 {
211  imu.b2i_set_current = set;
212 
213  if (imu.b2i_set_current) {
214  // adjust imu_to_body roll and pitch by current NedToBody roll and pitch
215  struct FloatEulers body_to_imu_eulers;
216  memcpy(&body_to_imu_eulers, orientationGetEulers_f(&imu.body_to_imu), sizeof(struct FloatEulers));
217  if (stateIsAttitudeValid()) {
218  // adjust imu_to_body roll and pitch by current NedToBody roll and pitch
219  body_to_imu_eulers.phi += stateGetNedToBodyEulers_f()->phi;
220  body_to_imu_eulers.theta += stateGetNedToBodyEulers_f()->theta;
221  orientationSetEulers_f(&imu.body_to_imu, &body_to_imu_eulers);
222 #if USE_IMU_FLOAT
223  orientationSetEulers_f(&imuf.body_to_imu, &body_to_imu_eulers);
224 #endif
225  } else {
226  // indicate that we couldn't set to current roll/pitch
227  imu.b2i_set_current = FALSE;
228  }
229  } else {
230  // reset to BODY_TO_IMU as defined in airframe file
231  struct FloatEulers body_to_imu_eulers =
232  {IMU_BODY_TO_IMU_PHI, IMU_BODY_TO_IMU_THETA, IMU_BODY_TO_IMU_PSI};
233  orientationSetEulers_f(&imu.body_to_imu, &body_to_imu_eulers);
234 #if USE_IMU_FLOAT
235  orientationSetEulers_f(&imuf.body_to_imu, &body_to_imu_eulers);
236 #endif
237  }
238 }

247 void WEAK imu_scale_gyro(struct Imu *_imu)
248 {
249  RATES_COPY(_imu->gyro_prev, _imu->gyro);
250  _imu->gyro.p = ((_imu->gyro_unscaled.p - _imu->gyro_neutral.p) * IMU_GYRO_P_SIGN *
251  IMU_GYRO_P_SENS_NUM) / IMU_GYRO_P_SENS_DEN;
252  _imu->gyro.q = ((_imu->gyro_unscaled.q - _imu->gyro_neutral.q) * IMU_GYRO_Q_SIGN *

271 void WEAK imu_scale_mag(struct Imu *_imu)
272 {
273  struct Int32Vect3 mag_correction;
274  mag_correction.x = (int32_t)(IMU_MAG_X_CURRENT_COEF * (float) electrical.current);
275  mag_correction.y = (int32_t)(IMU_MAG_Y_CURRENT_COEF * (float) electrical.current);
276  mag_correction.z = (int32_t)(IMU_MAG_Z_CURRENT_COEF * (float) electrical.current);
277  _imu->mag.x = (((_imu->mag_unscaled.x - mag_correction.x) - _imu->mag_neutral.x) *
278  IMU_MAG_X_SIGN * IMU_MAG_X_SENS_NUM) / IMU_MAG_X_SENS_DEN;
279  _imu->mag.y = (((_imu->mag_unscaled.y - mag_correction.y) - _imu->mag_neutral.y) *
280  IMU_MAG_Y_SIGN * IMU_MAG_Y_SENS_NUM) / IMU_MAG_Y_SENS_DEN;
281  _imu->mag.z = (((_imu->mag_unscaled.z - mag_correction.z) - _imu->mag_neutral.z) *
282  IMU_MAG_Z_SIGN * IMU_MAG_Z_SENS_NUM) / IMU_MAG_Z_SENS_DEN;
283 }
