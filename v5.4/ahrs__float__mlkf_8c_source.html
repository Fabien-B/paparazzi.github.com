149 static inline void propagate_ref(float dt) {
150  /* converts gyro to floating point */
151  struct FloatRates gyro_float;
152  RATES_FLOAT_OF_BFP(gyro_float, imu.gyro_prev);
153 
154  /* unbias measurement */
155  RATES_SUB(gyro_float, ahrs_impl.gyro_bias);
156 
157 #ifdef AHRS_PROPAGATE_LOW_PASS_RATES
158  /* lowpass angular rates */
159  const float alpha = 0.1;
160  FLOAT_RATES_LIN_CMB(ahrs_impl.imu_rate, ahrs_impl.imu_rate,
161  (1.-alpha), gyro_float, alpha);
162 #else
163  RATES_COPY(ahrs_impl.imu_rate, gyro_float);
164 #endif
165 
166  /* propagate reference quaternion */
167  float_quat_integrate(&ahrs_impl.ltp_to_imu_quat, &ahrs_impl.imu_rate, dt);
168 
169 }

175 static inline void propagate_state(float dt) {
176 
177  /* predict covariance */
178  const float dp = ahrs_impl.imu_rate.p*dt;
179  const float dq = ahrs_impl.imu_rate.q*dt;
180  const float dr = ahrs_impl.imu_rate.r*dt;
181 
182  float F[6][6] = {{ 1., dr, -dq, -dt, 0., 0. },
183  { -dr, 1., dp, 0., -dt, 0. },
184  { dq, -dp, 1., 0., 0., -dt },
185  { 0., 0., 0., 1., 0., 0. },
186  { 0., 0., 0., 0., 1., 0. },
187  { 0., 0., 0., 0., 0., 1. }};
188  // P = FPF' + GQG
189  float tmp[6][6];
190  MAT_MUL(6,6,6, tmp, F, ahrs_impl.P);
191  MAT_MUL_T(6,6,6, ahrs_impl.P, tmp, F);
192  const float dt2 = dt * dt;
193  const float GQG[6] = {dt2*10e-3, dt2*10e-3, dt2*10e-3, dt2*9e-6, dt2*9e-6, dt2*9e-6 };
194  for (int i=0;i<6;i++)
195  ahrs_impl.P[i][i] += GQG[i];
196 
197 }

203 static inline void update_state(const struct FloatVect3 *i_expected, struct FloatVect3* b_measured, struct FloatVect3* noise) {
204 
205  /* converted expected measurement from inertial to body frame */
206  struct FloatVect3 b_expected;
207  float_quat_vmult(&b_expected, &ahrs_impl.ltp_to_imu_quat, (struct FloatVect3*)i_expected);
208 
209  // S = HPH' + JRJ
210  float H[3][6] = {{ 0., -b_expected.z, b_expected.y, 0., 0., 0.},
211  { b_expected.z, 0., -b_expected.x, 0., 0., 0.},
212  {-b_expected.y, b_expected.x, 0., 0., 0., 0.}};
213  float tmp[3][6];
214  MAT_MUL(3,6,6, tmp, H, ahrs_impl.P);
215  float S[3][3];
216  MAT_MUL_T(3,6,3, S, tmp, H);
217 
218  /* add the measurement noise */
219  S[0][0] += noise->x;
220  S[1][1] += noise->y;
221  S[2][2] += noise->z;
222 
223  float invS[3][3];
224  MAT_INV33(invS, S);
225 
226  // K = PH'invS
227  float tmp2[6][3];
228  MAT_MUL_T(6,6,3, tmp2, ahrs_impl.P, H);
229  float K[6][3];
230  MAT_MUL(6,3,3, K, tmp2, invS);
231 
232  // P = (I-KH)P
233  float tmp3[6][6];
234  MAT_MUL(6,3,6, tmp3, K, H);
235  float I6[6][6] = {{ 1., 0., 0., 0., 0., 0. },
236  { 0., 1., 0., 0., 0., 0. },
237  { 0., 0., 1., 0., 0., 0. },
238  { 0., 0., 0., 1., 0., 0. },
239  { 0., 0., 0., 0., 1., 0. },
240  { 0., 0., 0., 0., 0., 1. }};
241  float tmp4[6][6];
242  MAT_SUB(6,6, tmp4, I6, tmp3);
243  float tmp5[6][6];
244  MAT_MUL(6,6,6, tmp5, tmp4, ahrs_impl.P);
245  memcpy(ahrs_impl.P, tmp5, sizeof(ahrs_impl.P));
246 
247  // X = X + Ke
248  struct FloatVect3 e;
249  VECT3_DIFF(e, *b_measured, b_expected);
250  ahrs_impl.gibbs_cor.qx += K[0][0]*e.x + K[0][1]*e.y + K[0][2]*e.z;
251  ahrs_impl.gibbs_cor.qy += K[1][0]*e.x + K[1][1]*e.y + K[1][2]*e.z;
252  ahrs_impl.gibbs_cor.qz += K[2][0]*e.x + K[2][1]*e.y + K[2][2]*e.z;
253  ahrs_impl.gyro_bias.p += K[3][0]*e.x + K[3][1]*e.y + K[3][2]*e.z;
254  ahrs_impl.gyro_bias.q += K[4][0]*e.x + K[4][1]*e.y + K[4][2]*e.z;
255  ahrs_impl.gyro_bias.r += K[5][0]*e.x + K[5][1]*e.y + K[5][2]*e.z;
256 
257 }
