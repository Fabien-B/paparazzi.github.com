98 static void pressure_abs_cb(uint8_t __attribute__((unused)) sender_id, const float *pressure)
99 {
100  air_data.pressure = *pressure;
101 
102  // calculate QNH from pressure and absolute alitude if that is available
103  if (air_data.calc_qnh_once && stateIsGlobalCoordinateValid()) {
104  float h = stateGetPositionLla_f()->alt;
105  air_data.qnh = pprz_isa_ref_pressure_of_height_full(air_data.pressure, h) / 100.f;
106  air_data.calc_qnh_once = FALSE;
107  }
108 
109  if (air_data.calc_amsl_baro && air_data.qnh > 0) {
110  air_data.amsl_baro = pprz_isa_height_of_pressure_full(air_data.pressure,
111  air_data.qnh * 100.f);
112  air_data.amsl_baro_valid = TRUE;
113  }
114 
115  /* reset baro health counter */
116  baro_health_counter = 10;
117 }

119 static void pressure_diff_cb(uint8_t __attribute__((unused)) sender_id, const float *pressure)
120 {
121  air_data.differential = *pressure;

123  air_data.airspeed = tas_from_dynamic_pressure(air_data.differential);

168 void air_data_init(void)
169 {
170  air_data.calc_airspeed = AIR_DATA_CALC_AIRSPEED;
171  air_data.calc_tas_factor = AIR_DATA_CALC_TAS_FACTOR;
172  air_data.calc_amsl_baro = AIR_DATA_CALC_AMSL_BARO;
173  air_data.tas_factor = AIR_DATA_TAS_FACTOR;
174  air_data.calc_qnh_once = TRUE;
175  air_data.amsl_baro_valid = FALSE;
176 
177  /* initialize the output variables
178  * pressure, qnh, temperature and airspeed to invalid values,
179  * rest to zero
180  */
181  air_data.pressure = -1.0f;
182  air_data.qnh = -1.0f;
183  air_data.airspeed = -1.0f;
184  air_data.temperature = -1000.0f;
185  air_data.differential = 0.0f;
186  air_data.amsl_baro = 0.0f;
187  air_data.aoa = 0.0f;
188  air_data.sideslip = 0.0f;
189  air_data.wind_speed = 0.0f;
190  air_data.wind_dir = 0.0f;

192  /* internal variables */
193  baro_health_counter = 0;
194 
195  AbiBindMsgBARO_ABS(AIR_DATA_BARO_ABS_ID, &pressure_abs_ev, pressure_abs_cb);
196  AbiBindMsgBARO_DIFF(AIR_DATA_BARO_DIFF_ID, &pressure_diff_ev, pressure_diff_cb);

240 float eas_from_dynamic_pressure(float q)
241 {
242  /* q (dynamic pressure) = total pressure - static pressure
243  * q = 1/2*rho*speed^2
244  * speed = sqrt(2*q/rho)
245  * With rho = air density at sea level.
246  * Lower bound of q at zero, no flying backwards guys...
247  */
248  const float two_div_rho_0 = 2.0 / PPRZ_ISA_AIR_DENSITY;
249  return sqrtf(Max(q * two_div_rho_0, 0));
250 }

268 float get_tas_factor(float p, float t)
269 {
270  /* factor to convert EAS to TAS:
271  * sqrt(rho0 / rho) = sqrt((p0 * T) / (p * T0))
272  * convert input temp to Kelvin
273  */
274  return sqrtf((PPRZ_ISA_SEA_LEVEL_PRESSURE * (t + 274.15)) /
275  (p * PPRZ_ISA_SEA_LEVEL_TEMP));
276 }
