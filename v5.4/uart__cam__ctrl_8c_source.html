71 void digital_cam_uart_event(void)
72 {
73  while (CameraLink(ChAvailable())) {
74  parse_mora(&mora_protocol, CameraLink(Getch()));
75  if (mora_protocol.msg_received) {
76  switch (mora_protocol.msg_id) {
77  case MORA_STATUS:
78  for (int i = 0; i < MORA_STATUS_MSG_SIZE; i++) {
79  mora_status_msg.bin[i] = mora_protocol.payload[i];
80  }
81  digital_cam_uart_status = mora_status_msg.data.shots;
82  break;
83  case MORA_PAYLOAD:
84  for (int i = 0; i < MORA_PAYLOAD_MSG_SIZE; i++) {
85  thumbs[thumb_pointer][i] = mora_protocol.payload[i];
86  }
87  break;
88  default:
89  break;
90  }
91  mora_protocol.msg_received = 0;
92  }
93  }
94 }

97 static void send_thumbnails(struct transport_tx *trans, struct link_device *dev)
98 {
99  static int cnt = 0;
100  if (digital_cam_uart_thumbnails > 0) {

102  cnt++;

104  cnt = 0;

107  }
108  pprz_msg_send_PAYLOAD(trans, dev, AC_ID, THUMB_MSG_SIZE, thumbs[thumb_pointer]);
109 
110  // Update the write/read pointer: if we receive a new thumb part, that will be sent, otherwise the oldest infor is repeated
111  thumb_pointer++;
112  if (thumb_pointer >= THUMB_COUNT) {
113  thumb_pointer = 0;
114  }
115 
116  MoraHeader(MORA_BUFFER_EMPTY, 0);

161  dc_shot_msg.data.vground = *stateGetHorizontalSpeedNorm_i();
162  dc_shot_msg.data.course = *stateGetHorizontalSpeedDir_i();
163  dc_shot_msg.data.groundalt = POS_BFP_OF_REAL(state.alt_agl_f);
164 
165  MoraHeader(MORA_SHOOT, MORA_SHOOT_MSG_SIZE);
166  for (int i = 0; i < (MORA_SHOOT_MSG_SIZE); i++) {
167  MoraPutUint8(dc_shot_msg.bin[i]);
168  }
169  MoraTrailer();
170  dc_send_shot_position();
171  break;
172  case DC_TALLER:
173  break;
174  case DC_WIDER:
175  break;
176  case DC_ON:
177  break;
178  case DC_OFF:
179  break;
180  default:
181  break;
