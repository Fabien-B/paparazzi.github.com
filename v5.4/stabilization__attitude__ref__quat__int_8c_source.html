177 #define DT_UPDATE (1./PERIODIC_FREQUENCY)
178 // CAUTION! Periodic frequency is assumed to be 512 Hz
179 // which is equal to >> 9
180 #define F_UPDATE_RES 9

182 #define OFFSET_AND_ROUND(_a, _b) (((_a)+(1<<((_b)-1)))>>(_b))

185 void stabilization_attitude_ref_update(void)
186 {
187 
188  /* integrate reference attitude */
189  const struct Int32Rates rate_ref_scaled = {
190  OFFSET_AND_ROUND(stab_att_ref_rate.p, (REF_RATE_FRAC - INT32_RATE_FRAC)),
191  OFFSET_AND_ROUND(stab_att_ref_rate.q, (REF_RATE_FRAC - INT32_RATE_FRAC)),
192  OFFSET_AND_ROUND(stab_att_ref_rate.r, (REF_RATE_FRAC - INT32_RATE_FRAC))
193  };
194  struct Int32Quat qdot;
195  int32_quat_derivative(&qdot, &rate_ref_scaled, &stab_att_ref_quat);
196  //QUAT_SMUL(qdot, qdot, DT_UPDATE);
197  qdot.qi = qdot.qi >> F_UPDATE_RES;
198  qdot.qx = qdot.qx >> F_UPDATE_RES;
199  qdot.qy = qdot.qy >> F_UPDATE_RES;
200  qdot.qz = qdot.qz >> F_UPDATE_RES;
201  QUAT_ADD(stab_att_ref_quat, qdot);
202  int32_quat_normalize(&stab_att_ref_quat);
203 
204  /* integrate reference rotational speeds
205  * delta rate = ref_accel * dt
206  * ref_rate = old_ref_rate + delta_rate
207  */
208  const struct Int32Rates delta_rate = {
209  stab_att_ref_accel.p >> (F_UPDATE_RES + REF_ACCEL_FRAC - REF_RATE_FRAC),
210  stab_att_ref_accel.q >> (F_UPDATE_RES + REF_ACCEL_FRAC - REF_RATE_FRAC),
211  stab_att_ref_accel.r >> (F_UPDATE_RES + REF_ACCEL_FRAC - REF_RATE_FRAC)
212  };
213  RATES_ADD(stab_att_ref_rate, delta_rate);
214 
215  /* compute reference angular accelerations */
216  struct Int32Quat err;
217  /* compute reference attitude error */
218  int32_quat_inv_comp(&err, &stab_att_sp_quat, &stab_att_ref_quat);
219  /* wrap it in the shortest direction */
220  int32_quat_wrap_shortest(&err);
221 
222  /* propagate the 2nd order linear model : accel = -2*zeta*omega * rate - omega^2 * angle */
223 
224  const struct Int32Rates accel_rate = {
225  (-two_zeta_omega.p * (stab_att_ref_rate.p >> (REF_RATE_FRAC - REF_ACCEL_FRAC))) >> (TWO_ZETA_OMEGA_RES),
226  (-two_zeta_omega.q * (stab_att_ref_rate.q >> (REF_RATE_FRAC - REF_ACCEL_FRAC))) >> (TWO_ZETA_OMEGA_RES),
227  (-two_zeta_omega.r * (stab_att_ref_rate.r >> (REF_RATE_FRAC - REF_ACCEL_FRAC))) >> (TWO_ZETA_OMEGA_RES)
228  };
229 
230  /* since error quaternion contains the half-angles we get 2*omega^2*err */
231  const struct Int32Rates accel_angle = {
232  (-two_omega_2.p * (err.qx >> (INT32_QUAT_FRAC - REF_ACCEL_FRAC))) >> (TWO_OMEGA_2_RES),
233  (-two_omega_2.q * (err.qy >> (INT32_QUAT_FRAC - REF_ACCEL_FRAC))) >> (TWO_OMEGA_2_RES),
234  (-two_omega_2.r * (err.qz >> (INT32_QUAT_FRAC - REF_ACCEL_FRAC))) >> (TWO_OMEGA_2_RES)
235  };
236 
237  RATES_SUM(stab_att_ref_accel, accel_rate, accel_angle);
238 
239 
240  /* saturate acceleration */
241  const struct Int32Rates MIN_ACCEL = { -REF_ACCEL_MAX_P, -REF_ACCEL_MAX_Q, -REF_ACCEL_MAX_R };
242  const struct Int32Rates MAX_ACCEL = { REF_ACCEL_MAX_P, REF_ACCEL_MAX_Q, REF_ACCEL_MAX_R };
243  RATES_BOUND_BOX(stab_att_ref_accel, MIN_ACCEL, MAX_ACCEL);
244 
245  /* saturate angular speed and trim accel accordingly */
246  SATURATE_SPEED_TRIM_ACCEL();
247 
248 
249  /* compute ref_euler for debugging and telemetry */
250  struct Int32Eulers ref_eul;
251  int32_eulers_of_quat(&ref_eul, &stab_att_ref_quat);
252  INT32_EULERS_LSHIFT(stab_att_ref_euler, ref_eul, (REF_ANGLE_FRAC - INT32_ANGLE_FRAC));
253 }
