221 inline char *get_dev_unique_id(char *s)
222 {
223 #if defined STM32F4
224  volatile uint8_t *unique_id = (volatile uint8_t *)0x1FFF7A10;
225 #else
226  volatile uint8_t *unique_id = (volatile uint8_t *)0x1FFFF7E8;
227 #endif
228  int i;
229 
230  // Fetch serial number from chip's unique ID
231  for (i = 0; i < 24; i += 2) {
232  s[i] = ((*unique_id >> 4) & 0xF) + '0';
233  s[i + 1] = (*unique_id++ & 0xF) + '0';
234  }
235  for (i = 0; i < 24; i++)
236  if (s[i] > '9') {
237  s[i] += 'A' - '9' - 1;
238  }
239  // add termination character
240  s[24] = '\0';
241  return s;
242 }

254 static int cdcacm_control_request(usbd_device *usbd_dev, struct usb_setup_data *req, uint8_t **buf,
255  uint16_t *len, void (**complete)(usbd_device *usbd_dev, struct usb_setup_data *req))
256 {
257  (void)complete;
258  (void)buf;
259  (void)usbd_dev;
260 
261  switch (req->bRequest) {
262  case USB_CDC_REQ_SET_CONTROL_LINE_STATE: {
263  /*
264  * This Linux cdc_acm driver requires this to be implemented
265  * even though it's optional in the CDC spec, and we don't
266  * advertise it in the ACM functional descriptor.
267  */
268  char local_buf[10];
269  struct usb_cdc_notification *notif = (void *)local_buf;
270 
271  /* We echo signals back to host as notification. */
272  notif->bmRequestType = 0xA1;
273  notif->bNotification = USB_CDC_NOTIFY_SERIAL_STATE;
274  notif->wValue = 0;
275  notif->wIndex = 0;
276  notif->wLength = 2;
277  local_buf[8] = req->wValue & 3;
278  local_buf[9] = 0;
279  usbd_ep_write_packet(usbd_dev, 0x83, local_buf, 10);
280  return 1;
281  }
282  case USB_CDC_REQ_SET_LINE_CODING:
283  if (*len < sizeof(struct usb_cdc_line_coding)) {
284  return 0;
285  }
286 
287  return 1;
288  default:
289  return 0;
290  }
291 }

352 bool_t fifo_put(fifo_t *fifo, uint8_t c)
353 {
354  int next;
355 
356  // check if FIFO has room
357  next = (fifo->head + 1) % VCOM_FIFO_SIZE;
358  if (next == fifo->tail) {
359  // full
360  return FALSE;
361  }
362 
363  fifo->buf[fifo->head] = c;
364  fifo->head = next;
365 
366  return TRUE;
367 }
368 
369 
370 bool_t fifo_get(fifo_t *fifo, uint8_t *pc)
371 {
372  int next;
373 
374  // check if FIFO has data
375  if (fifo->head == fifo->tail) {
376  return FALSE;
377  }
378 
379  next = (fifo->tail + 1) % VCOM_FIFO_SIZE;
380 
381  *pc = fifo->buf[fifo->tail];
382  fifo->tail = next;
383 
384  return TRUE;
385 }

409 int VCOM_putchar(int c)
410 {
411  if (usb_connected) {
412  // check if there are at least two more bytes left in queue
413  if (VCOM_check_free_space(2)) {
414  // if yes, add char
415  fifo_put(&txfifo, c);
416  } else {
417  // less than 2 bytes available, add byte and send data now
418  fifo_put(&txfifo, c);
419  VCOM_send_message();
420  }
421  return c;
422  }
423  return -1;
424 }

470 {
471  if (usb_connected) {
472  uint8_t buf[MAX_PACKET_SIZE];
