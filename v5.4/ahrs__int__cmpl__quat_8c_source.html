231 void ahrs_propagate(float dt) {
232  int32_t freq = (int32_t)(1./dt);
233 
234  /* unbias gyro */
235  struct Int32Rates omega;
236  RATES_DIFF(omega, imu.gyro_prev, ahrs_impl.gyro_bias);
237 
238  /* low pass rate */
239 #ifdef AHRS_PROPAGATE_LOW_PASS_RATES
240  RATES_SMUL(ahrs_impl.imu_rate, ahrs_impl.imu_rate,2);
241  RATES_ADD(ahrs_impl.imu_rate, omega);
242  RATES_SDIV(ahrs_impl.imu_rate, ahrs_impl.imu_rate, 3);
243 #else
244  RATES_COPY(ahrs_impl.imu_rate, omega);
245 #endif
246 
247  /* add correction */
248  RATES_ADD(omega, ahrs_impl.rate_correction);
249  /* and zeros it */
250  INT_RATES_ZERO(ahrs_impl.rate_correction);
251 
252  /* integrate quaternion */
253  int32_quat_integrate_fi(&ahrs_impl.ltp_to_imu_quat, &ahrs_impl.high_rez_quat,
254  &omega, freq);
255  int32_quat_normalize(&ahrs_impl.ltp_to_imu_quat);
256 
257  set_body_state_from_quat();
258 
259  // increase accel and mag propagation counters
260  ahrs_impl.accel_cnt++;
261  ahrs_impl.mag_cnt++;
262 }

265 void ahrs_set_accel_gains(void) {
266  /* Complementary filter proportionnal gain (without frequency correction)
267  * Kp = 2 * omega * zeta
268  *
269  * accel_inv_kp = 1 / (Kp * FRAC_conversion / cross_product_gain)
270  * accel_inv_kp = 4096 * 9.81 / Kp
271  */
272  ahrs_impl.accel_inv_kp = 4096 * 9.81 /
273  (2 * ahrs_impl.accel_omega * ahrs_impl.accel_zeta);
274 
275  /* Complementary filter integral gain
276  * Ki = omega^2
277  *
278  * accel_inv_ki = 2^5 / (Ki * FRAC_conversion / cross_product_gain)
279  * accel_inv_ki = 2^5 / 2^16 * 9.81 / Ki
280  * accel_inv_ki = 9.81 / 2048 / Ki
281  */
282  ahrs_impl.accel_inv_ki = 9.81 / 2048 / (ahrs_impl.accel_omega * ahrs_impl.accel_omega);
283 }

285 void ahrs_update_accel(float dt) {
286  // check if we had at least one propagation since last update
287  if (ahrs_impl.accel_cnt == 0)
288  return;
289 
290  // c2 = ltp z-axis in imu-frame
291  struct Int32RMat ltp_to_imu_rmat;
292  int32_rmat_of_quat(&ltp_to_imu_rmat, &ahrs_impl.ltp_to_imu_quat);
293  struct Int32Vect3 c2 = { RMAT_ELMT(ltp_to_imu_rmat, 0,2),
294  RMAT_ELMT(ltp_to_imu_rmat, 1,2),
295  RMAT_ELMT(ltp_to_imu_rmat, 2,2)};
296  struct Int32Vect3 residual;
297 
298  struct Int32Vect3 pseudo_gravity_measurement;
299 
300  if (ahrs_impl.correct_gravity && ahrs_impl.ltp_vel_norm_valid) {
301  /*
302  * centrifugal acceleration in body frame
303  * a_c_body = omega x (omega x r)
304  * (omega x r) = tangential velocity in body frame
305  * a_c_body = omega x vel_tangential_body
306  * assumption: tangential velocity only along body x-axis
307  */
308 
309  // FIXME: check overflows !
310 #define COMPUTATION_FRAC 16
311 #define ACC_FROM_CROSS_FRAC INT32_RATE_FRAC + INT32_SPEED_FRAC - INT32_ACCEL_FRAC - COMPUTATION_FRAC
312 
313  const struct Int32Vect3 vel_tangential_body =
314  {ahrs_impl.ltp_vel_norm >> COMPUTATION_FRAC, 0, 0};
315  struct Int32Vect3 acc_c_body;
316  VECT3_RATES_CROSS_VECT3(acc_c_body, (*stateGetBodyRates_i()), vel_tangential_body);
317  INT32_VECT3_RSHIFT(acc_c_body, acc_c_body, ACC_FROM_CROSS_FRAC);
318 
319  /* convert centrifucal acceleration from body to imu frame */
320  struct Int32Vect3 acc_c_imu;
321  struct Int32RMat *body_to_imu_rmat = orientationGetRMat_i(&imu.body_to_imu);
322  int32_rmat_vmult(&acc_c_imu, body_to_imu_rmat, &acc_c_body);
323 
324  /* and subtract it from imu measurement to get a corrected measurement
325  * of the gravity vector */
326  VECT3_DIFF(pseudo_gravity_measurement, imu.accel, acc_c_imu);
327  } else {
328  VECT3_COPY(pseudo_gravity_measurement, imu.accel);
329  }
330 
331  /* compute the residual of the pseudo gravity vector in imu frame */
332  VECT3_CROSS_PRODUCT(residual, pseudo_gravity_measurement, c2);
333 
334 
335  /* FIR filtered pseudo_gravity_measurement */
336  #define FIR_FILTER_SIZE 8
337  static struct Int32Vect3 filtered_gravity_measurement = {0, 0, 0};
338  VECT3_SMUL(filtered_gravity_measurement, filtered_gravity_measurement, FIR_FILTER_SIZE-1);
339  VECT3_ADD(filtered_gravity_measurement, pseudo_gravity_measurement);
340  VECT3_SDIV(filtered_gravity_measurement, filtered_gravity_measurement, FIR_FILTER_SIZE);
341 
342 
343  if (ahrs_impl.gravity_heuristic_factor) {
344  /* heuristic on acceleration (gravity estimate) norm */
345  /* Factor how strongly to change the weight.
346  * e.g. for gravity_heuristic_factor 30:
347  * <0.66G = 0, 1G = 1.0, >1.33G = 0
348  */
349 
350  struct FloatVect3 g_meas_f;
351  ACCELS_FLOAT_OF_BFP(g_meas_f, filtered_gravity_measurement);
352  const float g_meas_norm = FLOAT_VECT3_NORM(g_meas_f)/9.81;
353  ahrs_impl.weight = 1.0 - ahrs_impl.gravity_heuristic_factor * fabs(1.0 - g_meas_norm) / 10;
354  Bound(ahrs_impl.weight, 0.15, 1.0);
355  }
356  else {
357  ahrs_impl.weight = 1.0;
358  }
359 
360  /* Complementary filter proportional gain.
361  * Kp = 2 * zeta * omega
362  * final Kp with frequency correction = Kp * ahrs_impl.accel_cnt
363  * with ahrs_impl.accel_cnt beeing the number of propagations since last update
364  *
365  * residual FRAC : ACCEL_FRAC + TRIG_FRAC = 10 + 14 = 24
366  * rate_correction FRAC: RATE_FRAC = 12
367  * FRAC_conversion: 2^12 / 2^24 = 1 / 4096
368  * cross_product_gain : 9.81 m/s2
369  *
370  * accel_inv_kp = 1 / (Kp * FRAC_conversion / cross_product_gain)
371  * accel_inv_kp = 4096 * 9.81 / Kp
372  *
373  * inv_rate_scale = 1 / (weight * Kp * FRAC_conversion / cross_product_gain)
374  * inv_rate_scale = 1 / Kp / weight
375  * inv_rate_scale = accel_inv_kp / accel_cnt / weight
376  */
377  int32_t inv_rate_scale = (int32_t)(ahrs_impl.accel_inv_kp / ahrs_impl.accel_cnt
378  / ahrs_impl.weight);
379  Bound(inv_rate_scale, 8192, 4194304);
380 
381  ahrs_impl.rate_correction.p -= residual.x / inv_rate_scale;
382  ahrs_impl.rate_correction.q -= residual.y / inv_rate_scale;
383  ahrs_impl.rate_correction.r -= residual.z / inv_rate_scale;
384 
385  // reset accel propagation counter
386  ahrs_impl.accel_cnt = 0;
387 
388  /* Complementary filter integral gain
389  * Correct the gyro bias.
390  * Ki = omega^2 * dt
391  *
392  * residual FRAC = ACCEL_FRAC + TRIG_FRAC = 10 + 14 = 24
393  * high_rez_bias = RATE_FRAC+28 = 40
394  * FRAC_conversion: 2^40 / 2^24 = 2^16
395  * cross_product_gain : 9.81 m/s2
396  *
397  * accel_inv_ki = 2^5 / (Ki * FRAC_conversion / cross_product_gain)
398  * accel_inv_ki = 2^5 / 2^16 * 9.81 * Ki = 9.81 / 2^11 * Ki
399  *
400  * inv_bias_gain = 2^5 / (weight^2 * Ki * FRAC_conversion / cross_product_gain)
401  * inv_bias_gain = accel_inv_ki / weight^2
402  */
403 
404  int32_t inv_bias_gain = (int32_t)(ahrs_impl.accel_inv_ki /
405  (dt * ahrs_impl.weight * ahrs_impl.weight));
406  Bound(inv_bias_gain, 8, 65536)
407  ahrs_impl.high_rez_bias.p += (residual.x / inv_bias_gain) << 5;
408  ahrs_impl.high_rez_bias.q += (residual.y / inv_bias_gain) << 5;
409  ahrs_impl.high_rez_bias.r += (residual.z / inv_bias_gain) << 5;
410 
411  INT_RATES_RSHIFT(ahrs_impl.gyro_bias, ahrs_impl.high_rez_bias, 28);
412 
413 }

431  ahrs_impl.mag_kp = 2 * ahrs_impl.mag_zeta * ahrs_impl.mag_omega;

437 static inline void ahrs_update_mag_full(float dt) {
438 
439  struct Int32RMat ltp_to_imu_rmat;
440  int32_rmat_of_quat(&ltp_to_imu_rmat, &ahrs_impl.ltp_to_imu_quat);
441 
442  struct Int32Vect3 expected_imu;
443  int32_rmat_vmult(&expected_imu, &ltp_to_imu_rmat, &ahrs_impl.mag_h);
444 
445  struct Int32Vect3 residual;
446  VECT3_CROSS_PRODUCT(residual, imu.mag, expected_imu);
447 
448  /* Complementary filter proportionnal gain.
449  * Kp = 2 * mag_zeta * mag_omega
450  * final Kp with frequency correction = Kp * ahrs_impl.mag_cnt
451  * with ahrs_impl.mag_cnt beeing the number of propagations since last update
452  *
453  * residual FRAC: 2 * MAG_FRAC = 22
454  * rate_correction FRAC: RATE_FRAC = 12
455  * FRAC conversion: 2^12 / 2^22 = 1/1024
456  *
457  * inv_rate_gain = 1 / Kp / FRAC_conversion
458  * inv_rate_gain = 1024 / Kp
459  */
460 
461  const int32_t inv_rate_gain = (int32_t)(1024.0 / (ahrs_impl.mag_kp * ahrs_impl.mag_cnt));
462 
463  ahrs_impl.rate_correction.p += residual.x / inv_rate_gain;
464  ahrs_impl.rate_correction.q += residual.y / inv_rate_gain;
465  ahrs_impl.rate_correction.r += residual.z / inv_rate_gain;
466 
467  /* Complementary filter integral gain
468  * Correct the gyro bias.
469  * Ki = omega^2 * dt
470  *
471  * residual FRAC: 2* MAG_FRAC = 22
472  * high_rez_bias FRAC: RATE_FRAC+28 = 40
473  * FRAC conversion: 2^40 / 2^22 = 2^18
474  *
475  * bias_gain = Ki * FRAC_conversion = Ki * 2^18
476  */
477  const int32_t bias_gain = (int32_t)(ahrs_impl.mag_ki * dt * (1<<18));
478 
479  ahrs_impl.high_rez_bias.p -= residual.x * bias_gain;
480  ahrs_impl.high_rez_bias.q -= residual.y * bias_gain;
481  ahrs_impl.high_rez_bias.r -= residual.z * bias_gain;
482 
483 
484  INT_RATES_RSHIFT(ahrs_impl.gyro_bias, ahrs_impl.high_rez_bias, 28);
485 
486 }

489 static inline void ahrs_update_mag_2d(float dt) {
490 
491  struct Int32Vect2 expected_ltp = {ahrs_impl.mag_h.x, ahrs_impl.mag_h.y};
492  /* normalize expected ltp in 2D (x,y) */
493  int32_vect2_normalize(&expected_ltp, INT32_MAG_FRAC);
494 
495  struct Int32RMat ltp_to_imu_rmat;
496  int32_rmat_of_quat(&ltp_to_imu_rmat, &ahrs_impl.ltp_to_imu_quat);
497 
498  struct Int32Vect3 measured_ltp;
499  int32_rmat_transp_vmult(&measured_ltp, &ltp_to_imu_rmat, &imu.mag);
500 
501  /* normalize measured ltp in 2D (x,y) */
502  struct Int32Vect2 measured_ltp_2d = {measured_ltp.x, measured_ltp.y};
503  int32_vect2_normalize(&measured_ltp_2d, INT32_MAG_FRAC);
504 
505  /* residual_ltp FRAC: 2 * MAG_FRAC - 5 = 17 */
506  struct Int32Vect3 residual_ltp =
507  { 0,
508  0,
509  (measured_ltp_2d.x * expected_ltp.y - measured_ltp_2d.y * expected_ltp.x)/(1<<5)};
510 
511 
512  struct Int32Vect3 residual_imu;
513  int32_rmat_vmult(&residual_imu, &ltp_to_imu_rmat, &residual_ltp);
514 
515  /* Complementary filter proportionnal gain.
516  * Kp = 2 * mag_zeta * mag_omega
517  * final Kp with frequency correction = Kp * ahrs_impl.mag_cnt
518  * with ahrs_impl.mag_cnt beeing the number of propagations since last update
519  *
520  * residual_imu FRAC = residual_ltp FRAC = 17
521  * rate_correction FRAC: RATE_FRAC = 12
522  * FRAC conversion: 2^12 / 2^17 = 1/32
523  *
524  * inv_rate_gain = 1 / Kp / FRAC_conversion
525  * inv_rate_gain = 32 / Kp
526  */
527  int32_t inv_rate_gain = (int32_t)(32.0 / (ahrs_impl.mag_kp * ahrs_impl.mag_cnt));
528 
529  ahrs_impl.rate_correction.p += (residual_imu.x / inv_rate_gain);
530  ahrs_impl.rate_correction.q += (residual_imu.y / inv_rate_gain);
531  ahrs_impl.rate_correction.r += (residual_imu.z / inv_rate_gain);
532 
533  /* Complementary filter integral gain
534  * Correct the gyro bias.
535  * Ki = omega^2 * dt
536  *
537  * residual_imu FRAC = residual_ltp FRAC = 17
538  * high_rez_bias FRAC: RATE_FRAC+28 = 40
539  * FRAC conversion: 2^40 / 2^17 = 2^23
540  *
541  * bias_gain = Ki * FRAC_conversion = Ki * 2^23
542  */
543  int32_t bias_gain = (int32_t)(ahrs_impl.mag_ki * dt * (1 << 23));
544 
545  ahrs_impl.high_rez_bias.p -= (residual_imu.x * bias_gain);
546  ahrs_impl.high_rez_bias.q -= (residual_imu.y * bias_gain);
547  ahrs_impl.high_rez_bias.r -= (residual_imu.z * bias_gain);
548 
549  INT_RATES_RSHIFT(ahrs_impl.gyro_bias, ahrs_impl.high_rez_bias, 28);
550 
551 }

587 void ahrs_update_heading(int32_t heading) {
588 
589  INT32_ANGLE_NORMALIZE(heading);
590 
591  // row 0 of ltp_to_body_rmat = body x-axis in ltp frame
592  // we only consider x and y
593  struct Int32RMat* ltp_to_body_rmat = stateGetNedToBodyRMat_i();
594  struct Int32Vect2 expected_ltp =
595  { RMAT_ELMT((*ltp_to_body_rmat), 0, 0),
596  RMAT_ELMT((*ltp_to_body_rmat), 0, 1) };
597 
598  int32_t heading_x, heading_y;
599  PPRZ_ITRIG_COS(heading_x, heading); // measured course in x-direction
600  PPRZ_ITRIG_SIN(heading_y, heading); // measured course in y-direction
601 
602  // expected_heading cross measured_heading ??
603  struct Int32Vect3 residual_ltp =
604  { 0,
605  0,
606  (expected_ltp.x * heading_y - expected_ltp.y * heading_x)/(1<<INT32_ANGLE_FRAC)};
607 
608  struct Int32Vect3 residual_imu;
609  struct Int32RMat ltp_to_imu_rmat;
610  int32_rmat_of_quat(&ltp_to_imu_rmat, &ahrs_impl.ltp_to_imu_quat);
611  int32_rmat_vmult(&residual_imu, &ltp_to_imu_rmat, &residual_ltp);
612 
613  // residual FRAC = TRIG_FRAC + TRIG_FRAC = 14 + 14 = 28
614  // rate_correction FRAC = RATE_FRAC = 12
615  // 2^12 / 2^28 * 4.0 = 1/2^14
616  // (1<<INT32_ANGLE_FRAC)/2^14 = 1/4
617  ahrs_impl.rate_correction.p += residual_imu.x/4;
618  ahrs_impl.rate_correction.q += residual_imu.y/4;
619  ahrs_impl.rate_correction.r += residual_imu.z/4;
620 
621 
622  /* crude attempt to only update bias if deviation is small
623  * e.g. needed when you only have gps providing heading
624  * and the inital heading is totally different from
625  * the gps course information you get once you have a gps fix.
626  * Otherwise the bias will be falsely "corrected".
627  */
628  int32_t sin_max_angle_deviation;
629  PPRZ_ITRIG_SIN(sin_max_angle_deviation, TRIG_BFP_OF_REAL(RadOfDeg(AHRS_BIAS_UPDATE_HEADING_THRESHOLD)));
630  if (ABS(residual_ltp.z) < sin_max_angle_deviation)
631  {
632  // residual_ltp FRAC = 2 * TRIG_FRAC = 28
633  // high_rez_bias = RATE_FRAC+28 = 40
634  // 2^40 / 2^28 * 2.5e-4 = 1
635  ahrs_impl.high_rez_bias.p -= residual_imu.x*(1<<INT32_ANGLE_FRAC);
636  ahrs_impl.high_rez_bias.q -= residual_imu.y*(1<<INT32_ANGLE_FRAC);
637  ahrs_impl.high_rez_bias.r -= residual_imu.z*(1<<INT32_ANGLE_FRAC);
638 
639  INT_RATES_RSHIFT(ahrs_impl.gyro_bias, ahrs_impl.high_rez_bias, 28);
640  }
641 }

643 void ahrs_realign_heading(int32_t heading) {
644 
645  struct Int32Quat ltp_to_body_quat = *stateGetNedToBodyQuat_i();
646 
647  /* quaternion representing only the heading rotation from ltp to body */
648  struct Int32Quat q_h_new;
649  q_h_new.qx = 0;
650  q_h_new.qy = 0;
651  PPRZ_ITRIG_SIN(q_h_new.qz, heading/2);
652  PPRZ_ITRIG_COS(q_h_new.qi, heading/2);
653 
654  /* quaternion representing current heading only */
655  struct Int32Quat q_h;
656  QUAT_COPY(q_h, ltp_to_body_quat);
657  q_h.qx = 0;
658  q_h.qy = 0;
659  int32_quat_normalize(&q_h);
660 
661  /* quaternion representing rotation from current to new heading */
662  struct Int32Quat q_c;
663  int32_quat_inv_comp_norm_shortest(&q_c, &q_h, &q_h_new);
664 
665  /* correct current heading in body frame */
666  struct Int32Quat q;
667  int32_quat_comp_norm_shortest(&q, &q_c, &ltp_to_body_quat);
668  QUAT_COPY(ltp_to_body_quat, q);
669 
670  /* compute ltp to imu rotations */
671  struct Int32Quat *body_to_imu_quat = orientationGetQuat_i(&imu.body_to_imu);
672  int32_quat_comp(&ahrs_impl.ltp_to_imu_quat, &ltp_to_body_quat, body_to_imu_quat);
673 
674  /* Set state */
675  stateSetNedToBodyQuat_i(&ltp_to_body_quat);
676 
677  ahrs_impl.heading_aligned = TRUE;
678 }
