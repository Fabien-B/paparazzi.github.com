135 static inline float pprz_isa_height_of_pressure_full(float pressure, float ref_p)
136 {
137  if (ref_p > 0.) {

139  const float inv_expo = PPRZ_ISA_GAS_CONSTANT * PPRZ_ISA_TEMP_LAPS_RATE /
140  PPRZ_ISA_GRAVITY / PPRZ_ISA_MOLAR_MASS;

155 static inline float pprz_isa_ref_pressure_of_height_full(float pressure, float height)
