96 void *computervision_thread_main(void *data)
97 {
98  // Video Input
99  struct vid_struct vid;
100  vid.device = (char *)"/dev/video1";
101  vid.w = 1280;
102  vid.h = 720;
103  vid.n_buffers = 4;
104  if (video_init(&vid) < 0) {
105  printf("Error initialising video\n");
106  computervision_thread_status = -1;
107  return 0;
108  }
109 
110  // Video Grabbing
111  struct img_struct *img_new = video_create_image(&vid);
112 
113  // Video Resizing
114  uint8_t quality_factor = VIDEO_QUALITY_FACTOR;
115  uint8_t dri_jpeg_header = 0;
116  int microsleep = (int)(1000000. / VIDEO_FPS);
117 
118  struct img_struct small;
119  small.w = vid.w / VIDEO_DOWNSIZE_FACTOR;
120  small.h = vid.h / VIDEO_DOWNSIZE_FACTOR;
121  small.buf = (uint8_t *)malloc(small.w * small.h * 2);
122 
123  // Video Compression
124  uint8_t *jpegbuf = (uint8_t *)malloc(vid.h * vid.w * 2);
125 
126  // Network Transmit
127  struct UdpSocket *vsock;
128  vsock = udp_socket(VIDEO_SOCK_IP, VIDEO_SOCK_OUT, VIDEO_SOCK_IN, FMS_BROADCAST);
129 
130  // Create SPD file and make folder if necessary
131  FILE *sdp;
132  if (system("mkdir -p /data/video/sdp") == 0) {
133  sdp = fopen("/data/video/sdp/x86_config-mjpeg.sdp", "w");
134  if (sdp != NULL) {
135  fprintf(sdp, "v=0\n");
136  fprintf(sdp, "m=video %d RTP/AVP 26\n", (int)(VIDEO_SOCK_OUT));
137  fprintf(sdp, "c=IN IP4 0.0.0.0");
138  fclose(sdp);
139  }
140  }
141 
142  // file index (search from 0)
143  int file_index = 0;
144 
145  // time
146  struct timeval last_time;
147  gettimeofday(&last_time, NULL);
148 
149  while (computer_vision_thread_command > 0) {
150  // compute usleep to have a more stable frame rate
151  struct timeval time;
152  gettimeofday(&time, NULL);
153  int dt = (int)(time.tv_sec - last_time.tv_sec) * 1000000 + (int)(time.tv_usec - last_time.tv_usec);
154  if (dt < microsleep) { usleep(microsleep - dt); }
155  last_time = time;
156 
157  // Grab new frame
158  video_grab_image(&vid, img_new);
159 
160  // Save picture on disk
161  if (computer_vision_thread_command == 2) {
162  uint8_t *end = encode_image(img_new->buf, jpegbuf, 99, FOUR_TWO_TWO, vid.w, vid.h, 1);
163  uint32_t size = end - (jpegbuf);
164  FILE *save;
165  char save_name[128];
166  if (system("mkdir -p /data/video/images") == 0) {
167  // search available index (max is 99)
168  for (; file_index < 99; file_index++) {
169  printf("search %d\n", file_index);
170  sprintf(save_name, "/data/video/images/img_%02d.jpg", file_index);
171  // test if file exists or not
172  if (access(save_name, F_OK) == -1) {
173  printf("access\n");
174  save = fopen(save_name, "w");
175  if (save != NULL) {
176  fwrite(jpegbuf, sizeof(uint8_t), size, save);
177  fclose(save);
178  } else {
179  printf("Error when opening file %s\n", save_name);
180  }
181  // leave for loop
182  break;
183  } else {
184  printf("file exists\n");
185  }
186  }
187  }
188  computer_vision_thread_command = 1;
189  viewvideo_shot = 0;
190  }
191 
192  // Resize
193  resize_uyuv(img_new, &small, VIDEO_DOWNSIZE_FACTOR);
194 
195  // JPEG encode the image:
196  uint32_t image_format = FOUR_TWO_TWO; // format (in jpeg.h)
197  uint8_t *end = encode_image(small.buf, jpegbuf, quality_factor, image_format, small.w, small.h, dri_jpeg_header);
198  uint32_t size = end - (jpegbuf);
199 
200  // Send image with RTP
201  printf("Sending an image ...%u\n", size);
202  send_rtp_frame(
203  vsock, // UDP
204  jpegbuf, size, // JPEG
205  small.w, small.h, // Img Size
206  0, // Format 422
207  quality_factor, // Jpeg-Quality
208  dri_jpeg_header, // DRI Header
209  1 // 90kHz time increment
210  );
211  // Extra note: when the time increment is set to 0,
212  // it is automaticaly calculated by the send_rtp_frame function
213  // based on gettimeofday value. This seems to introduce some lag or jitter.
214  // An other way is to compute the time increment and set the correct value.
215  // It seems that a lower value is also working (when the frame is received
216  // the timestamp is always "late" so the frame is displayed immediately).
217  // Here, we set the time increment to the lowest possible value
218  // (1 = 1/90000 s) which is probably stupid but is actually working.
219  }
220  printf("Thread Closed\n");
221  video_close(&vid);
222  computervision_thread_status = -100;
223  return 0;
224 }
