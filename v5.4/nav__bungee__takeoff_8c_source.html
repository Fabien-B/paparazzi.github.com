79 bool_t nav_bungee_takeoff_setup(uint8_t BungeeWP)
80 {
81  float ThrottleB;
82 
83  initialx = stateGetPositionEnu_f()->x;
84  initialy = stateGetPositionEnu_f()->y;
85 
86  BungeeWaypoint = BungeeWP;
87 
88  //Takeoff_Distance can only be positive
89  TDistance = fabs(Takeoff_Distance);
90 
91  //Translate initial position so that the position of the bungee is (0,0)
92  float Currentx = initialx - (WaypointX(BungeeWaypoint));
93  float Currenty = initialy - (WaypointY(BungeeWaypoint));
94 
95  //Record bungee alt (which should be the ground alt at that point)
96  BungeeAlt = waypoints[BungeeWaypoint].a;
97 
98  //Find Launch line slope and Throttle line slope
99  float MLaunch = Currenty / Currentx;
100 
101  //Find Throttle Point (the point where the throttle line and launch line intersect)
102  if (Currentx < 0) {
103  throttlePx = TDistance / sqrt(MLaunch * MLaunch + 1);
104  } else {
105  throttlePx = -(TDistance / sqrt(MLaunch * MLaunch + 1));
106  }
107 
108  if (Currenty < 0) {
109  throttlePy = sqrt((TDistance * TDistance) - (throttlePx * throttlePx));
110  } else {
111  throttlePy = -sqrt((TDistance * TDistance) - (throttlePx * throttlePx));
112  }
113 
114  //Find ThrottleLine
115  ThrottleSlope = tan(atan2(Currenty, Currentx) + (3.14 / 2));
116  ThrottleB = (throttlePy - (ThrottleSlope * throttlePx));
117 
118  //Determine whether the UAV is below or above the throttle line
119  if (Currenty > ((ThrottleSlope * Currentx) + ThrottleB)) {
120  AboveLine = TRUE;
121  } else {
122  AboveLine = FALSE;
123  }
124 
125  //Enable Launch Status and turn kill throttle on
126  CTakeoffStatus = Launch;
127  kill_throttle = 1;
128 
129  //Translate the throttle point back
130  throttlePx = throttlePx + (WaypointX(BungeeWP));
131  throttlePy = throttlePy + (WaypointY(BungeeWP));
132 
133  return FALSE;
134 }

136 bool_t nav_bungee_takeoff_run(void)
137 {
138  //Translate current position so Throttle point is (0,0)
139  float Currentx = stateGetPositionEnu_f()->x - throttlePx;
140  float Currenty = stateGetPositionEnu_f()->y - throttlePy;
141  bool_t CurrentAboveLine;
142  float ThrottleB;
143 
144  switch (CTakeoffStatus) {
145  case Launch:
146  //Follow Launch Line
147  NavVerticalAutoThrottleMode(0);
148  NavVerticalAltitudeMode(BungeeAlt + Takeoff_Height, 0.);
149  nav_route_xy(initialx, initialy, throttlePx, throttlePy);
150 
151  kill_throttle = 1;
152 
153  //recalculate lines if below min speed
154  if ((*stateGetHorizontalSpeedNorm_f()) < Takeoff_MinSpeed) {
155  initialx = stateGetPositionEnu_f()->x;
156  initialy = stateGetPositionEnu_f()->y;
157 
158  //Translate initial position so that the position of the bungee is (0,0)
159  Currentx = initialx - (WaypointX(BungeeWaypoint));
160  Currenty = initialy - (WaypointY(BungeeWaypoint));
161 
162  //Find Launch line slope
163  float MLaunch = Currenty / Currentx;
164 
165  //Find Throttle Point (the point where the throttle line and launch line intersect)
166  if (Currentx < 0) {
167  throttlePx = TDistance / sqrt(MLaunch * MLaunch + 1);
168  } else {
169  throttlePx = -(TDistance / sqrt(MLaunch * MLaunch + 1));
170  }
171 
172  if (Currenty < 0) {
173  throttlePy = sqrt((TDistance * TDistance) - (throttlePx * throttlePx));
174  } else {
175  throttlePy = -sqrt((TDistance * TDistance) - (throttlePx * throttlePx));
176  }
177 
178  //Find ThrottleLine
179  ThrottleSlope = tan(atan2(Currenty, Currentx) + (3.14 / 2));
180  ThrottleB = (throttlePy - (ThrottleSlope * throttlePx));
181 
182  //Determine whether the UAV is below or above the throttle line
183  if (Currenty > ((ThrottleSlope * Currentx) + ThrottleB)) {
184  AboveLine = TRUE;
185  } else {
186  AboveLine = FALSE;
187  }
188 
189  //Translate the throttle point back
190  throttlePx = throttlePx + (WaypointX(BungeeWaypoint));
191  throttlePy = throttlePy + (WaypointY(BungeeWaypoint));
192  }
193 
194  //Find out if the UAV is currently above the line
195  if (Currenty > (ThrottleSlope * Currentx)) {
196  CurrentAboveLine = TRUE;
197  } else {
198  CurrentAboveLine = FALSE;
199  }
200 
201  //Find out if UAV has crossed the line
202  if (AboveLine != CurrentAboveLine && (*stateGetHorizontalSpeedNorm_f()) > Takeoff_MinSpeed) {
203  CTakeoffStatus = Throttle;
204  kill_throttle = 0;
205  nav_init_stage();
206  }
207  break;
208  case Throttle:
209  //Follow Launch Line
210  NavVerticalAutoThrottleMode(AGR_CLIMB_PITCH);
211  NavVerticalThrottleMode(9600 * (1));
212  nav_route_xy(initialx, initialy, throttlePx, throttlePy);
213  kill_throttle = 0;
214 
215  if ((stateGetPositionUtm_f()->alt > BungeeAlt + Takeoff_Height - 10)
216  && ((*stateGetHorizontalSpeedNorm_f()) > Takeoff_Speed)) {
217  CTakeoffStatus = Finished;
218  return FALSE;
219  } else {
220  return TRUE;
221  }
222  break;
223  default:
224  break;
225  }
226  return TRUE;
