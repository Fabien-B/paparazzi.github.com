91 {
92  while (w5100_ch_available()) {
93  w5100_receive(w5100_rx_buf, W5100_RX_BUFFER_SIZE);
94  int c = 0;
95  do {
96  parse_pprz(t, w5100_rx_buf[ c++ ]);
