45 bool_t spi_submit(struct spi_periph *p, struct spi_transaction *t)
46 {
47  int fd = (int)p->reg_addr;
48 
49  struct spi_ioc_transfer xfer;
50  memset(&xfer, 0, sizeof xfer);
51 
52  /* length in bytes of transaction */
53  uint8_t buf_len = Max(t->input_length, t->output_length);
54 
55  /* handle transactions with different input/output length */
56  if (buf_len > t->output_length) {
57  uint8_t tx_buf[buf_len];
58  memset(tx_buf, 0, sizeof tx_buf);
59  /* copy bytes to transmit to larger buffer, rest filled with zero */
60  memcpy(tx_buf, (void *)t->output_buf, t->output_length);
61  xfer.tx_buf = (unsigned long)tx_buf;
62  } else {
63  xfer.tx_buf = (unsigned long)t->output_buf;
64  }
65 
66  if (buf_len > t->input_length) {
67  uint8_t rx_buf[buf_len];
68  memset(rx_buf, 0, sizeof rx_buf);
69  xfer.rx_buf = (unsigned long)rx_buf;
70  } else {
71  xfer.rx_buf = (unsigned long)t->input_buf;
72  }
73 
74  xfer.len = buf_len;
75  /* fixed speed of 1Mhz for now, use SPIClockDiv?? */
76  xfer.speed_hz = 1000000;
77  xfer.delay_usecs = 0;
78  if (t->dss == SPIDss16bit) {
79  xfer.bits_per_word = 16;
80  } else {
81  xfer.bits_per_word = 8;
82  }
83  if (t->select == SPISelectUnselect || t->select == SPIUnselect) {
84  xfer.cs_change = 1;
85  }
86 
87  if (ioctl(fd, SPI_IOC_MESSAGE(1), xfer) < 0) {
88  t->status = SPITransFailed;
89  return FALSE;
90  }
91 
92  /* copy recieved data if we had to use an extra rx_buffer */
93  if (buf_len > t->input_length) {
94  memcpy((void *)t->input_buf, (void *)xfer.rx_buf, t->input_length);
95  }
96 
97  t->status = SPITransSuccess;
98  return TRUE;
99 }

114 #if USE_SPI0
115 void spi0_arch_init(void)
116 {
117  int fd = open("/dev/spidev1.0", O_RDWR);
118 
119  if (fd < 0) {
120  perror("Could not open SPI device /dev/spidev1.0");
121  spi0.reg_addr = NULL;
122  return;
123  }
124  spi0.reg_addr = (void *)fd;
125 
126  /* spi mode */
127  if (ioctl(fd, SPI_IOC_WR_MODE, (SPI_CPOL | SPI_CPHA)) < 0) {
128  perror("SPI0: can't set spi mode");
129  }
130 
131  /* set to MSB first */
132  if (ioctl(fd, SPI_IOC_WR_LSB_FIRST, 0) < 0) {
133  perror("SPI0: can't set spi bit justification");
134  }
135 
136  /* bits per word default to 8 */
137  if (ioctl(fd, SPI_IOC_WR_BITS_PER_WORD, 8) < 0) {
138  perror("SPI0: can't set bits per word");
139  }
140 
141  /* max speed in hz, 1MHz for now */
142  if (ioctl(fd, SPI_IOC_WR_MAX_SPEED_HZ, 1000000) < 0) {
143  perror("SPI0: can't set max speed hz");
144  }
145 }
146 #endif /* USE_SPI0 */
