112 void imu_navstik_event(void)
113 {
114  /* MPU-60x0 event taks */
115  mpu60x0_i2c_event(&imu_navstik.mpu);
116 
117  if (imu_navstik.mpu.data_available) {
118  /* default orientation as should be printed on the pcb, z-down, ICs down */
119  RATES_COPY(imu.gyro_unscaled, imu_navstik.mpu.data_rates.rates);
120  VECT3_COPY(imu.accel_unscaled, imu_navstik.mpu.data_accel.vect);
121 
122  imu_navstik.mpu.data_available = FALSE;
123  imu_navstik.gyro_valid = TRUE;
124  imu_navstik.accel_valid = TRUE;
125  }
126 
127  /* HMC58XX event task */
128  hmc58xx_event(&imu_navstik.hmc);
129  if (imu_navstik.hmc.data_available) {
130  imu.mag_unscaled.x = imu_navstik.hmc.data.vect.y;
131  imu.mag_unscaled.y = -imu_navstik.hmc.data.vect.x;
132  imu.mag_unscaled.z = imu_navstik.hmc.data.vect.z;
133 
134  imu_navstik.hmc.data_available = FALSE;
135  imu_navstik.mag_valid = TRUE;
136  }
137 }
