128 int formation_flight(void)
129 {
130 
131  static uint8_t _1Hz = 0;
132  uint8_t nb = 0, i;
133  float ch = cosf((*stateGetHorizontalSpeedDir_f()));
134  float sh = sinf((*stateGetHorizontalSpeedDir_f()));
135  form_n = 0.;
136  form_e = 0.;
137  form_a = 0.;
138  form_speed = (*stateGetHorizontalSpeedNorm_f());
139  form_speed_n = (*stateGetHorizontalSpeedNorm_f()) * ch;
140  form_speed_e = (*stateGetHorizontalSpeedNorm_f()) * sh;
141 
142  if (AC_ID == leader_id) {
143  stateGetPositionEnu_f()->x += formation[the_acs_id[AC_ID]].east;
144  stateGetPositionEnu_f()->y += formation[the_acs_id[AC_ID]].north;
145  }
146  // set info for this AC
147  SetAcInfo(AC_ID, stateGetPositionEnu_f()->x, stateGetPositionEnu_f()->y, (*stateGetHorizontalSpeedDir_f()),
148  stateGetPositionUtm_f()->alt, (*stateGetHorizontalSpeedNorm_f()), stateGetSpeedEnu_f()->z, gps.tow);
149 
150  // broadcast info
151  uint8_t ac_id = AC_ID;
152  enum slot_status status = formation[the_acs_id[AC_ID]].status;
153  DOWNLINK_SEND_FORMATION_STATUS_TM(DefaultChannel, DefaultDevice, &ac_id, &leader_id, &status);
154  if (++_1Hz >= 4) {
155  _1Hz = 0;
156  DOWNLINK_SEND_FORMATION_SLOT_TM(DefaultChannel, DefaultDevice, &ac_id, &form_mode,
157  &formation[the_acs_id[AC_ID]].east,
158  &formation[the_acs_id[AC_ID]].north,
159  &formation[the_acs_id[AC_ID]].alt);
160  }
161  if (formation[the_acs_id[AC_ID]].status != ACTIVE) { return FALSE; } // AC not ready
162 
163  // get leader info
164  struct ac_info_ * leader = get_ac_info(leader_id);
165  if (formation[the_acs_id[leader_id]].status == UNSET ||
166  formation[the_acs_id[leader_id]].status == IDLE) { return FALSE; } // leader not ready or not in formation
167 
168  // compute slots in the right reference frame
169  struct slot_ form[NB_ACS];
170  float cr = 0., sr = 1.;
171  if (form_mode == FORM_MODE_COURSE) {
172  cr = cosf(leader->course);
173  sr = sinf(leader->course);
174  }
175  for (i = 0; i < NB_ACS; ++i) {
176  if (formation[i].status == UNSET) { continue; }
177  form[i].east = formation[i].east * sr - formation[i].north * cr;
178  form[i].north = formation[i].east * cr + formation[i].north * sr;
179  form[i].alt = formation[i].alt;
180  }
181 
182  // compute control forces
183  for (i = 0; i < NB_ACS; ++i) {
184  if (the_acs[i].ac_id == AC_ID) { continue; }
185  struct ac_info_ * ac = get_ac_info(the_acs[i].ac_id);
186  float delta_t = Max((int)(gps.tow - ac->itow) / 1000., 0.);
187  if (delta_t > FORM_CARROT) {
188  // if AC not responding for too long
189  formation[i].status = LOST;
190  continue;
191  } else { formation[i].status = ACTIVE; }
192  // compute control if AC is ACTIVE and around the same altitude (maybe not so usefull)
193  if (formation[i].status == ACTIVE && fabs(stateGetPositionUtm_f()->alt - ac->alt) < form_prox && ac->alt > 0) {
194  form_e += (ac->east + ac->gspeed * sinf(ac->course) * delta_t - stateGetPositionEnu_f()->x)
195  - (form[i].east - form[the_acs_id[AC_ID]].east);
196  form_n += (ac->north + ac->gspeed * cosf(ac->course) * delta_t - stateGetPositionEnu_f()->y)
197  - (form[i].north - form[the_acs_id[AC_ID]].north);
198  form_a += (ac->alt - stateGetPositionUtm_f()->alt) - (formation[i].alt - formation[the_acs_id[AC_ID]].alt);
199  form_speed += ac->gspeed;
200  //form_speed_e += ac->gspeed * sinf(ac->course);
201  //form_speed_n += ac->gspeed * cosf(ac->course);
202  ++nb;
203  }
204  }
205  uint8_t _nb = Max(1, nb);
206  form_n /= _nb;
207  form_e /= _nb;
208  form_a /= _nb;
209  form_speed = form_speed / (nb + 1) - (*stateGetHorizontalSpeedNorm_f());
210  //form_speed_e = form_speed_e / (nb+1) - (*stateGetHorizontalSpeedNorm_f()) * sh;
211  //form_speed_n = form_speed_n / (nb+1) - (*stateGetHorizontalSpeedNorm_f()) * ch;
212 
213  // set commands
214  NavVerticalAutoThrottleMode(0.);
215 
216  // altitude loop
217  float alt = 0.;
218  if (AC_ID == leader_id) { alt = nav_altitude; }
219  else { alt = leader->alt - form[the_acs_id[leader_id]].alt; }
220  alt += formation[the_acs_id[AC_ID]].alt + coef_form_alt * form_a;
221  flight_altitude = Max(alt, ground_alt + SECURITY_HEIGHT);
222 
223  // carrot
224  if (AC_ID != leader_id) {
225  float dx = form[the_acs_id[AC_ID]].east - form[the_acs_id[leader_id]].east;
226  float dy = form[the_acs_id[AC_ID]].north - form[the_acs_id[leader_id]].north;
227  desired_x = leader->east + NOMINAL_AIRSPEED * form_carrot * sinf(leader->course) + dx;
228  desired_y = leader->north + NOMINAL_AIRSPEED * form_carrot * cosf(leader->course) + dy;
229  // fly to desired
230  fly_to_xy(desired_x, desired_y);
231  desired_x = leader->east + dx;
232  desired_y = leader->north + dy;
233  // lateral correction
234  //float diff_heading = asin((dx*ch - dy*sh) / sqrt(dx*dx + dy*dy));
235  //float diff_course = leader->course - (*stateGetHorizontalSpeedDir_f());
236  //NormRadAngle(diff_course);
237  //h_ctl_roll_setpoint += coef_form_course * diff_course;
238  //h_ctl_roll_setpoint += coef_form_course * diff_heading;
239  }
240  //BoundAbs(h_ctl_roll_setpoint, h_ctl_roll_max_setpoint);
241 
242  // speed loop
243  if (nb > 0) {
244  float cruise = V_CTL_AUTO_THROTTLE_NOMINAL_CRUISE_THROTTLE;
245  cruise += coef_form_pos * (form_n * ch + form_e * sh) + coef_form_speed * form_speed;
246  Bound(cruise, V_CTL_AUTO_THROTTLE_MIN_CRUISE_THROTTLE, V_CTL_AUTO_THROTTLE_MAX_CRUISE_THROTTLE);
247  v_ctl_auto_throttle_cruise_throttle = cruise;
248  }
249  return TRUE;
250 }
