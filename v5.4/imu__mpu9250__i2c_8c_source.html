116 void imu_mpu9250_event(void)
117 {
118  // If the MPU9250 I2C transaction has succeeded: convert the data
119  mpu9250_i2c_event(&imu_mpu9250.mpu);
120 
121  if (imu_mpu9250.mpu.data_available) {
122  // set channel order
123  struct Int32Vect3 accel = {
124  (int32_t)(imu_mpu9250.mpu.data_accel.value[IMU_MPU9250_CHAN_X]),
125  (int32_t)(imu_mpu9250.mpu.data_accel.value[IMU_MPU9250_CHAN_Y]),
126  (int32_t)(imu_mpu9250.mpu.data_accel.value[IMU_MPU9250_CHAN_Z])
127  };
128  struct Int32Rates rates = {
129  (int32_t)(imu_mpu9250.mpu.data_rates.value[IMU_MPU9250_CHAN_X]),
130  (int32_t)(imu_mpu9250.mpu.data_rates.value[IMU_MPU9250_CHAN_Y]),
131  (int32_t)(imu_mpu9250.mpu.data_rates.value[IMU_MPU9250_CHAN_Z])
132  };
133  // unscaled vector
134  VECT3_COPY(imu.accel_unscaled, accel);
135  RATES_COPY(imu.gyro_unscaled, rates);
136 
137  imu_mpu9250.mpu.data_available = FALSE;
138  imu_mpu9250.gyro_valid = TRUE;
139  imu_mpu9250.accel_valid = TRUE;
140  }
141  // Test if mag data are updated
142  if (imu_mpu9250.mpu.akm.data_available) {
143  struct Int32Vect3 mag = {
144  (int32_t)(imu_mpu9250.mpu.akm.data.value[IMU_MPU9250_CHAN_Y]),
145  (int32_t)(imu_mpu9250.mpu.akm.data.value[IMU_MPU9250_CHAN_X]),
146  -(int32_t)(imu_mpu9250.mpu.akm.data.value[IMU_MPU9250_CHAN_Z])
147  };
148  VECT3_COPY(imu.mag_unscaled, mag);
149  imu_mpu9250.mag_valid = TRUE;
150  imu_mpu9250.mpu.akm.data_available = FALSE;
151  }
152 
153 }
