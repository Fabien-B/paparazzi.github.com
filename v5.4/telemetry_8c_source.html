38 bool_t register_periodic_telemetry(struct pprz_telemetry *_pt, const char *_msg, telemetry_cb _cb)
39 {
40  // return FALSE if NULL is passed as pprz_telemetry
41  if (_pt == NULL) { return FALSE; }
42  // look for message name
43  uint8_t i;
44  for (i = 0; i < _pt->nb; i++) {
45  if (str_equal(_pt->msgs[i].msg, _msg)) {
46  // register callback if not already done
47  if (_pt->msgs[i].cb == NULL) {
48  _pt->msgs[i].cb = _cb;
49  return TRUE;
50  } else { return FALSE; }
51  }
52  }
53  // message name is not in telemetry file
54  return FALSE;
55 }
