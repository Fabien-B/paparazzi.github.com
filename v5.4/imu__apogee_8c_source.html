72  // MPU-60X0
73  mpu60x0_i2c_init(&imu_apogee.mpu, &(IMU_APOGEE_I2C_DEV), MPU60X0_ADDR_ALT);
74  // change the default configuration
75  imu_apogee.mpu.config.smplrt_div = APOGEE_SMPLRT_DIV;
76  imu_apogee.mpu.config.dlpf_cfg = APOGEE_LOWPASS_FILTER;

78  imu_apogee.mpu.config.accel_range = APOGEE_ACCEL_RANGE;

105 void imu_apogee_event(void)
106 {
107  // If the itg3200 I2C transaction has succeeded: convert the data
108  mpu60x0_i2c_event(&imu_apogee.mpu);
109  if (imu_apogee.mpu.data_available) {
110  RATES_ASSIGN(imu.gyro_unscaled, imu_apogee.mpu.data_rates.rates.p, -imu_apogee.mpu.data_rates.rates.q,
111  -imu_apogee.mpu.data_rates.rates.r);
112  VECT3_ASSIGN(imu.accel_unscaled, imu_apogee.mpu.data_accel.vect.x, -imu_apogee.mpu.data_accel.vect.y,
113  -imu_apogee.mpu.data_accel.vect.z);
114  imu_apogee.mpu.data_available = FALSE;
115  imu_apogee.gyr_valid = TRUE;
116  imu_apogee.acc_valid = TRUE;
117  }
118 }
