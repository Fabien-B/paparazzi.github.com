60 /* Big Endian to Float */
61 static inline float bef(volatile uint8_t *c) {
62  float f;
63  int8_t * p;
64  p = ((int8_t *)&f)+3;
65  *p-- = *c++;
66  *p-- = *c++;
67  *p-- = *c++;
68  *p = *c;
69  return f;
70 }

72 static inline bool_t gx3_verify_chk(volatile uint8_t *buff_add) {
73  uint16_t i,chk_calc;
74  chk_calc = 0;
75  for (i=0;i<GX3_MSG_LEN-2;i++) {
76  chk_calc += (uint8_t)*buff_add++;
77  }
78  return (chk_calc == ( (((uint16_t)*buff_add)<<8) + (uint8_t)*(buff_add+1) ));

295 /* GX3 Packet Collection */
296 void gx3_packet_parse( uint8_t c ) {
297  switch (ahrs_impl.gx3_packet.status) {
298  case GX3PacketWaiting:
299  ahrs_impl.gx3_packet.msg_idx = 0;
300  if (c == GX3_HEADER) {
301  ahrs_impl.gx3_packet.status++;
302  ahrs_impl.gx3_packet.msg_buf[ahrs_impl.gx3_packet.msg_idx] = c;
303  ahrs_impl.gx3_packet.msg_idx++;
304  } else {
305  ahrs_impl.gx3_packet.hdr_error++;
306  }
307  break;
308  case GX3PacketReading:
309  ahrs_impl.gx3_packet.msg_buf[ahrs_impl.gx3_packet.msg_idx] = c;
310  ahrs_impl.gx3_packet.msg_idx++;
311  if (ahrs_impl.gx3_packet.msg_idx == GX3_MSG_LEN) {
312  if (gx3_verify_chk(ahrs_impl.gx3_packet.msg_buf)) {
313  ahrs_impl.gx3_packet.msg_available = TRUE;
314  } else {
315  ahrs_impl.gx3_packet.msg_available = FALSE;
316  ahrs_impl.gx3_packet.chksm_error++;
317  }
318  ahrs_impl.gx3_packet.status = 0;
319  }
320  break;
321  default:
322  ahrs_impl.gx3_packet.status = 0;
323  ahrs_impl.gx3_packet.msg_idx = 0;
324  break;
325  }
326 }
