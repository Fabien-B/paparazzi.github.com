63 void udp_receive( void ) {
64  //First check if it isn't already having a message
65  if(dl_msg_available == TRUE) {
66  return;
67  }
68 
69  //Read from the network
70  uint16_t read = network_read(network, udp_read_buffer, TRANSPORT_PAYLOAD_LEN);
71  if(read > 0) {
72 
73  //Parse the packet
74  if(udp_read_buffer[0] == STX) {
75  uint8_t size = udp_read_buffer[1]-4; // minus STX, LENGTH, CK_A, CK_B
76  uint8_t ck_aa, ck_bb;
77  ck_aa = ck_bb = size+4;
78 
79  // index-offset plus 2 for STX and LENGTH
80  for (int i = 2; i < size+2; i++) {
81  dl_buffer[i-2] = udp_read_buffer[i];
82  ck_aa += udp_read_buffer[i];
83  ck_bb += ck_aa;
84  }
85 
86  // if both checksums are good, tell datalink that the message is available
87  if (udp_read_buffer[2+size] == ck_aa && udp_read_buffer[2+size+1] == ck_bb) {
88  dl_msg_available = TRUE;
89  }
90  }
91 
92  memset(&udp_read_buffer[0], 0, sizeof(udp_read_buffer));
93  }
94 }
