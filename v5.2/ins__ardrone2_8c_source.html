93 void ins_propagate() {
94  /* untilt accels and speeds */
95  FLOAT_RMAT_VECT3_TRANSP_MUL(ins_impl.ltp_accel, (*stateGetNedToBodyRMat_f()), ahrs_impl.accel);
96  FLOAT_RMAT_VECT3_TRANSP_MUL(ins_impl.ltp_speed, (*stateGetNedToBodyRMat_f()), ahrs_impl.speed);
97 
98  //Add g to the accelerations
99  ins_impl.ltp_accel.z += 9.81;
100 
101  //Save the accelerations and speeds
102  stateSetAccelNed_f(&ins_impl.ltp_accel);
103  stateSetSpeedNed_f(&ins_impl.ltp_speed);
104 
105  //Don't set the height if we use the one from the gps
106 #if !USE_GPS_HEIGHT
107  //Set the height and save the position
108  ins_impl.ltp_pos.z = -(ahrs_impl.altitude * INT32_POS_OF_CM_NUM) / INT32_POS_OF_CM_DEN;
109  stateSetPositionNed_i(&ins_impl.ltp_pos);
110 #endif
111 }

124  stateSetLocalOrigin_i(&ins_impl.ltp_def);

130 
131  //When we don't want to use the height of the navdata we can use the gps height
132 #if USE_GPS_HEIGHT

138  //Set the local origin
