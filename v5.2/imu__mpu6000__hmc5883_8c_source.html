103 void imu_mpu_hmc_event(void)
104 {
105  mpu60x0_spi_event(&imu_mpu_hmc.mpu);
106  if (imu_mpu_hmc.mpu.data_available) {
107  RATES_COPY(imu.gyro_unscaled, imu_mpu_hmc.mpu.data_rates.rates);
108  VECT3_COPY(imu.accel_unscaled, imu_mpu_hmc.mpu.data_accel.vect);
109  imu_mpu_hmc.mpu.data_available = FALSE;
110  imu_mpu_hmc.gyro_valid = TRUE;
111  imu_mpu_hmc.accel_valid = TRUE;
112  }
113 
114  /* HMC58XX event task */
115  hmc58xx_event(&imu_mpu_hmc.hmc);
116  if (imu_mpu_hmc.hmc.data_available) {
117  /* mag rotated by 90deg around z axis relative to MPU */
118  imu.mag_unscaled.x = imu_mpu_hmc.hmc.data.vect.y;
119  imu.mag_unscaled.y = -imu_mpu_hmc.hmc.data.vect.x;
120  imu.mag_unscaled.z = imu_mpu_hmc.hmc.data.vect.z;
121  imu_mpu_hmc.hmc.data_available = FALSE;
122  imu_mpu_hmc.mag_valid = TRUE;
123  }
124 }
