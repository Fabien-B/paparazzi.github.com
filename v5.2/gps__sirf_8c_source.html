50 void sirf_parse_char(uint8_t c) {
51  switch(gps_sirf.read_state) {
52  case UNINIT:
53  if(c == 0xA0) {
54  gps_sirf.msg_len = 0;
55  gps_sirf.msg_buf[gps_sirf.msg_len] = c;
56  gps_sirf.msg_len++;
57  gps_sirf.read_state = GOT_A0;
58  }
59  break;
60  case GOT_A0:
61  if(c == 0xA2) {
62  gps_sirf.msg_buf[gps_sirf.msg_len] = c;
63  gps_sirf.msg_len++;
64  gps_sirf.read_state = GOT_A2;
65  }
66  else
67  goto restart;
68  break;
69  case GOT_A2:
70  gps_sirf.msg_buf[gps_sirf.msg_len] = c;
71  gps_sirf.msg_len++;
72  if(c == 0xB0)
73  gps_sirf.read_state = GOT_B0;
74  break;
75  case GOT_B0:
76  if(c == 0xB3) {
77  gps_sirf.msg_buf[gps_sirf.msg_len] = c;
78  gps_sirf.msg_len++;
79  gps_sirf.msg_available = TRUE;
80  }
81  else
82  goto restart;
83  break;
84  default:
85  break;
86  }
87  return;
88 
89  restart:
90  gps_sirf.read_state = UNINIT;
91 }

98 void sirf_parse_41(void) {
99  struct sirf_msg_41* p = (struct sirf_msg_41*)&gps_sirf.msg_buf[4];
100 
101  gps.tow = Invert4Bytes(p->tow);
102  gps.hmsl = Invert4Bytes(p->alt_msl)*10;
103  gps.num_sv = p->num_sat;
104  gps.nb_channels = p ->num_sat;
105 
106  /* read latitude, longitude and altitude from packet */
107  gps.lla_pos.lat = Invert4Bytes(p->latitude);
108  gps.lla_pos.lon = Invert4Bytes(p->longitude);
109  gps.lla_pos.alt = Invert4Bytes(p->alt_ellipsoid) * 10;
110 
111 #if GPS_USE_LATLONG
112  /* convert to utm */
113  struct LlaCoor_f lla_f;
114  LLA_FLOAT_OF_BFP(lla_f, gps.lla_pos);
115  struct UtmCoor_f utm_f;
116  utm_f.zone = nav_utm_zone0;
117  utm_of_lla_f(&utm_f, &lla_f);
118 
119  /* copy results of utm conversion */
120  gps.utm_pos.east = utm_f.east*100;
121  gps.utm_pos.north = utm_f.north*100;
122  gps.utm_pos.alt = gps.lla_pos.alt;
123  gps.utm_pos.zone = nav_utm_zone0;
124 #endif
125 
126  gps.sacc = (Invert2Bytes(p->ehve)>>16);
127  gps.course = RadOfDeg(Invert2Bytes(p->cog))*pow(10, 5);
128  gps.gspeed = RadOfDeg(Invert2Bytes(p->sog))*pow(10, 5);
129  gps.cacc = RadOfDeg(Invert2Bytes(p->heading_err))*pow(10, 5);
130  gps.pacc = Invert4Bytes(p->ehpe);
131  gps.pdop = p->hdop * 20;
132 
133  if ((p->nav_type >> 8 & 0x7) >= 0x4)
134  gps.fix = GPS_FIX_3D;
135  else if((p->nav_type >> 8 & 0x7) >= 0x1)
136  gps.fix = GPS_FIX_2D;
137  else
138  gps.fix = GPS_FIX_NONE;
139 
140 
141  //Let gps_sirf know we have a position update
142  gps_sirf.pos_available = TRUE;
143 }

145 void sirf_parse_2(void) {
146  struct sirf_msg_2* p = (struct sirf_msg_2*)&gps_sirf.msg_buf[4];
147 
148  gps.week = Invert2Bytes(p->week);
149 
150  gps.ecef_pos.x = Invert4Bytes(p->x_pos) * 100;
151  gps.ecef_pos.y = Invert4Bytes(p->y_pos) * 100;
152  gps.ecef_pos.z = Invert4Bytes(p->z_pos) * 100;
153 
154  gps.ecef_vel.x = (Invert2Bytes(p->vx)>>16)*100/8;
155  gps.ecef_vel.y = (Invert2Bytes(p->vy)>>16)*100/8;
156  gps.ecef_vel.z = (Invert2Bytes(p->vz)>>16)*100/8;
157 
158  if(gps.fix == GPS_FIX_3D) {
159  ticks++;
160 #if DEBUG_SIRF
161  printf("GPS %i %i %i %i\n", ticks, (sys_time.nb_sec - start_time), ticks2, (sys_time.nb_sec - start_time2));
162 #endif
163  }
164  else if(sys_time.nb_sec - gps.last_3dfix_time > 10) {
165  start_time = sys_time.nb_sec;
166  ticks = 0;
167  }
168 
169 }

171 void sirf_parse_msg(void) {
172  //Set position available to false and check if it is a valid message
173  gps_sirf.pos_available = FALSE;
174  if(gps_sirf.msg_len < 8)
175  return;
176 
177  if(start_time2 == 0)
178  start_time2 = sys_time.nb_sec;
179  ticks2++;
180 
181  //Check the message id and parse the message
182  uint8_t message_id = gps_sirf.msg_buf[4];
183  switch(message_id) {
184  case 0x29:
185  sirf_parse_41();
186  break;
187  case 0x02:
188  sirf_parse_2();
189  break;
190  }

192 }
