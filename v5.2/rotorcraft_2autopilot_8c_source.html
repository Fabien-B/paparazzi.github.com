278 #define NAV_PRESCALER (PERIODIC_FREQUENCY / NAV_FREQ)
279 void autopilot_periodic(void) {
280 
281  RunOnceEvery(NAV_PRESCALER, compute_dist2_to_home());
282 
283  if (autopilot_in_flight && autopilot_mode == AP_MODE_NAV) {
284  if (too_far_from_home) {
285  if (dist2_to_home > failsafe_mode_dist2)
286  autopilot_set_mode(FAILSAFE_MODE_TOO_FAR_FROM_HOME);
287  else
288  autopilot_set_mode(AP_MODE_HOME);
289  }
290  }
291 
292  if (autopilot_mode == AP_MODE_HOME) {
293  RunOnceEvery(NAV_PRESCALER, nav_home());
294  }
295  else {
296  // otherwise always call nav_periodic_task so that carrot is always updated in GCS for other modes
297  RunOnceEvery(NAV_PRESCALER, nav_periodic_task());
298  }
299 
300 
301  /* If in FAILSAFE mode and either already not in_flight anymore
302  * or just "detected" ground, go to KILL mode.
303  */
304  if (autopilot_mode == AP_MODE_FAILSAFE) {
305  if (!autopilot_in_flight)
306  autopilot_set_mode(AP_MODE_KILL);
307 
308 #if FAILSAFE_GROUND_DETECT
309 INFO("Using FAILSAFE_GROUND_DETECT: KILL")
310  if (autopilot_ground_detected)
311  autopilot_set_mode(AP_MODE_KILL);
312 #endif
313  }
314 
315  /* Reset ground detection _after_ running flight plan
316  */
317  if (!autopilot_in_flight) {
318  autopilot_ground_detected = FALSE;
319  autopilot_detect_ground_once = FALSE;
320  }
321 
322  /* Set fixed "failsafe" commands from airframe file if in KILL mode.
323  * If in FAILSAFE mode, run normal loops with failsafe attitude and
324  * downwards velocity setpoints.
325  */
326  if (autopilot_mode == AP_MODE_KILL) {
327  SetCommands(commands_failsafe);
328  }
329  else {
330  guidance_v_run( autopilot_in_flight );
331  guidance_h_run( autopilot_in_flight );
332  SetRotorcraftCommands(stabilization_cmd, autopilot_in_flight, autopilot_motors_on);
333  }

338 void autopilot_set_mode(uint8_t new_autopilot_mode) {
339 
340  /* force kill mode as long as AHRS is not aligned */
341  if (!ahrs_is_aligned())
342  new_autopilot_mode = AP_MODE_KILL;
343 
344  if (new_autopilot_mode != autopilot_mode) {
345  /* horizontal mode */
346  switch (new_autopilot_mode) {
347  case AP_MODE_FAILSAFE:
348 #ifndef KILL_AS_FAILSAFE
349  stabilization_attitude_set_failsafe_setpoint();
350  guidance_h_mode_changed(GUIDANCE_H_MODE_ATTITUDE);
351  break;
352 #endif
353  case AP_MODE_KILL:
354  autopilot_in_flight = FALSE;
355  autopilot_in_flight_counter = 0;
356  guidance_h_mode_changed(GUIDANCE_H_MODE_KILL);
357  break;
358  case AP_MODE_RC_DIRECT:
359  guidance_h_mode_changed(GUIDANCE_H_MODE_RC_DIRECT);
360  break;
361  case AP_MODE_RATE_DIRECT:
362  case AP_MODE_RATE_Z_HOLD:
363  guidance_h_mode_changed(GUIDANCE_H_MODE_RATE);
364  break;
365  case AP_MODE_ATTITUDE_RC_CLIMB:
366  case AP_MODE_ATTITUDE_DIRECT:
367  case AP_MODE_ATTITUDE_CLIMB:
368  case AP_MODE_ATTITUDE_Z_HOLD:
369  guidance_h_mode_changed(GUIDANCE_H_MODE_ATTITUDE);
370  break;
371  case AP_MODE_FORWARD:
372  guidance_h_mode_changed(GUIDANCE_H_MODE_FORWARD);
373  break;
374  case AP_MODE_CARE_FREE_DIRECT:
375  guidance_h_mode_changed(GUIDANCE_H_MODE_CARE_FREE);
376  break;
377  case AP_MODE_HOVER_DIRECT:
378  case AP_MODE_HOVER_CLIMB:
379  case AP_MODE_HOVER_Z_HOLD:
380  guidance_h_mode_changed(GUIDANCE_H_MODE_HOVER);
381  break;
382  case AP_MODE_HOME:
383  case AP_MODE_NAV:
384  guidance_h_mode_changed(GUIDANCE_H_MODE_NAV);
385  break;
386  default:
387  break;
388  }
389  /* vertical mode */
390  switch (new_autopilot_mode) {
391  case AP_MODE_FAILSAFE:
392 #ifndef KILL_AS_FAILSAFE
393  guidance_v_mode_changed(GUIDANCE_V_MODE_CLIMB);
394  guidance_v_zd_sp = SPEED_BFP_OF_REAL(FAILSAFE_DESCENT_SPEED);
395  break;
396 #endif
397  case AP_MODE_KILL:
398  autopilot_set_motors_on(FALSE);
399  stabilization_cmd[COMMAND_THRUST] = 0;
400  guidance_v_mode_changed(GUIDANCE_V_MODE_KILL);
401  break;
402  case AP_MODE_RC_DIRECT:
403  case AP_MODE_RATE_DIRECT:
404  case AP_MODE_ATTITUDE_DIRECT:
405  case AP_MODE_HOVER_DIRECT:
406  case AP_MODE_CARE_FREE_DIRECT:
407  case AP_MODE_FORWARD:
408  guidance_v_mode_changed(GUIDANCE_V_MODE_RC_DIRECT);
409  break;
410  case AP_MODE_RATE_RC_CLIMB:
411  case AP_MODE_ATTITUDE_RC_CLIMB:
412  guidance_v_mode_changed(GUIDANCE_V_MODE_RC_CLIMB);
413  break;
414  case AP_MODE_ATTITUDE_CLIMB:
415  case AP_MODE_HOVER_CLIMB:
416  guidance_v_mode_changed(GUIDANCE_V_MODE_CLIMB);
417  break;
418  case AP_MODE_RATE_Z_HOLD:
419  case AP_MODE_ATTITUDE_Z_HOLD:
420  case AP_MODE_HOVER_Z_HOLD:
421  guidance_v_mode_changed(GUIDANCE_V_MODE_HOVER);
422  break;
423  case AP_MODE_HOME:
424  case AP_MODE_NAV:
425  guidance_v_mode_changed(GUIDANCE_V_MODE_NAV);
426  break;
427  default:
428  break;
429  }
430  autopilot_mode = new_autopilot_mode;
431  }

436 void autopilot_check_in_flight(bool_t motors_on) {
437  if (autopilot_in_flight) {
438  if (autopilot_in_flight_counter > 0) {
439  /* probably in_flight if thrust, speed and accel above IN_FLIGHT_MIN thresholds */
440  if ((stabilization_cmd[COMMAND_THRUST] <= AUTOPILOT_IN_FLIGHT_MIN_THRUST) &&
441  (abs(stateGetSpeedNed_f()->z) < AUTOPILOT_IN_FLIGHT_MIN_SPEED) &&
442  (abs(stateGetAccelNed_f()->z) < AUTOPILOT_IN_FLIGHT_MIN_ACCEL))
443  {
444  autopilot_in_flight_counter--;
445  if (autopilot_in_flight_counter == 0) {
446  autopilot_in_flight = FALSE;
447  }
448  }
449  else { /* thrust, speed or accel not above min threshold, reset counter */
450  autopilot_in_flight_counter = AUTOPILOT_IN_FLIGHT_TIME;
451  }
452  }
453  }
454  else { /* currently not in flight */
455  if (autopilot_in_flight_counter < AUTOPILOT_IN_FLIGHT_TIME &&
456  motors_on)
457  {
458  /* if thrust above min threshold, assume in_flight.
459  * Don't check for velocity and acceleration above threshold here...
460  */
461  if (stabilization_cmd[COMMAND_THRUST] > AUTOPILOT_IN_FLIGHT_MIN_THRUST) {
462  autopilot_in_flight_counter++;
463  if (autopilot_in_flight_counter == AUTOPILOT_IN_FLIGHT_TIME)
464  autopilot_in_flight = TRUE;
465  }
466  else { /* currently not in_flight and thrust below threshold, reset counter */
467  autopilot_in_flight_counter = 0;
468  }
469  }
470  }

484 void autopilot_on_rc_frame(void) {
485 
486  if (kill_switch_is_on()) {
487  autopilot_set_mode(AP_MODE_KILL);
488  }
489  else if ((autopilot_mode != AP_MODE_HOME)
490 #if UNLOCKED_HOME_MODE
491  || !too_far_from_home
492 #endif
493  )
494  {
495  uint8_t new_autopilot_mode = 0;
496  AP_MODE_OF_PPRZ(radio_control.values[RADIO_MODE], new_autopilot_mode);
497 
498 #if USE_GPS
499  /* don't enter NAV mode if GPS is lost (this also prevents mode oscillations) */
500  if (!(new_autopilot_mode == AP_MODE_NAV && GpsIsLost()))
501 #endif
502  autopilot_set_mode(new_autopilot_mode);
503  }
504 
505  /* if not in FAILSAFE or HOME mode check motor and in_flight status, read RC */
506  if (autopilot_mode != AP_MODE_FAILSAFE && autopilot_mode != AP_MODE_HOME) {
507 
508  /* if there are some commands that should always be set from RC, do it */
509 #ifdef SetAutoCommandsFromRC
510  SetAutoCommandsFromRC(commands, radio_control.values);
511 #endif
512 
513  /* if not in NAV_MODE set commands from the rc */
514 #ifdef SetCommandsFromRC
515  if (autopilot_mode != AP_MODE_NAV) {
516  SetCommandsFromRC(commands, radio_control.values);
517  }
518 #endif
519 
520  /* an arming sequence is used to start/stop motors */
521  autopilot_arming_check_motors_on();
522  kill_throttle = ! autopilot_motors_on;
523 
524  guidance_v_read_rc();
525  guidance_h_read_rc(autopilot_in_flight);
526  }
