212 void guidance_h_mode_changed(uint8_t new_mode)
213 {
214  if (new_mode == guidance_h_mode) {
215  return;
216  }
217 
218  if (new_mode != GUIDANCE_H_MODE_FORWARD && new_mode != GUIDANCE_H_MODE_RATE) {
219  transition_percentage = 0;
220  transition_theta_offset = 0;
221  }
222 
223  switch (new_mode) {
224  case GUIDANCE_H_MODE_RC_DIRECT:
225  stabilization_none_enter();
226  break;
227 
228  case GUIDANCE_H_MODE_RATE:
229  stabilization_rate_enter();
230  break;
231 
232  case GUIDANCE_H_MODE_CARE_FREE:
233  stabilization_attitude_reset_care_free_heading();
234  case GUIDANCE_H_MODE_FORWARD:
235  case GUIDANCE_H_MODE_ATTITUDE:
236 #if NO_ATTITUDE_RESET_ON_MODE_CHANGE
237  /* reset attitude stabilization if previous mode was not using it */
238  if (guidance_h_mode == GUIDANCE_H_MODE_KILL ||
239  guidance_h_mode == GUIDANCE_H_MODE_RATE ||
240  guidance_h_mode == GUIDANCE_H_MODE_RC_DIRECT)
241 #endif
242  stabilization_attitude_enter();
243  break;
244 
245  case GUIDANCE_H_MODE_HOVER:
246  guidance_h_hover_enter();
247 #if NO_ATTITUDE_RESET_ON_MODE_CHANGE
248  /* reset attitude stabilization if previous mode was not using it */
249  if (guidance_h_mode == GUIDANCE_H_MODE_KILL ||
250  guidance_h_mode == GUIDANCE_H_MODE_RATE ||
251  guidance_h_mode == GUIDANCE_H_MODE_RC_DIRECT)
252 #endif
253  stabilization_attitude_enter();
254  break;
255 
256 #if GUIDANCE_H_MODE_MODULE_SETTING == GUIDANCE_H_MODE_MODULE
257  case GUIDANCE_H_MODE_MODULE:
258  guidance_h_module_enter();
259  break;
260 #endif
261 
262  case GUIDANCE_H_MODE_NAV:
263  guidance_h_nav_enter();
264 #if NO_ATTITUDE_RESET_ON_MODE_CHANGE
265  /* reset attitude stabilization if previous mode was not using it */
266  if (guidance_h_mode == GUIDANCE_H_MODE_KILL ||
267  guidance_h_mode == GUIDANCE_H_MODE_RATE ||
268  guidance_h_mode == GUIDANCE_H_MODE_RC_DIRECT)
269 #endif
270  stabilization_attitude_enter();
271  break;
272 
273  default:
274  break;
275  }
276 
277  guidance_h_mode = new_mode;
278

334 void guidance_h_run(bool_t in_flight)
335 {
336  switch (guidance_h_mode) {
337 
338  case GUIDANCE_H_MODE_RC_DIRECT:
339  stabilization_none_run(in_flight);
340  break;
341 
342  case GUIDANCE_H_MODE_RATE:
343  stabilization_rate_run(in_flight);
344  break;
345 
346  case GUIDANCE_H_MODE_FORWARD:
347  if (transition_percentage < (100 << INT32_PERCENTAGE_FRAC)) {
348  transition_run();
349  }
350  case GUIDANCE_H_MODE_CARE_FREE:
351  case GUIDANCE_H_MODE_ATTITUDE:
352  stabilization_attitude_run(in_flight);
353  break;
354 
355  case GUIDANCE_H_MODE_HOVER:
356  if (!in_flight) {
357  guidance_h_hover_enter();
358  }
359 
360  guidance_h_update_reference();
361 
362  /* set psi command */
363  guidance_h_heading_sp = guidance_h_rc_sp.psi;
364  /* compute x,y earth commands */
365  guidance_h_traj_run(in_flight);
366  /* set final attitude setpoint */
367  stabilization_attitude_set_earth_cmd_i(&guidance_h_cmd_earth,
368  guidance_h_heading_sp);
369  stabilization_attitude_run(in_flight);
370  break;
371 
372  case GUIDANCE_H_MODE_NAV:
373  if (!in_flight) {
374  guidance_h_nav_enter();
375  }
376 
377  if (horizontal_mode == HORIZONTAL_MODE_ATTITUDE) {
378  struct Int32Eulers sp_cmd_i;
379  sp_cmd_i.phi = nav_roll;
380  sp_cmd_i.theta = nav_pitch;
384  sp_cmd_i.psi = stateGetNedToBodyEulers_i()->psi;
385  stabilization_attitude_set_rpy_setpoint_i(&sp_cmd_i);
386  } else {
387  INT32_VECT2_NED_OF_ENU(guidance_h_pos_sp, navigation_carrot);
388 
389  guidance_h_update_reference();
390 
391  /* set psi command */
392  guidance_h_heading_sp = nav_heading;
393  INT32_ANGLE_NORMALIZE(guidance_h_heading_sp);
394  /* compute x,y earth commands */
395  guidance_h_traj_run(in_flight);
396  /* set final attitude setpoint */
397  stabilization_attitude_set_earth_cmd_i(&guidance_h_cmd_earth,
398  guidance_h_heading_sp);
399  }
400  stabilization_attitude_run(in_flight);
401  break;
402 
403 #if GUIDANCE_H_MODE_MODULE_SETTING == GUIDANCE_H_MODE_MODULE
404  case GUIDANCE_H_MODE_MODULE:
405  guidance_h_module_run(in_flight);
406  break;
407 #endif
408 
409  default:
410  break;
411  }
412 }

441  VECT2_COPY(guidance_h_pos_sp, guidance_h_pos_ref); // for display only
442  }
443 #endif

447 #define MAX_POS_ERR POS_BFP_OF_REAL(16.)
448 #define MAX_SPEED_ERR SPEED_BFP_OF_REAL(16.)

450 #ifndef GUIDANCE_H_THRUST_CMD_FILTER
451 #define GUIDANCE_H_THRUST_CMD_FILTER 10
452 #endif
453 
454 /* with a pgain of 100 and a scale of 2,
455  * you get an angle of 5.6 degrees for 1m pos error */
456 #define GH_GAIN_SCALE 2

458 static void guidance_h_traj_run(bool_t in_flight)
459 {
460  /* maximum bank angle: default 20 deg, max 40 deg*/
461  static const int32_t traj_max_bank = Min(BFP_OF_REAL(GUIDANCE_H_MAX_BANK, INT32_ANGLE_FRAC),
462  BFP_OF_REAL(RadOfDeg(40), INT32_ANGLE_FRAC));
463  static const int32_t total_max_bank = BFP_OF_REAL(RadOfDeg(45), INT32_ANGLE_FRAC);
464 
465  /* compute position error */
466  VECT2_DIFF(guidance_h_pos_err, guidance_h_pos_ref, *stateGetPositionNed_i());
467  /* saturate it */
468  VECT2_STRIM(guidance_h_pos_err, -MAX_POS_ERR, MAX_POS_ERR);
469 
470  /* compute speed error */
471  VECT2_DIFF(guidance_h_speed_err, guidance_h_speed_ref, *stateGetSpeedNed_i());
472  /* saturate it */
473  VECT2_STRIM(guidance_h_speed_err, -MAX_SPEED_ERR, MAX_SPEED_ERR);
474 
475  /* run PID */
476  int32_t pd_x =
477  ((guidance_h_pgain * guidance_h_pos_err.x) >> (INT32_POS_FRAC - GH_GAIN_SCALE)) +
478  ((guidance_h_dgain * (guidance_h_speed_err.x >> 2)) >> (INT32_SPEED_FRAC - GH_GAIN_SCALE - 2));
479  int32_t pd_y =
480  ((guidance_h_pgain * guidance_h_pos_err.y) >> (INT32_POS_FRAC - GH_GAIN_SCALE)) +
481  ((guidance_h_dgain * (guidance_h_speed_err.y >> 2)) >> (INT32_SPEED_FRAC - GH_GAIN_SCALE - 2));
482  guidance_h_cmd_earth.x =
483  pd_x +
484  ((guidance_h_vgain * guidance_h_speed_ref.x) >> 17) + /* speed feedforward gain */
485  ((guidance_h_again * guidance_h_accel_ref.x) >> 8); /* acceleration feedforward gain */
486  guidance_h_cmd_earth.y =
487  pd_y +
488  ((guidance_h_vgain * guidance_h_speed_ref.y) >> 17) + /* speed feedforward gain */
489  ((guidance_h_again * guidance_h_accel_ref.y) >> 8); /* acceleration feedforward gain */
490 
491  /* trim max bank angle from PD */
492  VECT2_STRIM(guidance_h_cmd_earth, -traj_max_bank, traj_max_bank);
493 
494  /* Update pos & speed error integral, zero it if not in_flight.
495  * Integrate twice as fast when not only POS but also SPEED are wrong,
496  * but do not integrate POS errors when the SPEED is already catching up.
497  */
498  if (in_flight) {
499  /* ANGLE_FRAC (12) * GAIN (8) * LOOP_FREQ (9) -> INTEGRATOR HIGH RES ANGLE_FRAX (28) */
500  guidance_h_trim_att_integrator.x += (guidance_h_igain * pd_x);
501  guidance_h_trim_att_integrator.y += (guidance_h_igain * pd_y);
502  /* saturate it */
503  VECT2_STRIM(guidance_h_trim_att_integrator, -(traj_max_bank << 16), (traj_max_bank << 16));
504  /* add it to the command */
505  guidance_h_cmd_earth.x += (guidance_h_trim_att_integrator.x >> 16);
506  guidance_h_cmd_earth.y += (guidance_h_trim_att_integrator.y >> 16);
507  } else {
508  INT_VECT2_ZERO(guidance_h_trim_att_integrator);
509  }
510 
511  /* compute a better approximation of force commands by taking thrust into account */
512  if (guidance_h_approx_force_by_thrust && in_flight) {
513  static int32_t thrust_cmd_filt;
514  int32_t vertical_thrust = (stabilization_cmd[COMMAND_THRUST] * guidance_v_thrust_coeff) >> INT32_TRIG_FRAC;
515  thrust_cmd_filt = (thrust_cmd_filt * GUIDANCE_H_THRUST_CMD_FILTER + vertical_thrust) /
516  (GUIDANCE_H_THRUST_CMD_FILTER + 1);
517  guidance_h_cmd_earth.x = ANGLE_BFP_OF_REAL(atan2f((guidance_h_cmd_earth.x * MAX_PPRZ / INT32_ANGLE_PI_2),
518  thrust_cmd_filt));
519  guidance_h_cmd_earth.y = ANGLE_BFP_OF_REAL(atan2f((guidance_h_cmd_earth.y * MAX_PPRZ / INT32_ANGLE_PI_2),
520  thrust_cmd_filt));
521  }
522 
523  VECT2_STRIM(guidance_h_cmd_earth, -total_max_bank, total_max_bank);
524 }

548 static inline void transition_run(void)
549 {
550  //Add 0.00625%
551  transition_percentage += 1 << (INT32_PERCENTAGE_FRAC - 4);
