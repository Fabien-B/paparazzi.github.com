136 static void *viewvideo_thread(void *data __attribute__((unused)))
137 {
138  // Start the streaming of the V4L2 device
139  if (!v4l2_start_capture(viewvideo.dev)) {
140  printf("[viewvideo-thread] Could not start capture of %s.\n", viewvideo.dev->name);
141  return 0;
142  }
143 
144  // be nice to the more important stuff
145  set_nice_level(10);
146 
147  // Resize image if needed
148  struct image_t img_small;
149  image_create(&img_small,
150  viewvideo.dev->w / viewvideo.downsize_factor,
151  viewvideo.dev->h / viewvideo.downsize_factor,
152  IMAGE_YUV422);
153 
154  // Create the JPEG encoded image
155  struct image_t img_jpeg;
156  image_create(&img_jpeg, img_small.w, img_small.h, IMAGE_JPEG);
157 
158  // Initialize timing
159  uint32_t microsleep = (uint32_t)(1000000. / (float)viewvideo.fps);
160  struct timeval last_time;
161  gettimeofday(&last_time, NULL);
162 
163 #if VIEWVIDEO_USE_NETCAT
164  char nc_cmd[64];
165  sprintf(nc_cmd, "nc %s %d 2>/dev/null", STRINGIFY(VIEWVIDEO_HOST), VIEWVIDEO_PORT_OUT);
166 #else
167  struct UdpSocket video_sock;
168  udp_socket_create(&video_sock, STRINGIFY(VIEWVIDEO_HOST), VIEWVIDEO_PORT_OUT, -1, VIEWVIDEO_BROADCAST);
169 #endif
170 
171  // Start streaming
172  viewvideo.is_streaming = TRUE;
173  while (viewvideo.is_streaming) {
174  // compute usleep to have a more stable frame rate
175  struct timeval vision_thread_sleep_time;
176  gettimeofday(&vision_thread_sleep_time, NULL);
177  int dt = (int)(vision_thread_sleep_time.tv_sec - last_time.tv_sec) * 1000000 + (int)(vision_thread_sleep_time.tv_usec - last_time.tv_usec);
178  if (dt < microsleep) { usleep(microsleep - dt); }
179  last_time = vision_thread_sleep_time;
180 
181  // Wait for a new frame (blocking)
182  struct image_t img;
183  v4l2_image_get(viewvideo.dev, &img);
184 
185  // Check if we need to take a shot
186  if (viewvideo.take_shot) {
187  // Create a high quality image (99% JPEG encoded)
188  struct image_t jpeg_hr;
189  image_create(&jpeg_hr, img.w, img.h, IMAGE_JPEG);
190  jpeg_encode_image(&img, &jpeg_hr, 99, TRUE);
191 
192  // Search for a file where we can write to
193  char save_name[128];
194  for (; viewvideo.shot_number < 99999; viewvideo.shot_number++) {
195  sprintf(save_name, "%s/img_%05d.jpg", STRINGIFY(VIEWVIDEO_SHOT_PATH), viewvideo.shot_number);
196  // Check if file exists or not
197  if (access(save_name, F_OK) == -1) {
198  FILE *fp = fopen(save_name, "w");
199  if (fp == NULL) {
200  printf("[viewvideo-thread] Could not write shot %s.\n", save_name);
201  } else {
202  // Save it to the file and close it
203  fwrite(jpeg_hr.buf, sizeof(uint8_t), jpeg_hr.buf_size, fp);
204  fclose(fp);
205  }
206 
207  // We don't need to seek for a next index anymore
208  break;
209  }
210  }
211 
212  // We finished the shot
213  image_free(&jpeg_hr);
214  viewvideo.take_shot = FALSE;
215  }
216 
217  // Only resize when needed
218  if (viewvideo.downsize_factor != 1) {
219  image_yuv422_downsample(&img, &img_small, viewvideo.downsize_factor);
220  jpeg_encode_image(&img_small, &img_jpeg, VIEWVIDEO_QUALITY_FACTOR, VIEWVIDEO_USE_NETCAT);
221  } else {
222  jpeg_encode_image(&img, &img_jpeg, VIEWVIDEO_QUALITY_FACTOR, VIEWVIDEO_USE_NETCAT);
223  }
224 
225 #if VIEWVIDEO_USE_NETCAT
226  // Open process to send using netcat (in a fork because sometimes kills itself???)
227  pid_t pid = fork();
228 
229  if (pid < 0) {
230  printf("[viewvideo] Could not create netcat fork.\n");
231  } else if (pid == 0) {
232  // We are the child and want to send the image
233  FILE *netcat = popen(nc_cmd, "w");
234  if (netcat != NULL) {
235  fwrite(jpegbuf, sizeof(uint8_t), size, netcat);
236  pclose(netcat); // Ignore output, because it is too much when not connected
237  } else {
238  printf("[viewvideo] Failed to open netcat process.\n");
239  }
240 
241  // Exit the program since we don't want to continue after transmitting
242  exit(0);
243  } else {
244  // We want to wait until the child is finished
245  wait(NULL);
246  }
247 #else
248  // Send image with RTP
249  rtp_frame_send(
250  &video_sock, // UDP socket
251  &img_jpeg,
252  0, // Format 422
253  VIEWVIDEO_QUALITY_FACTOR, // Jpeg-Quality
254  0, // DRI Header
255  VIEWVIDEO_RTP_TIME_INC // 90kHz time increment
256  );
257  // Extra note: when the time increment is set to 0,
258  // it is automaticaly calculated by the send_rtp_frame function
259  // based on gettimeofday value. This seems to introduce some lag or jitter.
260  // An other way is to compute the time increment and set the correct value.
261  // It seems that a lower value is also working (when the frame is received
262  // the timestamp is always "late" so the frame is displayed immediately).
263  // Here, we set the time increment to the lowest possible value
264  // (1 = 1/90000 s) which is probably stupid but is actually working.
265 #endif
266 
267  // Free the image
268  v4l2_image_free(viewvideo.dev, &img);
269  }
270 
271  // Free all buffers
272  image_free(&img_jpeg);
273  image_free(&img_small);
274  return 0;
275 }

280 void viewvideo_init(void)
281 {
282 #ifdef VIEWVIDEO_SUBDEV
283  PRINT_CONFIG_MSG("[viewvideo] Configuring a subdevice!")
284  PRINT_CONFIG_VAR(VIEWVIDEO_SUBDEV)
285 
286  // Initialize the V4L2 subdevice (TODO: fix hardcoded path, which and code)
287  if (!v4l2_init_subdev(STRINGIFY(VIEWVIDEO_SUBDEV), 0, 1, V4L2_MBUS_FMT_UYVY8_2X8, VIEWVIDEO_DEVICE_SIZE)) {
288  printf("[viewvideo] Could not initialize the %s subdevice.\n", STRINGIFY(VIEWVIDEO_SUBDEV));
289  return;
290  }
291 #endif
292 
293  // Initialize the V4L2 device
294  viewvideo.dev = v4l2_init(STRINGIFY(VIEWVIDEO_DEVICE), VIEWVIDEO_DEVICE_SIZE, VIEWVIDEO_DEVICE_BUFFERS);
295  if (viewvideo.dev == NULL) {
296  printf("[viewvideo] Could not initialize the %s V4L2 device.\n", STRINGIFY(VIEWVIDEO_DEVICE));
297  return;
298  }
299 
300  // Create the shot directory
301  char save_name[128];
302  sprintf(save_name, "mkdir -p %s", STRINGIFY(VIEWVIDEO_SHOT_PATH));
303  if (system(save_name) != 0) {
304  printf("[viewvideo] Could not create shot directory %s.\n", STRINGIFY(VIEWVIDEO_SHOT_PATH));
305  return;
306  }
307 
308 #if VIEWVIDEO_USE_NETCAT
309  // Create an Netcat receiver file for the streaming
310  sprintf(save_name, "%s/netcat-recv.sh", STRINGIFY(VIEWVIDEO_SHOT_PATH));
311  FILE *fp = fopen(save_name, "w");
312  if (fp != NULL) {
313  fprintf(fp, "i=0\n");
314  fprintf(fp, "while true\n");
315  fprintf(fp, "do\n");
316  fprintf(fp, "\tn=$(printf \"%%04d\" $i)\n");
317  fprintf(fp, "\tnc -l 0.0.0.0 %d > img_${n}.jpg\n", (int)(VIEWVIDEO_PORT_OUT));
318  fprintf(fp, "\ti=$((i+1))\n");
319  fprintf(fp, "done\n");
320  fclose(fp);
321  }
322 #else
323  // Create an SDP file for the streaming
324  sprintf(save_name, "%s/stream.sdp", STRINGIFY(VIEWVIDEO_SHOT_PATH));
325  FILE *fp = fopen(save_name, "w");
326  if (fp != NULL) {
327  fprintf(fp, "v=0\n");
328  fprintf(fp, "m=video %d RTP/AVP 26\n", (int)(VIEWVIDEO_PORT_OUT));
329  fprintf(fp, "c=IN IP4 0.0.0.0\n");
330  fclose(fp);
331  }
332 #endif
333 }
