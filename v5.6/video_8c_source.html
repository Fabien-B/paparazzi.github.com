36 static bool_t write_reg(int fd, char *addr_val, uint8_t cnt)
37 {
38  char resp[cnt - 2];
39  uint8_t i;
40 
41  if (write(fd, addr_val, cnt) != cnt) {
42  printf("Write failed!\n");
43  return FALSE;
44  }
45  if (write(fd, addr_val, 2) != 2) {
46  printf("Write2 failed!\n");
47  return FALSE;
48  }
49  while (read(fd, resp, cnt - 2) != cnt - 2) { ; }
50  for (i = 0; i < cnt - 2; i++) {
51  if (resp[i] != addr_val[i + 2]) {
52  printf("[video] Could not write register %X%X\n", addr_val[0], addr_val[1]);
53  return write_reg(fd, addr_val, cnt);
54  }
55  }
56  return TRUE;
57 }
