130 void stabilization_opticflow_update(struct opticflow_result_t *result)
131 {
132  /* Check if we are in the correct AP_MODE before setting commands */
133  if (autopilot_mode != AP_MODE_MODULE) {
134  return;
135  }
136 
137  /* Calculate the error if we have enough flow */
138  float err_vx = 0;
139  float err_vy = 0;
140  if (result->tracked_cnt > 0) {
141  err_vx = opticflow_stab.desired_vx - result->vel_x;
142  err_vy = opticflow_stab.desired_vy - result->vel_y;
143  }
144 
145  /* Calculate the integrated errors (TODO: bound??) */
146  opticflow_stab.err_vx_int += err_vx / 100;
147  opticflow_stab.err_vy_int += err_vy / 100;
148 
149  /* Calculate the commands */
150  opticflow_stab.cmd.phi = opticflow_stab.phi_pgain * err_vx / 100
151  + opticflow_stab.phi_igain * opticflow_stab.err_vx_int;
152  opticflow_stab.cmd.theta = -(opticflow_stab.theta_pgain * err_vy / 100
153  + opticflow_stab.theta_igain * opticflow_stab.err_vy_int);
154 
155  /* Bound the roll and pitch commands */
156  BoundAbs(opticflow_stab.cmd.phi, CMD_OF_SAT);
157  BoundAbs(opticflow_stab.cmd.theta, CMD_OF_SAT);
158 }
