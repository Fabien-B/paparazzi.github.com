120  gps.utm_pos.zone = nav_utm_zone0;

138  gps.ned_vel.y = UBX_NAV_VELNED_VEL_E(gps_ubx.msg_buf);

147  gps_ubx.have_velned = 1;

149  gps.nb_channels = Min(UBX_NAV_SVINFO_NCH(gps_ubx.msg_buf), GPS_NB_CHANNELS);

157  gps.svinfos[i].azim = UBX_NAV_SVINFO_Azim(gps_ubx.msg_buf, i);
158  }

161  gps_ubx.status_flags = UBX_NAV_STATUS_Flags(gps_ubx.msg_buf);
162  gps_ubx.sol_flags = UBX_NAV_SOL_Flags(gps_ubx.msg_buf);

191 /* UBX parsing */
192 void gps_ubx_parse(uint8_t c)
193 {
194 #if LOG_RAW_GPS
195  sdLogWriteByte(pprzLogFile, c);
196 #endif
197  if (gps_ubx.status < GOT_PAYLOAD) {
198  gps_ubx.ck_a += c;
199  gps_ubx.ck_b += gps_ubx.ck_a;
200  }
201  switch (gps_ubx.status) {
202  case UNINIT:
203  if (c == UBX_SYNC1) {
204  gps_ubx.status++;
205  }
206  break;
207  case GOT_SYNC1:
208  if (c != UBX_SYNC2) {
209  gps_ubx.error_last = GPS_UBX_ERR_OUT_OF_SYNC;
210  goto error;
211  }
212  gps_ubx.ck_a = 0;
213  gps_ubx.ck_b = 0;
214  gps_ubx.status++;
215  break;
216  case GOT_SYNC2:
217  if (gps_ubx.msg_available) {
218  /* Previous message has not yet been parsed: discard this one */
219  gps_ubx.error_last = GPS_UBX_ERR_OVERRUN;
220  goto error;
221  }
222  gps_ubx.msg_class = c;
223  gps_ubx.status++;
224  break;
225  case GOT_CLASS:
226  gps_ubx.msg_id = c;
227  gps_ubx.status++;
228  break;
229  case GOT_ID:
230  gps_ubx.len = c;
231  gps_ubx.status++;
232  break;
233  case GOT_LEN1:
234  gps_ubx.len |= (c << 8);
235  if (gps_ubx.len > GPS_UBX_MAX_PAYLOAD) {
236  gps_ubx.error_last = GPS_UBX_ERR_MSG_TOO_LONG;
237  goto error;
238  }
239  gps_ubx.msg_idx = 0;
240  gps_ubx.status++;
241  break;
242  case GOT_LEN2:
243  gps_ubx.msg_buf[gps_ubx.msg_idx] = c;
244  gps_ubx.msg_idx++;
245  if (gps_ubx.msg_idx >= gps_ubx.len) {
246  gps_ubx.status++;
247  }
248  break;
249  case GOT_PAYLOAD:
250  if (c != gps_ubx.ck_a) {
251  gps_ubx.error_last = GPS_UBX_ERR_CHECKSUM;
252  goto error;
253  }
254  gps_ubx.status++;
255  break;
256  case GOT_CHECKSUM1:
257  if (c != gps_ubx.ck_b) {
258  gps_ubx.error_last = GPS_UBX_ERR_CHECKSUM;
259  goto error;
260  }
261  gps_ubx.msg_available = TRUE;
262  goto restart;
263  break;
264  default:
265  gps_ubx.error_last = GPS_UBX_ERR_UNEXPECTED;
266  goto error;
267  }
268  return;
269 error:
270  gps_ubx.error_cnt++;
271 restart:
272  gps_ubx.status = UNINIT;
273  return;
274 }

291  ubx_send_1byte(dev, (uint8_t)(len&0xFF));
292  ubx_send_1byte(dev, (uint8_t)(len>>8));
293 }

323 void gps_ubx_msg(void)
324 {
325  // current timestamp
326  uint32_t now_ts = get_sys_time_usec();
327 
328  gps.last_msg_ticks = sys_time.nb_sec_rem;
329  gps.last_msg_time = sys_time.nb_sec;
330  gps_ubx_read_message();
331  gps_ubx_ucenter_event();
332  if (gps_ubx.msg_class == UBX_NAV_ID &&
333  (gps_ubx.msg_id == UBX_NAV_VELNED_ID ||
334  (gps_ubx.msg_id == UBX_NAV_SOL_ID &&
335  gps_ubx.have_velned == 0))) {
336  if (gps.fix == GPS_FIX_3D) {
337  gps.last_3dfix_ticks = sys_time.nb_sec_rem;
338  gps.last_3dfix_time = sys_time.nb_sec;
339  }
340  AbiSendMsgGPS(GPS_UBX_ID, now_ts, &gps);
341  }
342  gps_ubx.msg_available = FALSE;
343 }
