82 static void start_message(struct xbee_transport *trans, struct link_device *dev, uint8_t payload_len)
83 {
84  downlink.nb_msgs++;
85  dev->put_byte(dev->periph, XBEE_START);
86  const uint16_t len = payload_len + XBEE_API_OVERHEAD;
87  dev->put_byte(dev->periph, (len >> 8));
88  dev->put_byte(dev->periph, (len & 0xff));
89  trans->cs_tx = 0;
90  const uint8_t header[] = XBEE_TX_HEADER;
91  put_bytes(trans, dev, DL_TYPE_UINT8, DL_FORMAT_SCALAR, XBEE_TX_OVERHEAD + 1, header);
92 }

119 static uint8_t xbee_text_reply_is_ok(struct link_device *dev)
120 {
121  char c[2];
122  int count = 0;
123 
124  while (dev->char_available(dev->periph)) {
125  char cc = dev->get_byte(dev->periph);
126  if (count < 2) {
127  c[count] = cc;
128  }
129  count++;
130  }
131 
132  if ((count > 2) && (c[0] == 'O') && (c[1] == 'K')) {
133  return TRUE;
134  }
135 
136  return FALSE;
137 }
