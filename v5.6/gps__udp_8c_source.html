55 void gps_parse(void)
56 {
57  if (gps_network == NULL) { return; }
58 
59  //Read from the network
60  int size = network_read(gps_network, &gps_udp_read_buffer[0], 256);
61 
62  if (size > 0) {
63  // Correct MSG
64  if ((size == GPS_UDP_MSG_LEN) && (gps_udp_read_buffer[0] == STX)) {
65  gps.lla_pos.lat = UDP_GPS_INT(gps_udp_read_buffer + 4);
66  gps.lla_pos.lon = UDP_GPS_INT(gps_udp_read_buffer + 8);
67  gps.lla_pos.alt = UDP_GPS_INT(gps_udp_read_buffer + 12);
68  gps.hmsl = UDP_GPS_INT(gps_udp_read_buffer + 16);
69 
70  gps.ecef_pos.x = UDP_GPS_INT(gps_udp_read_buffer + 20);
71  gps.ecef_pos.y = UDP_GPS_INT(gps_udp_read_buffer + 24);
72  gps.ecef_pos.z = UDP_GPS_INT(gps_udp_read_buffer + 28);
73 
74  gps.ecef_vel.x = UDP_GPS_INT(gps_udp_read_buffer + 32);
75  gps.ecef_vel.y = UDP_GPS_INT(gps_udp_read_buffer + 36);
76  gps.ecef_vel.z = UDP_GPS_INT(gps_udp_read_buffer + 40);
77 
78  gps.fix = GPS_FIX_3D;
79 
80 #if GPS_USE_LATLONG
81  // Computes from (lat, long) in the referenced UTM zone
82  struct LlaCoor_f lla_f;
83  LLA_FLOAT_OF_BFP(lla_f, gps.lla_pos);
84  struct UtmCoor_f utm_f;

90  gps.utm_pos.north = utm_f.north * 100;

94 
95  // publish new GPS data

97  gps.last_msg_ticks = sys_time.nb_sec_rem;
98  gps.last_msg_time = sys_time.nb_sec;
99  if (gps.fix == GPS_FIX_3D) {
100  gps.last_3dfix_ticks = sys_time.nb_sec_rem;
101  gps.last_3dfix_time = sys_time.nb_sec;
102  }
103  AbiSendMsgGPS(GPS_UDP_ID, now_ts, &gps);
104 
105  } else {
106  printf("gps_udp error: msg len invalid %d bytes\n", size);
107  }
108  memset(&gps_udp_read_buffer[0], 0, sizeof(gps_udp_read_buffer));
109  }
110 }
