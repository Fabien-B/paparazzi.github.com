186 void stabilization_rate_run(bool_t in_flight)
187 {
188  /* compute feed-back command */
189  struct Int32Rates _error;
190  struct Int32Rates *body_rate = stateGetBodyRates_i();
191  RATES_DIFF(_error, stabilization_rate_sp, (*body_rate));
192  if (in_flight) {
193  /* update integrator */
194  RATES_ADD(stabilization_rate_sum_err, _error);
195  RATES_BOUND_CUBE(stabilization_rate_sum_err, -MAX_SUM_ERR, MAX_SUM_ERR);
196  } else {
197  INT_RATES_ZERO(stabilization_rate_sum_err);
198  }
199 
200  /* PI */
201  stabilization_rate_fb_cmd.p = stabilization_rate_gain.p * _error.p +
202  OFFSET_AND_ROUND2((stabilization_rate_igain.p * stabilization_rate_sum_err.p), 10);
203 
204  stabilization_rate_fb_cmd.q = stabilization_rate_gain.q * _error.q +
205  OFFSET_AND_ROUND2((stabilization_rate_igain.q * stabilization_rate_sum_err.q), 10);
206 
207  stabilization_rate_fb_cmd.r = stabilization_rate_gain.r * _error.r +
208  OFFSET_AND_ROUND2((stabilization_rate_igain.r * stabilization_rate_sum_err.r), 10);
209 
210  stabilization_cmd[COMMAND_ROLL] = stabilization_rate_fb_cmd.p >> 11;
211  stabilization_cmd[COMMAND_PITCH] = stabilization_rate_fb_cmd.q >> 11;
212  stabilization_cmd[COMMAND_YAW] = stabilization_rate_fb_cmd.r >> 11;
213 
214  /* bound the result */
215  BoundAbs(stabilization_cmd[COMMAND_ROLL], MAX_PPRZ);
216  BoundAbs(stabilization_cmd[COMMAND_PITCH], MAX_PPRZ);
217  BoundAbs(stabilization_cmd[COMMAND_YAW], MAX_PPRZ);
218 
219 }
