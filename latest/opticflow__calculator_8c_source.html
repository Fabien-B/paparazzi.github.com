155 void opticflow_calc_frame(struct opticflow_t *opticflow, struct opticflow_state_t *state, struct image_t *img, struct opticflow_result_t *result)
156 {
157  // variables for size_divergence:
158  float size_divergence; int n_samples;
159 
160  // variables for linear flow fit:
161  float error_threshold; int n_iterations_RANSAC, n_samples_RANSAC, success_fit; struct linear_flow_fit_info fit_info;
162 
163  // Update FPS for information
164  result->fps = 1 / (timeval_diff(&opticflow->prev_timestamp, &img->ts) / 1000.);
165  memcpy(&opticflow->prev_timestamp, &img->ts, sizeof(struct timeval));
166 
167  // Convert image to grayscale
168  image_to_grayscale(img, &opticflow->img_gray);
169 
170  // Copy to previous image if not set
171  if (!opticflow->got_first_img) {
172  image_copy(&opticflow->img_gray, &opticflow->prev_img_gray);
173  opticflow->got_first_img = TRUE;
174  }
175 
176  // *************************************************************************************
177  // Corner detection
178  // *************************************************************************************
179 
180  // FAST corner detection (TODO: non fixed threshold)
181  struct point_t *corners = fast9_detect(img, opticflow->fast9_threshold, opticflow->fast9_min_distance,
182  20, 20, &result->corner_cnt);
183 
184  // Adaptive threshold
185  if (opticflow->fast9_adaptive) {
186 
187  // Decrease and increase the threshold based on previous values
188  if (result->corner_cnt < 40 && opticflow->fast9_threshold > 5) {
189  opticflow->fast9_threshold--;
190  } else if (result->corner_cnt > 50 && opticflow->fast9_threshold < 60) {
191  opticflow->fast9_threshold++;
192  }
193  }
194 
195 #if OPTICFLOW_DEBUG && OPTICFLOW_SHOW_CORNERS
196  image_show_points(img, corners, result->corner_cnt);
197 #endif
198 
199  // Check if we found some corners to track
200  if (result->corner_cnt < 1) {
201  free(corners);
202  image_copy(&opticflow->img_gray, &opticflow->prev_img_gray);
203  return;
204  }
205 
206  // *************************************************************************************
207  // Corner Tracking
208  // *************************************************************************************
209 
210  // Execute a Lucas Kanade optical flow
211  result->tracked_cnt = result->corner_cnt;
212  struct flow_t *vectors = opticFlowLK(&opticflow->img_gray, &opticflow->prev_img_gray, corners, &result->tracked_cnt,
213  opticflow->window_size / 2, opticflow->subpixel_factor, opticflow->max_iterations,
214  opticflow->threshold_vec, opticflow->max_track_corners);
215 
216 #if OPTICFLOW_DEBUG && OPTICFLOW_SHOW_FLOW
217  image_show_flow(img, vectors, result->tracked_cnt, opticflow->subpixel_factor);
218 #endif
219 
220  // Estimate size divergence:
221  if (SIZE_DIV) {
222  n_samples = 100;
223  size_divergence = get_size_divergence(vectors, result->tracked_cnt, n_samples);
224  result->div_size = size_divergence;
225  } else {
226  result->div_size = 0.0f;
227  }
228  if (LINEAR_FIT) {
229  // Linear flow fit (normally derotation should be performed before):
230  error_threshold = 10.0f;
231  n_iterations_RANSAC = 20;
232  n_samples_RANSAC = 5;
233  success_fit = analyze_linear_flow_field(vectors, result->tracked_cnt, error_threshold, n_iterations_RANSAC, n_samples_RANSAC, img->w, img->h, &fit_info);
234 
235  if (!success_fit) {
236  fit_info.divergence = 0.0f;
237  fit_info.surface_roughness = 0.0f;
238  }
239 
240  result->divergence = fit_info.divergence;
241  result->surface_roughness = fit_info.surface_roughness;
242  } else {
243  result->divergence = 0.0f;
244  result->surface_roughness = 0.0f;
245  }
246 
247 
248  // Get the median flow
249  qsort(vectors, result->tracked_cnt, sizeof(struct flow_t), cmp_flow);
250  if (result->tracked_cnt == 0) {
251  // We got no flow
252  result->flow_x = 0;
253  result->flow_y = 0;
254  } else if (result->tracked_cnt > 3) {
255  // Take the average of the 3 median points
256  result->flow_x = vectors[result->tracked_cnt / 2 - 1].flow_x;
257  result->flow_y = vectors[result->tracked_cnt / 2 - 1].flow_y;
258  result->flow_x += vectors[result->tracked_cnt / 2].flow_x;
259  result->flow_y += vectors[result->tracked_cnt / 2].flow_y;
260  result->flow_x += vectors[result->tracked_cnt / 2 + 1].flow_x;
261  result->flow_y += vectors[result->tracked_cnt / 2 + 1].flow_y;
262  result->flow_x /= 3;
263  result->flow_y /= 3;
264  } else {
265  // Take the median point
266  result->flow_x = vectors[result->tracked_cnt / 2].flow_x;
267  result->flow_y = vectors[result->tracked_cnt / 2].flow_y;
268  }
269 
270  // Flow Derotation
271  float diff_flow_x = (state->phi - opticflow->prev_phi) * img->w / OPTICFLOW_FOV_W;
272  float diff_flow_y = (state->theta - opticflow->prev_theta) * img->h / OPTICFLOW_FOV_H;
273  result->flow_der_x = result->flow_x - diff_flow_x * opticflow->subpixel_factor;
274  result->flow_der_y = result->flow_y - diff_flow_y * opticflow->subpixel_factor;
275  opticflow->prev_phi = state->phi;
276  opticflow->prev_theta = state->theta;
277 
278  // Velocity calculation
279  result->vel_x = -result->flow_der_x * result->fps * state->agl / opticflow->subpixel_factor * img->w / OPTICFLOW_FX;
280  result->vel_y = result->flow_der_y * result->fps * state->agl / opticflow->subpixel_factor * img->h / OPTICFLOW_FY;
281 
282  // *************************************************************************************
283  // Next Loop Preparation
284  // *************************************************************************************
285  free(corners);
286  free(vectors);
287  image_switch(&opticflow->img_gray, &opticflow->prev_img_gray);
288 }

296 static uint32_t timeval_diff(struct timeval *starttime, struct timeval *finishtime)
297 {
298  uint32_t msec;
299  msec = (finishtime->tv_sec - starttime->tv_sec) * 1000;
300  msec += (finishtime->tv_usec - starttime->tv_usec) / 1000;
301  return msec;
302 }

311 static int cmp_flow(const void *a, const void *b)
312 {
313  const struct flow_t *a_p = (const struct flow_t *)a;
314  const struct flow_t *b_p = (const struct flow_t *)b;
315  return (a_p->flow_x * a_p->flow_x + a_p->flow_y * a_p->flow_y) - (b_p->flow_x * b_p->flow_x + b_p->flow_y * b_p->flow_y);
316 }
