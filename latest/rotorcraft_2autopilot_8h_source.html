139 static inline void DetectGroundEvent(void)
140 {
141  if (autopilot_mode == AP_MODE_FAILSAFE || autopilot_detect_ground_once) {
142  struct NedCoor_f *accel = stateGetAccelNed_f();
143  if (accel->z < -THRESHOLD_GROUND_DETECT ||
144  accel->z > THRESHOLD_GROUND_DETECT) {
145  autopilot_ground_detected = TRUE;
146  autopilot_detect_ground_once = FALSE;
147  }
148  }
149 }
