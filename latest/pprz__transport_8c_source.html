82 static void start_message(struct pprz_transport *trans, struct link_device *dev, uint8_t payload_len)
83 {
84  downlink.nb_msgs++;
85  dev->put_byte(dev->periph, STX);
86  const uint8_t msg_len = size_of(trans, payload_len);
87  dev->put_byte(dev->periph, msg_len);
88  trans->ck_a_tx = msg_len;
89  trans->ck_b_tx = msg_len;
90 }
