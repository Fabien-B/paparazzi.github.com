111 static void gyro_cb(uint8_t __attribute__((unused)) sender_id,
112  uint32_t stamp, struct Int32Rates *gyro)
113 {
114  ahrs_fc_last_stamp = stamp;
115 #if USE_AUTO_AHRS_FREQ || !defined(AHRS_PROPAGATE_FREQUENCY)
116  PRINT_CONFIG_MSG("Calculating dt for AHRS_FC propagation.")
117  /* timestamp in usec when last callback was received */
118  static uint32_t last_stamp = 0;
119 
120  if (last_stamp > 0 && ahrs_fc.is_aligned) {
121  float dt = (float)(stamp - last_stamp) * 1e-6;
122  ahrs_fc_propagate(gyro, dt);
123  compute_body_orientation_and_rates();
124  }
125  last_stamp = stamp;
126 #else
127  PRINT_CONFIG_MSG("Using fixed AHRS_PROPAGATE_FREQUENCY for AHRS_FC propagation.")
128  PRINT_CONFIG_VAR(AHRS_PROPAGATE_FREQUENCY)
129  if (ahrs_fc.status == AHRS_FC_RUNNING) {
130  const float dt = 1. / (AHRS_PROPAGATE_FREQUENCY);
131  ahrs_fc_propagate(gyro, dt);
132  compute_body_orientation_and_rates();
133  }
134 #endif
135 }

221 static void compute_body_orientation_and_rates(void)
222 {
223  if (ahrs_fc_output_enabled) {
224  /* Compute LTP to BODY quaternion */
225  struct FloatQuat ltp_to_body_quat;
226  struct FloatQuat *body_to_imu_quat = orientationGetQuat_f(&ahrs_fc.body_to_imu);
227  float_quat_comp_inv(&ltp_to_body_quat, &ahrs_fc.ltp_to_imu_quat, body_to_imu_quat);
228  /* Set state */
229  stateSetNedToBodyQuat_f(&ltp_to_body_quat);
230 
231  /* compute body rates */
232  struct FloatRates body_rate;
233  struct FloatRMat *body_to_imu_rmat = orientationGetRMat_f(&ahrs_fc.body_to_imu);
234  float_rmat_transp_ratemult(&body_rate, body_to_imu_rmat, &ahrs_fc.imu_rate);
235  stateSetBodyRates_f(&body_rate);
236  }
