99 /*
100  * Callback functions to interpret SBP messages.
101  * Every message ID has a callback associated with it to
102  * receive and interpret the message payload.
103  */
104 static void sbp_pos_ecef_callback(uint16_t sender_id __attribute__((unused)),
105  uint8_t len __attribute__((unused)),
106  uint8_t msg[],
107  void *context __attribute__((unused)))
108 {
109  static uint8_t last_flags = 0;
110  msg_pos_ecef_t pos_ecef = *(msg_pos_ecef_t *)msg;
111 
112  // Check if we got RTK fix (FIXME when libsbp has a nicer way of doing this)
113  if(pos_ecef.flags > 0 ){//|| last_flags == 0) {
114  gps.ecef_pos.x = (int32_t)(pos_ecef.x * 100.0);
115  gps.ecef_pos.y = (int32_t)(pos_ecef.y * 100.0);
116  gps.ecef_pos.z = (int32_t)(pos_ecef.z * 100.0);
117  gps.pacc = (uint32_t)(pos_ecef.accuracy);// FIXME not implemented yet by libswiftnav
118  gps.num_sv = pos_ecef.n_sats;
119  gps.tow = pos_ecef.tow;
120 
121  if(pos_ecef.flags == 1)
122  gps.fix = GPS_FIX_RTK;
123  else if(pos_ecef.flags == 2)
124  gps.fix = GPS_FIX_DGPS;
125  else
126  gps.fix = GPS_FIX_3D;
127  }
128  last_flags = pos_ecef.flags;
129 
130  if(pos_ecef.flags > 0) gps_piksi_publish(); // Only if RTK position
131 }

148 static void sbp_pos_llh_callback(uint16_t sender_id __attribute__((unused)),
149  uint8_t len __attribute__((unused)),
150  uint8_t msg[],
151  void *context __attribute__((unused)))
152 {
153  static uint8_t last_flags = 0;
154  msg_pos_llh_t pos_llh = *(msg_pos_llh_t *)msg;
155 
156  // Check if we got RTK fix (FIXME when libsbp has a nicer way of doing this)
157  if(pos_llh.flags > 0 || last_flags == 0) {
158  gps.lla_pos.lat = (int32_t)(pos_llh.lat * 1e7);
159  gps.lla_pos.lon = (int32_t)(pos_llh.lon * 1e7);
160  int32_t alt = (int32_t)(pos_llh.height * 1000.);
161 #if GPS_USE_LATLONG
162  /* Computes from (lat, long) in the referenced UTM zone */
163  struct LlaCoor_f lla_f;
164  LLA_FLOAT_OF_BFP(lla_f, gps.lla_pos);
165  struct UtmCoor_f utm_f;
166  utm_f.zone = nav_utm_zone0;
167  /* convert to utm */
168  utm_of_lla_f(&utm_f, &lla_f);
169  /* copy results of utm conversion */
170  gps.utm_pos.east = utm_f.east * 100;
171  gps.utm_pos.north = utm_f.north * 100;
172  gps.utm_pos.alt = gps.lla_pos.alt;
173  gps.utm_pos.zone = nav_utm_zone0;
174  // height is above ellipsoid or MSL according to bit flag (but not both are available)
175  // 0: above ellipsoid
176  // 1: above MSL
177  // we have to get the HMSL from the flight plan for now
178  if (bit_is_set(pos_llh.flags, 3)) {
179  gps.hmsl = alt;
180  gps.lla_pos.alt = alt + NAV_MSL0;
181  } else {
182  gps.lla_pos.alt = alt;
183  gps.hmsl = alt - NAV_MSL0;
184  }
185 #else
186  // but here we fill the two alt with the same value since we don't know HMSL
187  gps.lla_pos.alt = alt;
188  gps.hmsl = alt;
189 #endif
190  }
191  last_flags = pos_llh.flags;
192 }

270  sbp_register_callback(&sbp_state, SBP_MSG_VEL_ECEF, &sbp_vel_ecef_callback, NULL, &vel_ecef_node);
271  sbp_register_callback(&sbp_state, SBP_MSG_POS_LLH, &sbp_pos_llh_callback, NULL, &pos_llh_node);
272  sbp_register_callback(&sbp_state, SBP_MSG_VEL_NED, &sbp_vel_ned_callback, NULL, &vel_ned_node);
273  sbp_register_callback(&sbp_state, SBP_MSG_DOPS, &sbp_dops_callback, NULL, &dops_node);
274  sbp_register_callback(&sbp_state, SBP_MSG_GPS_TIME, &sbp_gps_time_callback, NULL, &gps_time_node);
275  sbp_register_callback(&sbp_state, SBP_MSG_TRACKING_STATE, &sbp_tracking_state_callback, NULL, &tracking_state_node);
276  sbp_register_callback(&sbp_state, SBP_MSG_TRACKING_STATE_DEP_A, &sbp_tracking_state_dep_a_callback, NULL, &tracking_state_dep_a_node);

278  /* Write settings */

306 static void gps_piksi_publish(void)
307 {
308  // current timestamp
309  uint32_t now_ts = get_sys_time_usec();
310 
311  gps.last_msg_ticks = sys_time.nb_sec_rem;
312  gps.last_msg_time = sys_time.nb_sec;

315  gps.last_3dfix_time = sys_time.nb_sec;
316  }
317  AbiSendMsgGPS(GPS_PIKSI_ID, now_ts, &gps);
318 }

320 /*
321  * Read bytes from the Piksi UART connection
322  * This is a wrapper functions used in the libsbp library
323  */
324 uint32_t gps_piksi_read(uint8_t *buff, uint32_t n, void *context __attribute__((unused)))
325 {
326  uint32_t i;
327  for (i = 0; i < n; i++) {
328  if (!uart_char_available(&(GPS_LINK)))
329  break;
330 
331  buff[i] = uart_getch(&(GPS_LINK));
332  }
333  return i;
334 }

341 {

343  for (i = 0; i < n; i++) {
344  uart_put_byte(&(GPS_LINK), buff[i]);
