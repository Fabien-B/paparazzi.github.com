124 static void sbp_pos_llh_callback(uint16_t sender_id __attribute__((unused)),
125  uint8_t len __attribute__((unused)),
126  uint8_t msg[],
127  void *context __attribute__((unused)))
128 {
129  sbp_pos_llh_t pos_llh = *(sbp_pos_llh_t *)msg;
130  gps.lla_pos.lat = (int32_t)(pos_llh.lat * 1e7);
131  gps.lla_pos.lon = (int32_t)(pos_llh.lon * 1e7);
132  int32_t alt = (int32_t)(pos_llh.height * 1000.);
133 #if GPS_USE_LATLONG
134  /* Computes from (lat, long) in the referenced UTM zone */
135  struct LlaCoor_f lla_f;
136  LLA_FLOAT_OF_BFP(lla_f, gps.lla_pos);
137  struct UtmCoor_f utm_f;
138  utm_f.zone = nav_utm_zone0;
139  /* convert to utm */
140  utm_of_lla_f(&utm_f, &lla_f);
141  /* copy results of utm conversion */
142  gps.utm_pos.east = utm_f.east * 100;
143  gps.utm_pos.north = utm_f.north * 100;
144  gps.utm_pos.alt = gps.lla_pos.alt;
145  gps.utm_pos.zone = nav_utm_zone0;
146  // height is above ellipsoid or MSL according to bit flag (but not both are available)
147  // 0: above ellipsoid
148  // 1: above MSL
149  // we have to get the HMSL from the flight plan for now
150  if (bit_is_set(pos_llh.flags, 3)) {
151  gps.hmsl = alt;
152  gps.lla_pos.alt = alt + NAV_MSL0;
153  } else {
154  gps.lla_pos.alt = alt;
155  gps.hmsl = alt - NAV_MSL0;
156  }
157 #else
158  // but here we fill the two alt with the same value since we don't know HMSL
159  gps.lla_pos.alt = alt;
160  gps.hmsl = alt;
161 #endif
162 }

247 /*
248  * Event function
249  */
250 void gps_piksi_event(void)
251 {
252  // fill fifo with new uart bytes
253  while (uart_char_available(&(GPS_LINK))) {
254  uint8_t c = uart_getch(&(GPS_LINK));
255  fifo_write(c);
256  }
257  // call sbp event function
258  sbp_process(&sbp_state, &fifo_read);
259 }

264 uint16_t head = 0;

267 /* Return 1 if true, 0 otherwise. */
268 uint8_t fifo_empty(void)
269 {
270  if (head == tail) {
271  return 1;
272  }
273  return 0;
274 }

276 /*
277  * Append a character to our SBP message fifo.
278  * Returns 1 if char successfully appended to fifo.
279  * Returns 0 if fifo is full.
280  */
281 uint8_t fifo_write(char c)
282 {
283  if (fifo_full()) {
284  return 0;
285  }
286  sbp_msg_fifo[tail] = c;
287  tail = (tail + 1) % FIFO_LEN;
288  return 1;
289 }
290 
291 /*
292  * Read 1 char from fifo.
293  * Returns 0 if fifo is empty, otherwise 1.
294  */
295 uint8_t fifo_read_char(char *c)
296 {
297  if (fifo_empty()) {
298  return 0;
299  }
300  *c = sbp_msg_fifo[head];
301  head = (head + 1) % FIFO_LEN;
302  return 1;
303 }
304 
305 /*
306  * Read arbitrary number of chars from FIFO. Must conform to
307  * function definition that is passed to the function
308  * sbp_process().
309  * Returns the number of characters successfully read.
310  */
311 uint32_t fifo_read(uint8_t *buff, uint32_t n, void *context __attribute__((unused)))
312 {
313  uint32_t i;
314  for (i = 0; i < n; i++) {
315  if (!fifo_read_char((char *)(buff + i))) {
316  break;
317  }
318  }
319  return i;
320 }
321 
322 /* Return 1 if true, 0 otherwise. */
323 uint8_t fifo_full(void)
324 {
325  if (((tail + 1) % FIFO_LEN) == head) {
326  return 1;
327  }
328  return 0;
329 }
