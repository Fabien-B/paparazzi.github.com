134 static void gyro_cb(uint8_t __attribute__((unused)) sender_id,
135  uint32_t stamp, struct Int32Rates *gyro)
136 {
137  ahrs_icq_last_stamp = stamp;
138 #if USE_AUTO_AHRS_FREQ || !defined(AHRS_PROPAGATE_FREQUENCY)
139  PRINT_CONFIG_MSG("Calculating dt for AHRS_ICQ propagation.")
140  /* timestamp in usec when last callback was received */
141  static uint32_t last_stamp = 0;
142 
143  if (last_stamp > 0 && ahrs_icq.is_aligned) {
144  float dt = (float)(stamp - last_stamp) * 1e-6;
145  ahrs_icq_propagate(gyro, dt);
146  set_body_state_from_quat();
147  }
148  last_stamp = stamp;
149 #else
150  PRINT_CONFIG_MSG("Using fixed AHRS_PROPAGATE_FREQUENCY for AHRS_ICQ propagation.")
151  PRINT_CONFIG_VAR(AHRS_PROPAGATE_FREQUENCY)
152  if (ahrs_icq.status == AHRS_ICQ_RUNNING) {
153  const float dt = 1. / (AHRS_PROPAGATE_FREQUENCY);
154  ahrs_icq_propagate(gyro, dt);
155  set_body_state_from_quat();
156  }
157 #endif
158 }

246 static void set_body_state_from_quat(void)
247 {
248  if (ahrs_icq_output_enabled) {
249  /* Compute LTP to BODY quaternion */
250  struct Int32Quat ltp_to_body_quat;
251  struct Int32Quat *body_to_imu_quat = orientationGetQuat_i(&ahrs_icq.body_to_imu);
252  int32_quat_comp_inv(&ltp_to_body_quat, &ahrs_icq.ltp_to_imu_quat, body_to_imu_quat);
253  /* Set state */
254  stateSetNedToBodyQuat_i(&ltp_to_body_quat);
255 
256  /* compute body rates */
257  struct Int32Rates body_rate;
258  struct Int32RMat *body_to_imu_rmat = orientationGetRMat_i(&ahrs_icq.body_to_imu);
259  int32_rmat_transp_ratemult(&body_rate, body_to_imu_rmat, &ahrs_icq.imu_rate);
260  /* Set state */
261  stateSetBodyRates_i(&body_rate);
262  }
