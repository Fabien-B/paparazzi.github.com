111  max1168_status = MAX1168_IDLE;
112  imu_hbmini.gyr_valid = TRUE;
113  imu_hbmini.acc_valid = TRUE;
114  }
115 
116  // HMC58XX event task
117  hmc58xx_event(&imu_hbmini.hmc);
118  if (imu_hbmini.hmc.data_available) {
119  imu.mag_unscaled.z = imu_hbmini.hmc.data.value[IMU_MAG_X_CHAN];
120  imu.mag_unscaled.y = imu_hbmini.hmc.data.value[IMU_MAG_Y_CHAN];
121  imu.mag_unscaled.x = imu_hbmini.hmc.data.value[IMU_MAG_Z_CHAN];
122 
123  imu_hbmini.hmc.data_available = FALSE;
124  imu_hbmini.mag_valid = TRUE;
125  }
