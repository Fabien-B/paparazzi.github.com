216 void guidance_h_mode_changed(uint8_t new_mode)
217 {
218  if (new_mode == guidance_h.mode) {
219  return;
220  }
221 
222  if (new_mode != GUIDANCE_H_MODE_FORWARD && new_mode != GUIDANCE_H_MODE_RATE) {
223  transition_percentage = 0;
224  transition_theta_offset = 0;
225  }
226 
227  switch (new_mode) {
228  case GUIDANCE_H_MODE_RC_DIRECT:
229  stabilization_none_enter();
230  break;
231 
232  case GUIDANCE_H_MODE_RATE:
233  stabilization_rate_enter();
234  break;
235 
236  case GUIDANCE_H_MODE_CARE_FREE:
237  stabilization_attitude_reset_care_free_heading();
238  case GUIDANCE_H_MODE_FORWARD:
239  case GUIDANCE_H_MODE_ATTITUDE:
240 #if NO_ATTITUDE_RESET_ON_MODE_CHANGE
241  /* reset attitude stabilization if previous mode was not using it */
242  if (guidance_h.mode == GUIDANCE_H_MODE_KILL ||
243  guidance_h.mode == GUIDANCE_H_MODE_RATE ||
244  guidance_h.mode == GUIDANCE_H_MODE_RC_DIRECT)
245 #endif
246  stabilization_attitude_enter();
247  break;
248 
249  case GUIDANCE_H_MODE_HOVER:
250 #if GUIDANCE_INDI
251  guidance_indi_enter();
252 #endif
253  guidance_h_hover_enter();
254 #if NO_ATTITUDE_RESET_ON_MODE_CHANGE
255  /* reset attitude stabilization if previous mode was not using it */
256  if (guidance_h.mode == GUIDANCE_H_MODE_KILL ||
257  guidance_h.mode == GUIDANCE_H_MODE_RATE ||
258  guidance_h.mode == GUIDANCE_H_MODE_RC_DIRECT)
259 #endif
260  stabilization_attitude_enter();
261  break;
262 
263 #if GUIDANCE_H_MODE_MODULE_SETTING == GUIDANCE_H_MODE_MODULE
264  case GUIDANCE_H_MODE_MODULE:
265  guidance_h_module_enter();
266  break;
267 #endif
268 
269  case GUIDANCE_H_MODE_NAV:
270  guidance_h_nav_enter();
271 #if NO_ATTITUDE_RESET_ON_MODE_CHANGE
272  /* reset attitude stabilization if previous mode was not using it */
273  if (guidance_h.mode == GUIDANCE_H_MODE_KILL ||
274  guidance_h.mode == GUIDANCE_H_MODE_RATE ||
275  guidance_h.mode == GUIDANCE_H_MODE_RC_DIRECT)
276 #endif
277  stabilization_attitude_enter();
278  break;
279 
280  case GUIDANCE_H_MODE_FLIP:
281  guidance_flip_enter();
282  break;
283 
284  default:
285  break;
286  }
287 
288  guidance_h.mode = new_mode;
289

347 void guidance_h_run(bool_t in_flight)
348 {
349  switch (guidance_h.mode) {
350 
351  case GUIDANCE_H_MODE_RC_DIRECT:
352  stabilization_none_run(in_flight);
353  break;
354 
355  case GUIDANCE_H_MODE_RATE:
356  stabilization_rate_run(in_flight);
357  break;
358 
359  case GUIDANCE_H_MODE_FORWARD:
360  if (transition_percentage < (100 << INT32_PERCENTAGE_FRAC)) {
361  transition_run();
362  }
363  case GUIDANCE_H_MODE_CARE_FREE:
364  case GUIDANCE_H_MODE_ATTITUDE:
365  stabilization_attitude_run(in_flight);
366  break;
367 
368  case GUIDANCE_H_MODE_HOVER:
369  if (!in_flight) {
370  guidance_h_hover_enter();
371  }
372 
373  guidance_h_update_reference();
374 
375  /* set psi command */
376  guidance_h.sp.heading = guidance_h.rc_sp.psi;
377 
378 #if GUIDANCE_INDI
379  guidance_indi_run(in_flight, guidance_h.sp.heading);
380 #else
381  /* compute x,y earth commands */
382  guidance_h_traj_run(in_flight);
383  /* set final attitude setpoint */
384  stabilization_attitude_set_earth_cmd_i(&guidance_h_cmd_earth,
385  guidance_h.sp.heading);
386 #endif
387  stabilization_attitude_run(in_flight);
388  break;
389 
390  case GUIDANCE_H_MODE_NAV:
391  if (!in_flight) {
392  guidance_h_nav_enter();
393  }
394 
395  if (horizontal_mode == HORIZONTAL_MODE_ATTITUDE) {
396  struct Int32Eulers sp_cmd_i;
397  sp_cmd_i.phi = nav_roll;
398  sp_cmd_i.theta = nav_pitch;
399  sp_cmd_i.psi = nav_heading;
400  stabilization_attitude_set_rpy_setpoint_i(&sp_cmd_i);
401  } else {
402  INT32_VECT2_NED_OF_ENU(guidance_h.sp.pos, navigation_carrot);
403 
404  guidance_h_update_reference();
405 
406  /* set psi command */
407  guidance_h.sp.heading = nav_heading;
408  INT32_ANGLE_NORMALIZE(guidance_h.sp.heading);
409 #if GUIDANCE_INDI
410  guidance_indi_run(in_flight, guidance_h.sp.heading);
411 #else
412  /* compute x,y earth commands */
413  guidance_h_traj_run(in_flight);
414  /* set final attitude setpoint */
415  stabilization_attitude_set_earth_cmd_i(&guidance_h_cmd_earth,
416  guidance_h.sp.heading);
417 #endif
418  }
419  stabilization_attitude_run(in_flight);
420  break;
421 
422 #if GUIDANCE_H_MODE_MODULE_SETTING == GUIDANCE_H_MODE_MODULE
423  case GUIDANCE_H_MODE_MODULE:
424  guidance_h_module_run(in_flight);
425  break;
426 #endif
427 
428  case GUIDANCE_H_MODE_FLIP:
429  guidance_flip_run();
430  break;
431 
432  default:
433  break;
434  }
435 }

438 static void guidance_h_update_reference(void)
439 {
440  /* compute reference even if usage temporarily disabled via guidance_h_use_ref */
441 #if GUIDANCE_H_USE_REF
442 #if GUIDANCE_H_USE_SPEED_REF
443  if (guidance_h.mode == GUIDANCE_H_MODE_HOVER) {
444  gh_update_ref_from_speed_sp(guidance_h.sp.speed);
445  } else
446 #endif
447  gh_update_ref_from_pos_sp(guidance_h.sp.pos);
448 #endif
449 
450  /* either use the reference or simply copy the pos setpoint */
451  if (guidance_h.use_ref) {
452  /* convert our reference to generic representation */
453  INT32_VECT2_RSHIFT(guidance_h.ref.pos, gh_ref.pos, (GH_POS_REF_FRAC - INT32_POS_FRAC));
454  INT32_VECT2_LSHIFT(guidance_h.ref.speed, gh_ref.speed, (INT32_SPEED_FRAC - GH_SPEED_REF_FRAC));
455  INT32_VECT2_LSHIFT(guidance_h.ref.accel, gh_ref.accel, (INT32_ACCEL_FRAC - GH_ACCEL_REF_FRAC));
456  } else {
457  VECT2_COPY(guidance_h.ref.pos, guidance_h.sp.pos);
458  INT_VECT2_ZERO(guidance_h.ref.speed);
459  INT_VECT2_ZERO(guidance_h.ref.accel);
460  }
461 
462 #if GUIDANCE_H_USE_SPEED_REF
463  if (guidance_h.mode == GUIDANCE_H_MODE_HOVER) {
464  VECT2_COPY(guidance_h.sp.pos, guidance_h.ref.pos); // for display only
465  }
466 #endif
467 }

470 #define MAX_POS_ERR POS_BFP_OF_REAL(16.)
471 #define MAX_SPEED_ERR SPEED_BFP_OF_REAL(16.)
472 
473 #ifndef GUIDANCE_H_THRUST_CMD_FILTER
474 #define GUIDANCE_H_THRUST_CMD_FILTER 10
475 #endif
476 
477 /* with a pgain of 100 and a scale of 2,
478  * you get an angle of 5.6 degrees for 1m pos error */
479 #define GH_GAIN_SCALE 2
480 
481 #if !GUIDANCE_INDI
482 static void guidance_h_traj_run(bool_t in_flight)
483 {
484  /* maximum bank angle: default 20 deg, max 40 deg*/
485  static const int32_t traj_max_bank = Min(BFP_OF_REAL(GUIDANCE_H_MAX_BANK, INT32_ANGLE_FRAC),
486  BFP_OF_REAL(RadOfDeg(40), INT32_ANGLE_FRAC));
487  static const int32_t total_max_bank = BFP_OF_REAL(RadOfDeg(45), INT32_ANGLE_FRAC);
488 
489  /* compute position error */
490  VECT2_DIFF(guidance_h_pos_err, guidance_h.ref.pos, *stateGetPositionNed_i());
491  /* saturate it */
492  VECT2_STRIM(guidance_h_pos_err, -MAX_POS_ERR, MAX_POS_ERR);
493 
494  /* compute speed error */
495  VECT2_DIFF(guidance_h_speed_err, guidance_h.ref.speed, *stateGetSpeedNed_i());
496  /* saturate it */
497  VECT2_STRIM(guidance_h_speed_err, -MAX_SPEED_ERR, MAX_SPEED_ERR);
498 
499  /* run PID */
500  int32_t pd_x =
501  ((guidance_h.gains.p * guidance_h_pos_err.x) >> (INT32_POS_FRAC - GH_GAIN_SCALE)) +
502  ((guidance_h.gains.d * (guidance_h_speed_err.x >> 2)) >> (INT32_SPEED_FRAC - GH_GAIN_SCALE - 2));
503  int32_t pd_y =
504  ((guidance_h.gains.p * guidance_h_pos_err.y) >> (INT32_POS_FRAC - GH_GAIN_SCALE)) +
505  ((guidance_h.gains.d * (guidance_h_speed_err.y >> 2)) >> (INT32_SPEED_FRAC - GH_GAIN_SCALE - 2));
506  guidance_h_cmd_earth.x = pd_x +
507  ((guidance_h.gains.v * guidance_h.ref.speed.x) >> 17) + /* speed feedforward gain */
508  ((guidance_h.gains.a * guidance_h.ref.accel.x) >> 8); /* acceleration feedforward gain */
509  guidance_h_cmd_earth.y = pd_y +
510  ((guidance_h.gains.v * guidance_h.ref.speed.y) >> 17) + /* speed feedforward gain */
511  ((guidance_h.gains.a * guidance_h.ref.accel.y) >> 8); /* acceleration feedforward gain */
512 
513  /* trim max bank angle from PD */
514  VECT2_STRIM(guidance_h_cmd_earth, -traj_max_bank, traj_max_bank);
515 
516  /* Update pos & speed error integral, zero it if not in_flight.
517  * Integrate twice as fast when not only POS but also SPEED are wrong,
518  * but do not integrate POS errors when the SPEED is already catching up.
519  */
520  if (in_flight) {
521  /* ANGLE_FRAC (12) * GAIN (8) * LOOP_FREQ (9) -> INTEGRATOR HIGH RES ANGLE_FRAX (28) */
522  guidance_h_trim_att_integrator.x += (guidance_h.gains.i * pd_x);
523  guidance_h_trim_att_integrator.y += (guidance_h.gains.i * pd_y);
524  /* saturate it */
525  VECT2_STRIM(guidance_h_trim_att_integrator, -(traj_max_bank << 16), (traj_max_bank << 16));
526  /* add it to the command */
527  guidance_h_cmd_earth.x += (guidance_h_trim_att_integrator.x >> 16);
528  guidance_h_cmd_earth.y += (guidance_h_trim_att_integrator.y >> 16);
529  } else {
530  INT_VECT2_ZERO(guidance_h_trim_att_integrator);
531  }
532 
533  /* compute a better approximation of force commands by taking thrust into account */
534  if (guidance_h.approx_force_by_thrust && in_flight) {
535  static int32_t thrust_cmd_filt;
536  int32_t vertical_thrust = (stabilization_cmd[COMMAND_THRUST] * guidance_v_thrust_coeff) >> INT32_TRIG_FRAC;
537  thrust_cmd_filt = (thrust_cmd_filt * GUIDANCE_H_THRUST_CMD_FILTER + vertical_thrust) /
538  (GUIDANCE_H_THRUST_CMD_FILTER + 1);
539  guidance_h_cmd_earth.x = ANGLE_BFP_OF_REAL(atan2f((guidance_h_cmd_earth.x * MAX_PPRZ / INT32_ANGLE_PI_2),
540  thrust_cmd_filt));
541  guidance_h_cmd_earth.y = ANGLE_BFP_OF_REAL(atan2f((guidance_h_cmd_earth.y * MAX_PPRZ / INT32_ANGLE_PI_2),
542  thrust_cmd_filt));
543  }
544 
545  VECT2_STRIM(guidance_h_cmd_earth, -total_max_bank, total_max_bank);
546 }
547 #endif

584 static void read_rc_setpoint_speed_i(struct Int32Vect2 *speed_sp, bool_t in_flight)

589  int64_t rc_y = radio_control.values[RADIO_ROLL];
590  DeadBand(rc_x, MAX_PPRZ / 20);
591  DeadBand(rc_y, MAX_PPRZ / 20);
592 
593  // convert input from MAX_PPRZ range to SPEED_BFP
594  int32_t max_speed = SPEED_BFP_OF_REAL(GUIDANCE_H_REF_MAX_SPEED);
596  //int32_t rc_norm = sqrtf(rc_x * rc_x + rc_y * rc_y);
597  //int32_t max_pprz = rc_norm * MAX_PPRZ / Max(abs(rc_x), abs(rc_y);
598  rc_x = rc_x * max_speed / MAX_PPRZ;
599  rc_y = rc_y * max_speed / MAX_PPRZ;
600 
601  /* Rotate from body to NED frame by negative psi angle */
602  int32_t psi = -stateGetNedToBodyEulers_i()->psi;
603  int32_t s_psi, c_psi;
604  PPRZ_ITRIG_SIN(s_psi, psi);
605  PPRZ_ITRIG_COS(c_psi, psi);
606  speed_sp->x = (int32_t)(((int64_t)c_psi * rc_x + (int64_t)s_psi * rc_y) >> INT32_TRIG_FRAC);
607  speed_sp->y = (int32_t)((-(int64_t)s_psi * rc_x + (int64_t)c_psi * rc_y) >> INT32_TRIG_FRAC);
608  } else {
609  speed_sp->x = 0;
610  speed_sp->y = 0;
611  }
612 }
