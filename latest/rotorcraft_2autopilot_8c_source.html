329 #define NAV_PRESCALER (PERIODIC_FREQUENCY / NAV_FREQ)
330 void autopilot_periodic(void)
331 {
332 
333  RunOnceEvery(NAV_PRESCALER, compute_dist2_to_home());
334 
335  if (autopilot_in_flight && autopilot_mode == AP_MODE_NAV) {
336  if (too_far_from_home) {
337  if (dist2_to_home > failsafe_mode_dist2) {
338  autopilot_set_mode(FAILSAFE_MODE_TOO_FAR_FROM_HOME);
339  } else {
340  autopilot_set_mode(AP_MODE_HOME);
341  }
342  }
343  }
344 
345  if (autopilot_mode == AP_MODE_HOME) {
346  RunOnceEvery(NAV_PRESCALER, nav_home());
347  } else {
348  // otherwise always call nav_periodic_task so that carrot is always updated in GCS for other modes
349  RunOnceEvery(NAV_PRESCALER, nav_periodic_task());
350  }
351 
352 
353  /* If in FAILSAFE mode and either already not in_flight anymore
354  * or just "detected" ground, go to KILL mode.
355  */
356  if (autopilot_mode == AP_MODE_FAILSAFE) {
357  if (!autopilot_in_flight) {
358  autopilot_set_mode(AP_MODE_KILL);
359  }
360 
361 #if FAILSAFE_GROUND_DETECT
362  INFO("Using FAILSAFE_GROUND_DETECT: KILL")
363  if (autopilot_ground_detected) {
364  autopilot_set_mode(AP_MODE_KILL);
365  }
366 #endif
367  }
368 
369  /* Reset ground detection _after_ running flight plan
370  */
371  if (!autopilot_in_flight) {
372  autopilot_ground_detected = FALSE;
373  autopilot_detect_ground_once = FALSE;
374  }
375 
376  /* Set fixed "failsafe" commands from airframe file if in KILL mode.
377  * If in FAILSAFE mode, run normal loops with failsafe attitude and
378  * downwards velocity setpoints.
379  */
380  if (autopilot_mode == AP_MODE_KILL) {
381  SetCommands(commands_failsafe);
382  } else {
383  guidance_v_run(autopilot_in_flight);
384  guidance_h_run(autopilot_in_flight);
385  SetRotorcraftCommands(stabilization_cmd, autopilot_in_flight, autopilot_motors_on);
386  }
387 
388 }

391 void autopilot_set_mode(uint8_t new_autopilot_mode)
392 {
393 
394  /* force startup mode (default is kill) as long as AHRS is not aligned */
395  if (!ahrs_is_aligned()) {
396  new_autopilot_mode = MODE_STARTUP;
397  }
398 
399  if (new_autopilot_mode != autopilot_mode) {
400  /* horizontal mode */
401  switch (new_autopilot_mode) {
402  case AP_MODE_FAILSAFE:
403 #ifndef KILL_AS_FAILSAFE
404  stabilization_attitude_set_failsafe_setpoint();
405  guidance_h_mode_changed(GUIDANCE_H_MODE_ATTITUDE);
406  break;
407 #endif
408  case AP_MODE_KILL:
409  autopilot_in_flight = FALSE;
410  autopilot_in_flight_counter = 0;
411  guidance_h_mode_changed(GUIDANCE_H_MODE_KILL);
412  break;
413  case AP_MODE_RC_DIRECT:
414  guidance_h_mode_changed(GUIDANCE_H_MODE_RC_DIRECT);
415  break;
416  case AP_MODE_RATE_DIRECT:
417  case AP_MODE_RATE_Z_HOLD:
418  guidance_h_mode_changed(GUIDANCE_H_MODE_RATE);
419  break;
420  case AP_MODE_ATTITUDE_RC_CLIMB:
421  case AP_MODE_ATTITUDE_DIRECT:
422  case AP_MODE_ATTITUDE_CLIMB:
423  case AP_MODE_ATTITUDE_Z_HOLD:
424  guidance_h_mode_changed(GUIDANCE_H_MODE_ATTITUDE);
425  break;
426  case AP_MODE_FORWARD:
427  guidance_h_mode_changed(GUIDANCE_H_MODE_FORWARD);
428  break;
429  case AP_MODE_CARE_FREE_DIRECT:
430  guidance_h_mode_changed(GUIDANCE_H_MODE_CARE_FREE);
431  break;
432  case AP_MODE_HOVER_DIRECT:
433  case AP_MODE_HOVER_CLIMB:
434  case AP_MODE_HOVER_Z_HOLD:
435  guidance_h_mode_changed(GUIDANCE_H_MODE_HOVER);
436  break;
437  case AP_MODE_HOME:
438  case AP_MODE_NAV:
439  guidance_h_mode_changed(GUIDANCE_H_MODE_NAV);
440  break;
441  case AP_MODE_MODULE:
442 #ifdef GUIDANCE_H_MODE_MODULE_SETTING
443  guidance_h_mode_changed(GUIDANCE_H_MODE_MODULE_SETTING);
444 #endif
445  break;
446  case AP_MODE_FLIP:
447  guidance_h_mode_changed(GUIDANCE_H_MODE_FLIP);
448  break;
449  default:
450  break;
451  }
452  /* vertical mode */
453  switch (new_autopilot_mode) {
454  case AP_MODE_FAILSAFE:
455 #ifndef KILL_AS_FAILSAFE
456  guidance_v_mode_changed(GUIDANCE_V_MODE_CLIMB);
457  guidance_v_zd_sp = SPEED_BFP_OF_REAL(FAILSAFE_DESCENT_SPEED);
458  break;
459 #endif
460  case AP_MODE_KILL:
461  autopilot_set_motors_on(FALSE);
462  stabilization_cmd[COMMAND_THRUST] = 0;
463  guidance_v_mode_changed(GUIDANCE_V_MODE_KILL);
464  break;
465  case AP_MODE_RC_DIRECT:
466  case AP_MODE_RATE_DIRECT:
467  case AP_MODE_ATTITUDE_DIRECT:
468  case AP_MODE_HOVER_DIRECT:
469  case AP_MODE_CARE_FREE_DIRECT:
470  case AP_MODE_FORWARD:
471  guidance_v_mode_changed(GUIDANCE_V_MODE_RC_DIRECT);
472  break;
473  case AP_MODE_RATE_RC_CLIMB:
474  case AP_MODE_ATTITUDE_RC_CLIMB:
475  guidance_v_mode_changed(GUIDANCE_V_MODE_RC_CLIMB);
476  break;
477  case AP_MODE_ATTITUDE_CLIMB:
478  case AP_MODE_HOVER_CLIMB:
479  guidance_v_mode_changed(GUIDANCE_V_MODE_CLIMB);
480  break;
481  case AP_MODE_RATE_Z_HOLD:
482  case AP_MODE_ATTITUDE_Z_HOLD:
483  case AP_MODE_HOVER_Z_HOLD:
484  guidance_v_mode_changed(GUIDANCE_V_MODE_HOVER);
485  break;
486  case AP_MODE_HOME:
487  case AP_MODE_NAV:
488  guidance_v_mode_changed(GUIDANCE_V_MODE_NAV);
489  break;
490  case AP_MODE_MODULE:
491 #ifdef GUIDANCE_V_MODE_MODULE_SETTING
492  guidance_v_mode_changed(GUIDANCE_V_MODE_MODULE_SETTING);
493 #endif
494  break;
495  case AP_MODE_FLIP:
496  guidance_v_mode_changed(GUIDANCE_V_MODE_FLIP);
497  break;
498  default:
499  break;
500  }
501  autopilot_mode = new_autopilot_mode;
502  }

507 void autopilot_check_in_flight(bool_t motors_on)
508 {
509  if (autopilot_in_flight) {
510  if (autopilot_in_flight_counter > 0) {
511  /* probably in_flight if thrust, speed and accel above IN_FLIGHT_MIN thresholds */
512  if ((stabilization_cmd[COMMAND_THRUST] <= AUTOPILOT_IN_FLIGHT_MIN_THRUST) &&
513  (abs(stateGetSpeedNed_f()->z) < AUTOPILOT_IN_FLIGHT_MIN_SPEED) &&
514  (abs(stateGetAccelNed_f()->z) < AUTOPILOT_IN_FLIGHT_MIN_ACCEL)) {
515  autopilot_in_flight_counter--;
516  if (autopilot_in_flight_counter == 0) {
517  autopilot_in_flight = FALSE;
518  }
519  } else { /* thrust, speed or accel not above min threshold, reset counter */
520  autopilot_in_flight_counter = AUTOPILOT_IN_FLIGHT_TIME;
521  }
522  }
523  } else { /* currently not in flight */
524  if (autopilot_in_flight_counter < AUTOPILOT_IN_FLIGHT_TIME &&
525  motors_on) {
526  /* if thrust above min threshold, assume in_flight.
527  * Don't check for velocity and acceleration above threshold here...
528  */
529  if (stabilization_cmd[COMMAND_THRUST] > AUTOPILOT_IN_FLIGHT_MIN_THRUST) {
530  autopilot_in_flight_counter++;
531  if (autopilot_in_flight_counter == AUTOPILOT_IN_FLIGHT_TIME) {
532  autopilot_in_flight = TRUE;
533  }
534  } else { /* currently not in_flight and thrust below threshold, reset counter */
535  autopilot_in_flight_counter = 0;
536  }
537  }
538  }

599 void autopilot_on_rc_frame(void)
600 {
601 
602  if (kill_switch_is_on()) {
603  autopilot_set_mode(AP_MODE_KILL);
604  } else {
605 #ifdef RADIO_AUTO_MODE
606  INFO("Using RADIO_AUTO_MODE to switch between AUTO1 and AUTO2.")
607  uint8_t new_autopilot_mode = ap_mode_of_two_switches();
608 #else
609  uint8_t new_autopilot_mode = ap_mode_of_3way_switch();
610 #endif
611 
612  /* don't enter NAV mode if GPS is lost (this also prevents mode oscillations) */
613  if (!(new_autopilot_mode == AP_MODE_NAV && GpsIsLost())) {
614  /* always allow to switch to manual */
615  if (new_autopilot_mode == MODE_MANUAL) {
616  autopilot_set_mode(new_autopilot_mode);
617  }
618  /* if in HOME mode, don't allow switching to non-manual modes */

622  || !too_far_from_home
623 #endif

626  }
627  }

631  * only allow arming if ahrs is aligned
632  */

634  autopilot_arming_check_motors_on();
635  kill_throttle = ! autopilot_motors_on;
636  }
637

639  if (autopilot_mode != AP_MODE_FAILSAFE && autopilot_mode != AP_MODE_HOME) {
640

644 #endif

650  }
