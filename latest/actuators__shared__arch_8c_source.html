69 void set_servo_timer(uint32_t timer, uint32_t period, uint8_t channels_mask) {
70  timer_reset(timer);
71 
72  /* Timer global mode:
73  * - No divider.
74  * - Alignement edge.
75  * - Direction up.
76  */
77  if ((timer == TIM9) || (timer == TIM12))
78  //There are no EDGE and DIR settings in TIM9 and TIM12
79  timer_set_mode(timer, TIM_CR1_CKD_CK_INT, 0, 0);
80  else
81  timer_set_mode(timer, TIM_CR1_CKD_CK_INT, TIM_CR1_CMS_EDGE, TIM_CR1_DIR_UP);
82 
83 
84  // TIM1, 8 and 9 use APB2 clock, all others APB1
85  // By default the PWM_BASE_FREQ is set to 1MHz thus the timer tick period is 1uS
86  if (timer != TIM1 && timer != TIM8 && timer != TIM9) {
87  timer_set_prescaler(timer, (TIMER_APB1_CLK / PWM_BASE_FREQ) - 1);
88  } else {
89  // TIM9, 1 and 8 use APB2 clock
90  timer_set_prescaler(timer, (TIMER_APB2_CLK / PWM_BASE_FREQ) - 1);
91  }
92 
93  timer_disable_preload(timer);
94 
95  timer_continuous_mode(timer);
96 
97  timer_set_period(timer, (PWM_BASE_FREQ / period) - 1);
98 
99  /* Disable outputs and configure channel if needed. */
100  if (bit_is_set(channels_mask, 0)) {
101  actuators_pwm_arch_channel_init(timer, TIM_OC1);
102  }
103  if (bit_is_set(channels_mask, 1)) {
104  actuators_pwm_arch_channel_init(timer, TIM_OC2);
105  }
106  if (bit_is_set(channels_mask, 2)) {
107  actuators_pwm_arch_channel_init(timer, TIM_OC3);
108  }
109  if (bit_is_set(channels_mask, 3)) {
110  actuators_pwm_arch_channel_init(timer, TIM_OC4);
111  }
112 
113  /*
114  * Set initial output compare values.
115  * Note: Maybe we should preload the compare registers with some sensible
116  * values before we enable the timer?
117  */
118  //timer_set_oc_value(timer, TIM_OC1, 1000);
119  //timer_set_oc_value(timer, TIM_OC2, 1000);
120  //timer_set_oc_value(timer, TIM_OC3, 1000);
121  //timer_set_oc_value(timer, TIM_OC4, 1000);
122 
123  /* -- Enable timer -- */
124  /*
125  * ARR reload enable.
126  * Note: In our case it does not matter much if we do preload or not. As it
127  * is unlikely we will want to change the frequency of the timer during

129  */
