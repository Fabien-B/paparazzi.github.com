50 bool_t nav_spiral_setup(uint8_t center_wp, uint8_t edge_wp, float radius_start, float radius_inc, float segments)
51 {
52  VECT2_COPY(nav_spiral.center, waypoints[center_wp]); // center of the helix
53  nav_spiral.center.z = waypoints[center_wp].a;
54  nav_spiral.radius_start = radius_start; // start radius of the helix
55  nav_spiral.segments = segments;
56  nav_spiral.radius_min = NAV_SPIRAL_MIN_CIRCLE_RADIUS;
57  if (nav_spiral.radius_start < nav_spiral.radius_min)
58  nav_spiral.radius_start = nav_spiral.radius_min;
59  nav_spiral.radius_increment = radius_inc; // multiplier for increasing the spiral
60 
61  struct FloatVect2 edge;
62  VECT2_DIFF(edge, waypoints[edge_wp], nav_spiral.center);
63 
64  FLOAT_VECT2_NORM(nav_spiral.radius, edge);
65 
66  // get a copy of the current position
67  struct EnuCoor_f pos_enu;
68  memcpy(&pos_enu, stateGetPositionEnu_f(), sizeof(struct EnuCoor_f));
69 
70  VECT2_DIFF(nav_spiral.trans_current, pos_enu, nav_spiral.center);
71  nav_spiral.trans_current.z = stateGetPositionUtm_f()->alt - nav_spiral.center.z;
72 
73  nav_spiral.dist_from_center = FLOAT_VECT3_NORM(nav_spiral.trans_current);
74 
75  // nav_spiral.alpha_limit denotes angle, where the radius will be increased
76  nav_spiral.alpha_limit = 2*M_PI / nav_spiral.segments;
77  //current position
78  nav_spiral.fly_from.x = stateGetPositionEnu_f()->x;
79  nav_spiral.fly_from.y = stateGetPositionEnu_f()->y;
80 
81  if(nav_spiral.dist_from_center > nav_spiral.radius)
82  nav_spiral.status = SpiralOutside;
83  return FALSE;
84 }

86 bool_t nav_spiral_run(void)
87 {
88  struct EnuCoor_f pos_enu;
89  memcpy(&pos_enu, stateGetPositionEnu_f(), sizeof(struct EnuCoor_f));
90 
91  VECT2_DIFF(nav_spiral.trans_current, pos_enu, nav_spiral.center);
92  nav_spiral.dist_from_center = FLOAT_VECT3_NORM(nav_spiral.trans_current);
93 
94  float DistanceStartEstim;
95  float CircleAlpha;
96 
97  switch(nav_spiral.status)
98  {
99  case SpiralOutside:
100  //flys until center of the helix is reached an start helix
101  nav_route_xy(nav_spiral.fly_from.x, nav_spiral.fly_from.y, nav_spiral.center.x, nav_spiral.center.y);
102  // center reached?
103  if (nav_approaching_xy(nav_spiral.center.x, nav_spiral.center.y, nav_spiral.fly_from.x, nav_spiral.fly_from.y, 0)) {
104  // nadir image
105 #ifdef DIGITAL_CAM
106  dc_send_command(DC_SHOOT);
107 #endif
108  nav_spiral.status = SpiralStartCircle;
109  }
110  break;
111  case SpiralStartCircle:
112  // Starts helix
113  // storage of current coordinates
114  // calculation needed, State switch to SpiralCircle
115  nav_circle_XY(nav_spiral.center.y, nav_spiral.center.y, nav_spiral.radius_start);
116  if(nav_spiral.dist_from_center >= nav_spiral.radius_start){
117  VECT2_COPY(nav_spiral.last_circle, pos_enu);
118  nav_spiral.status = SpiralCircle;
119  // Start helix
120 #ifdef DIGITAL_CAM
121  dc_Circle(360/nav_spiral.segments);
122 #endif
123  }
124  break;
125  case SpiralCircle: {
126  nav_circle_XY(nav_spiral.center.x, nav_spiral.center.y, nav_spiral.radius_start);
127  // Trigonometrische Berechnung des bereits geflogenen Winkels alpha
128  // equation:
129  // alpha = 2 * asin ( |Starting position angular - current positon| / (2* nav_spiral.radius_start)
130  // if alphamax already reached, increase radius.
131 
132  //DistanceStartEstim = |Starting position angular - current positon|
133  struct FloatVect2 pos_diff;
134  VECT2_DIFF(pos_diff, nav_spiral.last_circle, pos_enu);
135  FLOAT_VECT2_NORM(DistanceStartEstim, pos_diff);
136  CircleAlpha = (2.0 * asin (DistanceStartEstim / (2 * nav_spiral.radius_start)));
137  if (CircleAlpha >= nav_spiral.alpha_limit) {
138  VECT2_COPY(nav_spiral.last_circle, pos_enu);
139  nav_spiral.status = SpiralInc;
140  }
141  break;
142  }
143  case SpiralInc:
144  // increasing circle radius as long as it is smaller than max helix radius
145  if(nav_spiral.radius_start + nav_spiral.radius_increment < nav_spiral.radius)
146  {
147  nav_spiral.radius_start = nav_spiral.radius_start + nav_spiral.radius_increment;
148 #ifdef DIGITAL_CAM
149  if (dc_cam_tracing) {
150  // calculating Cam angle for camera alignment
151  nav_spiral.trans_current.z = stateGetPositionUtm_f()->alt - nav_spiral.center.z;
152  dc_cam_angle = atan(nav_spiral.radius_start/nav_spiral.trans_current.z) * 180 / M_PI;
153  }
154 #endif
155  }
156  else {
157  nav_spiral.radius_start = nav_spiral.radius;
158 #ifdef DIGITAL_CAM
159  // Stopps DC
160  dc_stop();
161 #endif
162  }
163  nav_spiral.status = SpiralCircle;
164  break;
165  default:
166  break;
167  }
168 
169  NavVerticalAutoThrottleMode(0.); /* No pitch */
170  NavVerticalAltitudeMode(nav_spiral.center.z, 0.); /* No preclimb */
171 
172  return TRUE;
173 }
