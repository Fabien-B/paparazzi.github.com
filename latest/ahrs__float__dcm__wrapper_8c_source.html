52 {
53  uint8_t id = AHRS_DCM_FILTER_ID;

86 static void gyro_cb(uint8_t __attribute__((unused)) sender_id,
87  uint32_t stamp, struct Int32Rates *gyro)
88 {
89  ahrs_dcm_last_stamp = stamp;
90 #if USE_AUTO_AHRS_FREQ || !defined(AHRS_PROPAGATE_FREQUENCY)
91  PRINT_CONFIG_MSG("Calculating dt for AHRS dcm propagation.")
92  /* timestamp in usec when last callback was received */
93  static uint32_t last_stamp = 0;
94  if (last_stamp > 0 && ahrs_dcm.is_aligned) {
95  float dt = (float)(stamp - last_stamp) * 1e-6;
96  ahrs_dcm_propagate(gyro, dt);
97  set_body_orientation_and_rates();
98  }
99  last_stamp = stamp;
100 #else
101  PRINT_CONFIG_MSG("Using fixed AHRS_PROPAGATE_FREQUENCY for AHRS dcm propagation.")
102  PRINT_CONFIG_VAR(AHRS_PROPAGATE_FREQUENCY)
103  if (ahrs_dcm.is_aligned) {
104  const float dt = 1. / (AHRS_PROPAGATE_FREQUENCY);
105  ahrs_dcm_propagate(gyro, dt);
106  set_body_orientation_and_rates();
107  }
108 #endif
109 }
