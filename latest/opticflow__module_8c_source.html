110 void opticflow_module_init(void)
111 {
112  // Subscribe ABI messages
113  AbiBindMsgAGL(OPTICFLOW_AGL_ID, &opticflow_agl_ev, opticflow_agl_cb); // ABI to the altitude above ground level
114  AbiBindMsgIMU_ACCEL_INT32(OPTICFLOW_IMU_ID, &opticflow_imu_accel_ev,
115  &opticflow_imu_accel_cb); // ABI to the IMU accelerometer measurements
116  AbiBindMsgBODY_TO_IMU_QUAT(OPTICFLOW_BODY_TO_IMU_ID, &opticflow_body_to_imu_ev,
117  &opticflow_body_to_imu_cb); // ABI to the quaternion of body to imu
118 
119  // Set the opticflow state to 0
120  FLOAT_RATES_ZERO(opticflow_state.rates);
121  float_quat_identity(&opticflow_state.imu_to_body_quat);
122  INT_VECT3_ZERO(opticflow_state.accel_imu_meas);
123  opticflow_state.agl = 0;
124 
125  // Initialize the opticflow calculation
126  opticflow_got_result = false;
127 
128  cv_add_to_device(&OPTICFLOW_CAMERA, opticflow_module_calc);
129 
130 #if PERIODIC_TELEMETRY
131  register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_OPTIC_FLOW_EST, opticflow_telem_send);
132 #endif
133 
134 }

140 void opticflow_module_run(void)
141 {
142  pthread_mutex_lock(&opticflow_mutex);
143  // Update the stabilization loops on the current calculation
144  if (opticflow_got_result) {
145  uint32_t now_ts = get_sys_time_usec();
146  AbiSendMsgOPTICAL_FLOW(OPTICFLOW_SEND_ABI_ID, now_ts,
147  opticflow_result.flow_x,
148  opticflow_result.flow_y,
149  opticflow_result.flow_der_x,
150  opticflow_result.flow_der_y,
151  opticflow_result.noise_measurement,// FIXME, scale to some quality measure 0-255
152  opticflow_result.div_size,
153  opticflow_state.agl);
154  //TODO Find an appropiate quality measure for the noise model in the state filter, for now it is tracked_cnt
155  if (opticflow_result.noise_measurement < 0.8) {
156  AbiSendMsgVELOCITY_ESTIMATE(OPTICFLOW_SEND_ABI_ID, now_ts,
157  opticflow_result.vel_body_x,
158  opticflow_result.vel_body_y,
159  0.0f,
160  opticflow_result.noise_measurement
161  );
162  }
163  opticflow_got_result = false;
164  }
165  pthread_mutex_unlock(&opticflow_mutex);
166 }

175 struct image_t *opticflow_module_calc(struct image_t *img)
176 {
177  // Copy the state
178  // TODO : put accelerometer values at pose of img timestamp
179  struct opticflow_state_t temp_state;
180  struct pose_t pose = get_rotation_at_timestamp(img->pprz_ts);
181  temp_state = opticflow_state;
182  temp_state.rates = pose.rates;
183 
184  // Do the optical flow calculation
185  struct opticflow_result_t temp_result = {}; // new initialization
186  opticflow_calc_frame(&opticflow, &temp_state, img, &temp_result);
187 
188  // Copy the result if finished
189  pthread_mutex_lock(&opticflow_mutex);
190  opticflow_result = temp_result;
191  opticflow_got_result = true;
192 
193 
194  // release the mutex as we are done with editing the opticflow result
195  pthread_mutex_unlock(&opticflow_mutex);
196  return img;
197 }
