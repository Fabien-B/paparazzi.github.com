109 {

113  // Set the opticflow state to 0

115  opticflow_state.theta = 0;
116  opticflow_state.agl = 0;

118  // Initialize the opticflow calculation
119  opticflow_calc_init(&opticflow, 320, 240);
120  opticflow_got_result = FALSE;

122 #ifdef OPTICFLOW_SUBDEV

148 void opticflow_module_run(void)
149 {
150  pthread_mutex_lock(&opticflow_mutex);
151  // Send Updated data to thread
152  opticflow_state.phi = stateGetNedToBodyEulers_f()->phi;
153  opticflow_state.theta = stateGetNedToBodyEulers_f()->theta;
154 
155  // Update the stabilization loops on the current calculation
156  if (opticflow_got_result) {
157  uint32_t now_ts = get_sys_time_usec();
158  uint8_t quality = opticflow_result.divergence; // FIXME, scale to some quality measure 0-255
159  AbiSendMsgOPTICAL_FLOW(OPTICFLOW_SENDER_ID, now_ts,
160  opticflow_result.flow_x,
161  opticflow_result.flow_y,
162  opticflow_result.flow_der_x,
163  opticflow_result.flow_der_x,
164  quality,
165  opticflow_state.agl);
166  if (opticflow_result.tracked_cnt > 0) {
167  AbiSendMsgVELOCITY_ESTIMATE(OPTICFLOW_SENDER_ID, now_ts,
168  opticflow_result.vel_x,
169  opticflow_result.vel_y,
170  0.0f);
171  }
172  opticflow_got_result = FALSE;
173  }
174  pthread_mutex_unlock(&opticflow_mutex);
175 }

212 static void *opticflow_module_calc(void *data __attribute__((unused)))
213 {
214  // Start the streaming on the V4L2 device
215  if (!v4l2_start_capture(opticflow_dev)) {
216  printf("[opticflow_module] Could not start capture of the camera\n");
217  return 0;
218  }
219 
220 #if OPTICFLOW_DEBUG
221  // Create a new JPEG image
222  struct image_t img_jpeg;
223  image_create(&img_jpeg, opticflow_dev->w, opticflow_dev->h, IMAGE_JPEG);
224 #endif
225 
226  /* Main loop of the optical flow calculation */
227  while (TRUE) {
228  // Try to fetch an image
229  struct image_t img;
230  v4l2_image_get(opticflow_dev, &img);
231 
232  // Copy the state
233  pthread_mutex_lock(&opticflow_mutex);
234  struct opticflow_state_t temp_state;
235  memcpy(&temp_state, &opticflow_state, sizeof(struct opticflow_state_t));
236  pthread_mutex_unlock(&opticflow_mutex);
237 
238  // Do the optical flow calculation
239  struct opticflow_result_t temp_result;
240  opticflow_calc_frame(&opticflow, &temp_state, &img, &temp_result);
241 
242  // Copy the result if finished
243  pthread_mutex_lock(&opticflow_mutex);
244  memcpy(&opticflow_result, &temp_result, sizeof(struct opticflow_result_t));
245  opticflow_got_result = TRUE;
246  pthread_mutex_unlock(&opticflow_mutex);
247 
248 #if OPTICFLOW_DEBUG
249  jpeg_encode_image(&img, &img_jpeg, 70, FALSE);
250  rtp_frame_send(
251  &VIEWVIDEO_DEV, // UDP device
252  &img_jpeg,
253  0, // Format 422
254  70, // Jpeg-Quality
255  0, // DRI Header
256  0 // 90kHz time increment
257  );
258 #endif
259 
260  // Free the image
261  v4l2_image_free(opticflow_dev, &img);
262  }
263 
264 #if OPTICFLOW_DEBUG
265  image_free(&img_jpeg);
266 #endif
267 }
