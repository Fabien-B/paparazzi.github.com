142 void meteo_stick_periodic(void)
143 {
144  // Read ADC
145  ads1220_periodic(&meteo_stick.pressure);
146  ads1220_periodic(&meteo_stick.diff_pressure);
147  ads1220_periodic(&meteo_stick.temperature);
148  // Read PWM
149  meteo_stick.humidity_period = pwm_input_period_tics[MS_HUMIDITY_PWM_INPUT];
150 
151  // Log data
152 #if LOG_MS
153  if (pprzLogFile.fs != NULL) {
154  if (!log_ptu_started) {
155  sdLogWriteLog(&pprzLogFile,
156  "P(adc) T(adc) H(ticks) P_diff(adc) GPS_fix TOW(ms) Week Lat(1e7rad) Lon(1e7rad) HMSL(mm) gpseed(cm/s) course(1e7rad) climb(cm/s)\n");
157  log_ptu_started = TRUE;
158  } else {
159  sdLogWriteLog(&pprzLogFile, "%d %d %d %d %d %d %d %d %d %d %d %d %d\n",
160  meteo_stick.pressure.data, meteo_stick.temperature.data,
161  meteo_stick.humidity_period, meteo_stick.diff_pressure.data,
162  gps.fix, gps.tow, gps.week,
163  gps.lla_pos.lat, gps.lla_pos.lon, gps.hmsl,
164  gps.gspeed, gps.course, -gps.ned_vel.z);
165  }
166  }
167 #endif
168 
169  // Send data
170 #if SEND_MS
171  meteo_stick_send_data();
172 #endif
173 }

177 void meteo_stick_event(void)
178 {
179  ads1220_event(&meteo_stick.pressure);
180  ads1220_event(&meteo_stick.diff_pressure);
181  ads1220_event(&meteo_stick.temperature);
182 
183  // send absolute pressure data over ABI as soon as available
184  if (meteo_stick.pressure.data_available) {
185  float abs = MS_PRESSURE_SCALE * (float)((int32_t)meteo_stick.pressure.data - MS_PRESSURE_OFFSET);
186  AbiSendMsgBARO_ABS(METEO_STICK_SENDER_ID, &abs);
187  meteo_stick.pressure.data_available = FALSE;
188  }
189 
190  // send differential pressure data over ABI as soon as available
191  if (meteo_stick.diff_pressure.data_available) {
192  float diff = MS_DIFF_PRESSURE_SCALE * (float)((int32_t)meteo_stick.diff_pressure.data - MS_DIFF_PRESSURE_OFFSET);
193  AbiSendMsgBARO_DIFF(METEO_STICK_SENDER_ID, &diff);
194  meteo_stick.diff_pressure.data_available = FALSE;
195  }
196 }
