58 static int32_t seq_in_array(uint8_t *array, uint16_t array_size, uint8_t *sequence, uint16_t sequence_size)
59 {
60  uint16_t i;
61  static uint16_t current_sequence_id = 0;
62  static uint16_t count_ff = 0;
63 
64  for (i = 0; i < array_size; i++) {
65 
66  // Detect stop sequence
67  if (array[i] == sequence[current_sequence_id]) {
68  current_sequence_id++;
69  if (current_sequence_id >= sequence_size) {
70  count_ff = 0;
71  current_sequence_id = 0;
72  return i;
73  }
74  } else {
75  current_sequence_id = 0;
76  }
77 
78  // Detect ff sequence
79  if (array[i] == 0xFF) {
80  count_ff++;
81 
82  if (count_ff >= 1000) {
83  count_ff = 0;
84  current_sequence_id = 0;
85  return i;
86  }
87  } else {
88  count_ff = 0;
89  }
90  }
91 
92  return -1;
93 }

103 void direct_memory_logger_periodic(void)
104 {
105  int32_t seq_idx;
106  uint16_t i, end_idx;
107 
108  // Switch the different statusses
109  switch (dml.status) {
110  case DML_IDLE:
111  // Do nothing
112  break;
113 
114  // Stopping
115  case DML_STOP:
116  if (dml.sst.status != SST25VFXXXX_IDLE) {
117  break;
118  }
119 
120  dml.status = DML_IDLE;
121  dml.write_addr = dml.sst.flash_addr + 6;
122  sst25vfxxxx_write(&dml.sst, stop_log_sequence, 6);
123  break;
124 
125  // Logging
126  case DML_START:
127  dm_counter = 0;
128  dml.status = DML_LOGGING;
129  case DML_LOGGING:
130  // Check if too slow TODO fix error
131  dm_counter++;
132  if (dml.sst.status != SST25VFXXXX_IDLE) {
133  break;
134  }
135 
136  // Set the log values
137  log_struct.counter++;
138  log_struct.accel_z = imu.accel.z;
139  log_struct.gyro_p = imu.gyro.p;
140  log_struct.gyro_q = imu.gyro.q;
141  log_struct.gyro_r = imu.gyro.r;
142  log_struct.thrust = stabilization_cmd[COMMAND_THRUST];
143 
144  sst25vfxxxx_write(&dml.sst, (uint8_t *) &log_struct, sizeof(struct LogStruct));
145  break;
146 
147  // Reading
148  case DML_READ:
149  dml.status = DML_READING;
150  case DML_READING:
151 
152  if (DMLoggerLink(TxRunning) || dml.sst.status != SST25VFXXXX_IDLE) {
153  break;
154  }
155 
156  // Detect end sequence
157  seq_idx = seq_in_array(&(dml.buffer[5]), DML_BUF_SIZE - 5, stop_log_sequence, 6);
158  if (seq_idx < 0) {
159  end_idx = DML_BUF_SIZE;
160  } else {
161  end_idx = seq_idx + 5;
162  dml.status = DML_IDLE;
163  }
164 
165  for (i = 5; i < end_idx; i++) {
166  DMLoggerLink(Transmit(dml.buffer[i]));
167  }
168 
169  // Read next bytes
170  dml.sst.flash_addr += end_idx - 5;
171  if (seq_idx < 0) {
172  sst25vfxxxx_read(&dml.sst, dml.buffer, DML_BUF_SIZE - 5);
173  }
174  break;
175 
176  default:
177  if (dml.sst.status == SST25VFXXXX_IDLE) {
178  dml.status = DML_IDLE;
179  }
180  break;
181  }
182 }

184 void direct_memory_logger_set(uint8_t val)
185 {
186  // First handle stopping while logging
187  if (dml.status == DML_LOGGING && val == DML_STOP) {
188  dml.status = DML_STOP;
189  return;
190  }
191 
192  // Handle only while idle
193  if (dml.status != DML_IDLE) {
194  return;
195  }
196 
197  // Handle all the statuses
198  dml.status = val;
199  switch (dml.status) {
200  case DML_ERASE:
201  dml.sst.flash_addr = 0x0;
202  dml.write_addr = 0x0;
203  sst25vfxxxx_chip_erase(&dml.sst);
204  break;
205  case DML_START:
206  dml.sst.flash_addr = dml.write_addr;
207  sst25vfxxxx_write(&dml.sst, start_log_sequence, 6);
208  break;
209  case DML_READ:
210  dml.sst.flash_addr = 0x0;
211  sst25vfxxxx_read(&dml.sst, dml.buffer, DML_BUF_SIZE - 5);
212  break;
213  default:
214  break;
215  }
216 }
