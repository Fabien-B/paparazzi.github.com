62  if (rxRuntimeConfig.rcFrameStatusFn(&rxRuntimeConfig) & RX_FRAME_COMPLETE) {
63  rxRuntimeConfig.rcProcessFrameFn(&rxRuntimeConfig);
64  radio_control.frame_cpt++;
65  radio_control.time_since_last_frame = 0;
66  if (radio_control.radio_ok_cpt > 0) {
67  radio_control.radio_ok_cpt--;
68  } else {
69  radio_control.status = RC_OK;
70  for (int i = 0; i < RADIO_CONTROL_NB_CHANNEL; ++i) {
71  frsky_raw[i] = rxRuntimeConfig.rcReadRawFn(&rxRuntimeConfig, i);
72  }
73  NormalizePpmIIR(frsky_raw, radio_control);
74  _received_frame_handler();
