91 {
92  uint8_t id = AHRS_ICE_FILTER_ID;
93  uint8_t mde = 3;

96  uint32_t t_diff = get_sys_time_usec() - ahrs_ice_last_stamp;
97  /* set lost if no new gyro measurements for 50ms */
98  if (t_diff > 50000) { mde = 5; }

176 /* Rotate angles and rates from imu to body frame and set state */
177 static void set_body_state_from_euler(void)
178 {
179  if (ahrs_ice_output_enabled) {
180  struct Int32RMat *body_to_imu_rmat = orientationGetRMat_i(&ahrs_ice.body_to_imu);
181  struct Int32RMat ltp_to_imu_rmat, ltp_to_body_rmat;
182  /* Compute LTP to IMU rotation matrix */
183  int32_rmat_of_eulers(&ltp_to_imu_rmat, &ahrs_ice.ltp_to_imu_euler);
184  /* Compute LTP to BODY rotation matrix */
185  int32_rmat_comp_inv(&ltp_to_body_rmat, &ltp_to_imu_rmat, body_to_imu_rmat);
186  /* Set state */
187  stateSetNedToBodyRMat_i(&ltp_to_body_rmat);
188 
189  struct Int32Rates body_rate;
190  /* compute body rates */
191  int32_rmat_transp_ratemult(&body_rate, body_to_imu_rmat, &ahrs_ice.imu_rate);
192  /* Set state */
193  stateSetBodyRates_i(&body_rate);
194  }
195 }
