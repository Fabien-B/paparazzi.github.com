59 {

61  // NB if BLUEGIGA_BUFFER_SIZE is smaller than 256 then an additional check is needed that len < BLUEGIGA_BUFFER_SIZE

64  }
65 
66  return FALSE;

123 void bluegiga_init(struct bluegiga_periph *p)
124 {
125 #ifdef MODEM_LED
126  LED_INIT(MODEM_LED);
127 #endif
128 
129  // configure the SPI bus
130  bluegiga_spi.input_buf = p->work_rx;
131  bluegiga_spi.output_buf = p->work_tx;
132  bluegiga_spi.input_length = 20;
133  bluegiga_spi.output_length = 20;
134  bluegiga_spi.slave_idx = 0; // Not used for SPI-Slave: always NSS pin
135  bluegiga_spi.select = SPISelectUnselect;
136  bluegiga_spi.cpol = SPICpolIdleHigh;
137  bluegiga_spi.cpha = SPICphaEdge2;
138  bluegiga_spi.dss = SPIDss8bit;
139  bluegiga_spi.bitorder = SPIMSBFirst;
140  bluegiga_spi.cdiv = SPIDiv64;
141  bluegiga_spi.after_cb = (SPICallback) trans_cb;
142 
143  // Configure generic link device
144  p->device.periph = (void *)(p);
145  p->device.check_free_space = (check_free_space_t) dev_check_free_space;
146  p->device.put_byte = (put_byte_t) dev_put_byte;
147  p->device.send_message = (send_message_t) dev_send_message;
148  p->device.char_available = (char_available_t) dev_char_available;
149  p->device.get_byte = (get_byte_t) dev_get_byte;
150 
151  // initialize peripheral variables
152  p->rx_insert_idx = 0;
153  p->rx_extract_idx = 0;
154  p->tx_insert_idx = 0;
155  p->tx_extract_idx = 0;
156 
157  for (int i = 0; i < bluegiga_spi.input_length; i++) {
158  p->work_rx[i] = 0;
159  }
160  for (int i = 0; i < bluegiga_spi.output_length; i++) {
161  p->work_tx[i] = 0;
162  }
163  for (int i = 0; i < 255; i++) {
164  bluegiga_rssi[i] = 127;
165  }
166 
167  p->bytes_recvd_since_last = 0;
168  p->end_of_msg = p->tx_insert_idx;
169 
170  // set DRDY interrupt pin for spi master triggered on falling edge
171  gpio_setup_output(BLUEGIGA_DRDY_GPIO, BLUEGIGA_DRDY_GPIO_PIN);
172  gpio_set(BLUEGIGA_DRDY_GPIO, BLUEGIGA_DRDY_GPIO_PIN);
173 
174  coms_status = BLUEGIGA_UNINIT;
175 
176 #if PERIODIC_TELEMETRY
177  register_periodic_telemetry(DefaultPeriodic, "BLUEGIGA", send_bluegiga);
178 #endif
179 
180  // register spi slave read for transaction
181  spi_slave_register(&(BLUEGIGA_SPI_DEV), &bluegiga_spi);
182 }

185 void bluegiga_transmit(struct bluegiga_periph *p, uint8_t data)
186 {
187  if (dev_check_free_space(p, 1) && coms_status != BLUEGIGA_UNINIT) {
188  p->tx_buf[p->tx_insert_idx] = data;

193 /* Load waiting data into tx peripheral buffer */
194 void bluegiga_load_tx(struct bluegiga_periph *p)
195 {
196  // check data available in buffer to send
197  uint8_t packet_len = ((p->end_of_msg - p->tx_extract_idx + BLUEGIGA_BUFFER_SIZE) % BLUEGIGA_BUFFER_SIZE);
198  if (packet_len > 19) {
199  packet_len = 19;
200  }
201 
202  if (packet_len && coms_status == BLUEGIGA_IDLE) {
203  uint8_t i;
204  // attach header with data length of real data in 20 char data string
205  p->work_tx[0] = packet_len;
206 
207  // copy data from working buffer to spi output buffer
208  for (i = 0; i < packet_len; i++) {
209  p->work_tx[i + 1] = p->tx_buf[(p->tx_extract_idx + i) % BLUEGIGA_BUFFER_SIZE];
210  }
211  bluegiga_increment_buf(&p->tx_extract_idx, packet_len);
212 
213  // clear unused bytes
214  for (i = packet_len + 1; i < bluegiga_spi.output_length; i++) {
215  p->work_tx[i] = 0;
216  }
217 
218  coms_status = BLUEGIGA_SENDING;
219  }
220 }

222 /* read data from dma if available, set as call back of successful spi exchange
223  *
224  * TODO Remove use of bluegiga_p global in following function
225  */
226 void bluegiga_receive(struct spi_transaction *trans)
227 {
228  if (trans->status == SPITransSuccess) {
229  // handle successful msg send
230  if (coms_status == BLUEGIGA_SENDING) {
231  // empty transfer buffer
232  for (uint8_t i = 0; i < trans->output_length; i++) {
233  trans->output_buf[i] = 0;
234  }
235  }
236 
237  /*
238  * 0xff communication lost with ground station
239  * 0xfe RSSI value from broadcaster
240  * 0xfd Change in broadcast mode
241  * 0xfc Receive all recorded RSSI
242  * <=20 Data package from ground station
243  */
244 
245  uint8_t packet_len = 0;
246  uint8_t read_offset = 0;
247  switch (trans->input_buf[0]) {
248  case 0xff: // communication lost with ground station
249 #ifdef MODEM_LED
250  LED_OFF(MODEM_LED);
251 #endif
252  coms_status = BLUEGIGA_UNINIT;
253  gpio_set(BLUEGIGA_DRDY_GPIO, BLUEGIGA_DRDY_GPIO_PIN); // Reset interrupt pin
254  break;
255  case 0xfe: // RSSI value from broadcaster
256  bluegiga_rssi[trans->input_buf[1]] = trans->input_buf[2];
257  packet_len = trans->input_buf[3];
258  read_offset = 4;
259  break;
260  case 0xfd: // Change in broadcast mode
261  gpio_set(BLUEGIGA_DRDY_GPIO, BLUEGIGA_DRDY_GPIO_PIN); // Reset interrupt pin
262 
263  // fetch scan status
264  if (trans->input_buf[1] == 1) {
265  coms_status = BLUEGIGA_BROADCASTING;
266  } else {
267  coms_status = BLUEGIGA_UNINIT;
268  }
269  break;
270  case 0xfc: // Receive all recorded RSSI
271  for (uint8_t i = 0; i < trans->input_buf[1]; i++) {
272  bluegiga_rssi[trans->input_buf[2] + i] = trans->input_buf[3 + i];
273  }
274  break;
275  default:
276  packet_len = trans->input_buf[0]; // length of transmitted message
277  read_offset = 1;
278  }
279 
280  // handle incoming datalink message
281  if (packet_len > 0 && packet_len <= trans->input_length) {
282  // Handle received message
283  for (uint8_t i = 0; i < packet_len; i++) {
284  bluegiga_p.rx_buf[(bluegiga_p.rx_insert_idx + i) % BLUEGIGA_BUFFER_SIZE] = trans->input_buf[i + read_offset];
285  }
286  bluegiga_increment_buf(&bluegiga_p.rx_insert_idx, packet_len);
287  bluegiga_p.bytes_recvd_since_last += packet_len;
288  coms_status = BLUEGIGA_IDLE;
289 
290  for (uint8_t i = 0; i < trans->input_length; i++) {
291  telemetry_copy[i] = trans->input_buf[i];
292  }
293  } else {
294  coms_status = BLUEGIGA_IDLE;
295  }
296 
297  // load next message to be sent into work buffer, needs to be loaded before calling spi_slave_register
298  bluegiga_load_tx(&bluegiga_p);
299 
300  // register spi slave read for next transaction
301  spi_slave_register(&(BLUEGIGA_SPI_DEV), &bluegiga_spi);
302  }
303 }
