56 static int dev_check_free_space(struct bluegiga_periph *p, uint8_t len)
57 {
58  // check if there is enough space for message
59  // NB if BLUEGIGA_BUFFER_SIZE is smaller than 256 then an additional check is needed that len < BLUEGIGA_BUFFER_SIZE
60  if ( len - 1 <= ((p->tx_extract_idx - p->tx_insert_idx - 1 + BLUEGIGA_BUFFER_SIZE) % BLUEGIGA_BUFFER_SIZE))
61  return TRUE;
62 
63  return FALSE;
64 }

102 void bluegiga_init(struct bluegiga_periph *p)
103 {
104 #ifdef MODEM_LED
105  LED_INIT(MODEM_LED);
106 #endif
107 
108  // configure the SPI bus
109  bluegiga_spi.input_buf = p->work_rx;
110  bluegiga_spi.output_buf = p->work_tx;
111  bluegiga_spi.input_length = 20;
112  bluegiga_spi.output_length = 20;
113  bluegiga_spi.slave_idx = 0; // Not used for SPI-Slave: always NSS pin
114  bluegiga_spi.select = SPISelectUnselect;
115  bluegiga_spi.cpol = SPICpolIdleHigh;
116  bluegiga_spi.cpha = SPICphaEdge2;
117  bluegiga_spi.dss = SPIDss8bit;
118  bluegiga_spi.bitorder = SPIMSBFirst;
119  bluegiga_spi.cdiv = SPIDiv64;
120  bluegiga_spi.after_cb = (SPICallback) trans_cb;
121 
122  // Configure generic link device
123  p->device.periph = (void *)(p);
124  p->device.check_free_space = (check_free_space_t) dev_check_free_space;
125  p->device.put_byte = (put_byte_t) dev_put_byte;
126  p->device.send_message = (send_message_t) dev_send_message;
127  p->device.char_available = (char_available_t) dev_char_available;
128  p->device.get_byte = (get_byte_t) dev_get_byte;
129 
130  // initialize peripheral variables
131  p->rx_insert_idx = 0;
132  p->rx_extract_idx = 0;
133  p->tx_insert_idx = 0;
134  p->tx_extract_idx = 0;
135 
136  for (int i = 0; i < bluegiga_spi.input_length; i++) {
137  p->work_rx[i] = 0;
138  }
139  for (int i = 0; i < bluegiga_spi.output_length; i++) {
140  p->work_tx[i] = 0;
141  }
142  for (int i = 0; i < 255; i++) {
143  bluegiga_rssi[i] = 127;
144  }
145 
146  // set DRDY interrupt pin for spi master triggered on falling edge
147  gpio_setup_output(BLUEGIGA_DRDY_GPIO, BLUEGIGA_DRDY_GPIO_PIN);
148  gpio_set(BLUEGIGA_DRDY_GPIO, BLUEGIGA_DRDY_GPIO_PIN);
149 
150  // register spi slave read for transaction
151  spi_slave_register(&(BLUEGIGA_SPI_DEV), &bluegiga_spi);
152 
153  coms_status = BLUEGIGA_UNINIT;
154 }

156 /* Add one byte to the end of tx circular buffer */
157 void bluegiga_transmit(struct bluegiga_periph *p, uint8_t data)
158 {
159  if (dev_check_free_space(p, 1) && coms_status != BLUEGIGA_UNINIT) {
160  p->tx_buf[p->tx_insert_idx] = data;
161  bluegiga_increment_buf(&p->tx_insert_idx, 1);
162  }
163 }

165 /* Send data in transmit buffer to spi master */
166 void bluegiga_send(struct bluegiga_periph *p)
167 {
168  uint8_t packet_len;
169 
170  // check data available in buffer to send
171  packet_len = ((p->tx_insert_idx - p->tx_extract_idx + BLUEGIGA_BUFFER_SIZE) % BLUEGIGA_BUFFER_SIZE);
172  if (packet_len > 18) {
173  packet_len = 18;
174  }
175 
176  if (packet_len && coms_status == BLUEGIGA_IDLE) {
177  uint8_t i;
178  // attach header with data length of real data in 20 char data string
179  p->work_tx[1] = packet_len;
180 
181  // copy data from working buffer to spi output buffer
182  for (i = 0; i < packet_len; i++) {
183  p->work_tx[i + 2] = p->tx_buf[(p->tx_extract_idx + i) % BLUEGIGA_BUFFER_SIZE];
184  }
185  bluegiga_increment_buf(&p->tx_extract_idx, packet_len);
186 
187  // clear unused bytes
188  for (i = packet_len + 2; i < bluegiga_spi.output_length; i++) {
189  p->work_tx[i] = 0;
190  }
191 
192  coms_status = BLUEGIGA_SENDING;
193  }
194 }

196 /* read data from dma if available, set as call back of successful spi exchange */
197 void bluegiga_receive(struct spi_transaction *trans)
198 {
199  if (trans->status == SPITransSuccess) {
200  if (coms_status == BLUEGIGA_SENDING) {
201  // Handle successful sent message
202  for (uint8_t i = 0; i < trans->output_length; i++) { // Clear tx buffer
203  trans->output_buf[i] = 0;
204  }
205  }
206 
207  uint8_t packet_len = trans->input_buf[3]; // length of transmitted message
208 
209  if (packet_len > trans->input_length) {
210  // Direct message from Bluegiga
211  // int k_rssi, i;
212  switch (packet_len) {
213  case 0xff: // Connection lost with ground station!
214 #ifdef MODEM_LED
215  LED_OFF(MODEM_LED);
216 #endif
217  coms_status = BLUEGIGA_UNINIT;
218  gpio_set(BLUEGIGA_DRDY_GPIO, BLUEGIGA_DRDY_GPIO_PIN); // Reset interrupt pin
219  break;
220  /*case 0xfe: // rssi data
221  k_rssi = trans->input_buf[2];
222  for (i = 0; i < k_rssi; i++) {
223  bluegiga_rssi[i] = trans->input_buf[3 + i];
224  }
225  break;*/
226  case 0xfd: // interrupt handled on bluegiga
227  gpio_set(BLUEGIGA_DRDY_GPIO, BLUEGIGA_DRDY_GPIO_PIN); // Reset interrupt pin
228 
229  // fetch scan status
230  if (trans->input_buf[2] == 1) {
231  coms_status = BLUEGIGA_SCANNING;
232  } else {
233  coms_status = BLUEGIGA_UNINIT;
234  }
235  break;
236  default:
237  break;
238  }
239  }
240 
241  // handle incoming datalink message
242  else if (packet_len > 0) {
243  bluegiga_rssi[trans->input_buf[1]] = trans->input_buf[2];
244  // Handle received message
245  for (uint8_t i = 0; i < packet_len; i++) {
246  bluegiga_p.rx_buf[(bluegiga_p.rx_insert_idx + i) % BLUEGIGA_BUFFER_SIZE] = trans->input_buf[i + 4];
247  }
248  bluegiga_increment_buf(&bluegiga_p.rx_insert_idx, packet_len);
249  coms_status = BLUEGIGA_IDLE;
250  } else {
251  coms_status = BLUEGIGA_IDLE;
252  }
253 
254  // clear rx buffer
255  for (uint8_t i = 0; i < trans->input_length; i++) {
256  trans->input_buf[i] = 0;
257  }
258  // register spi slave read for next transaction
259  spi_slave_register(&(BLUEGIGA_SPI_DEV), &bluegiga_spi);
260  }
261 }
