54 static void range_msg_callback(uint8_t sender_id __attribute__((unused)),
55  uint8_t ac_id, float range, float tracked_v_north, float tracked_v_east, float tracked_h)
56 {
57  int idx = -1; // Initialize the index of all tracked drones (-1 for null assumption of no drone found)
58 
59  // Check if a new aircraft ID is present, if it's a new ID we start a new EKF for it.
60  if (!int32_vect_find(id_array, ac_id, &idx, RL_NUAVS) &&
61  (number_filters < RL_NUAVS)) {
62  id_array[number_filters] = ac_id;
63  discrete_ekf_new(&ekf_rl[number_filters]);
64  number_filters++;
65  } else if (idx != -1) {
66  range_array[idx] = range;
67  ekf_rl[idx].dt = (get_sys_time_usec() - latest_update_time[idx]) / pow(10, 6); // Update the time between messages
68 
69  // Measurement Vector Z = [range owvVx(NED) ownVy(NED) tracked_v_north(NED) tracked_v_east(NED) dh]
70  float Z[EKF_M] = {range, stateGetSpeedEnu_f()->y, stateGetSpeedEnu_f()->x, tracked_v_north, tracked_v_east, tracked_h - stateGetPositionEnu_f()->z};
71 
72  discrete_ekf_predict(&ekf_rl[idx]);
73  discrete_ekf_update(&ekf_rl[idx], Z);
74  }
75 
76  latest_update_time[idx] = get_sys_time_usec();
77 };

80 static void send_relative_localization_data(struct transport_tx *trans, struct link_device *dev)
81 {
82  pprzmsg_cnt++;
83  if (pprzmsg_cnt >= number_filters) {
84  pprzmsg_cnt = 0;
85  }
86 
87  if (number_filters > 0) {

91  &ekf_rl[pprzmsg_cnt].X[2], &ekf_rl[pprzmsg_cnt].X[3], // vx vy (own)
92  &ekf_rl[pprzmsg_cnt].X[4], &ekf_rl[pprzmsg_cnt].X[5], // vx vy (tracked)
