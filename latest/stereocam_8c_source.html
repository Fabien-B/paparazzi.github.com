119 void get_msg_properties(uint8_t *raw, MsgProperties *properties, uint16_t start)
120 {
121  *properties = (MsgProperties) {start, 0, 0};
122  uint16_t i = start, startOfLine = start;
123  while (1) {
124  // Check the first 3 bytes for the pattern 255-0-0, then check what special byte is encoded next
125  if ((raw[i] == 255) && (raw[add(i, 1)] == 0) && (raw[add(i, 2)] == 0)) {
126  if (raw[add(i, 3)] == 171) { // End of image
127  break;
128  }
129  if (raw[add(i, 3)] == 128) { // Start of line
130  startOfLine = i;
131  }
132  if (raw[add(i, 3)] == 218) { // End of line
133  properties->height++;
134  properties->width = diff(i, startOfLine + 4); // removed 4 for the indication bits at the end of line
135  }
136  }
137  i = add(i, 1);
138  }
139 }

145 static uint8_t handleStereoPackage(void)
146 {
147  // read all data from the stereo com link, check that don't overtake extract
148  while (dev->char_available(dev->periph) && add(insert_loc, 1) != extract_loc) {
149  ser_read_buf[insert_loc] = dev->get_byte(dev->periph);
150  insert_loc = add(insert_loc, 1);
151  }
152 
153  // search for complete message in buffer, if found increments read location and returns immediately
154  // stay length of line header behind insert (NB. code can be changed to look back instead of forward)
155  while (diff(insert_loc, extract_loc) > 3) {
156  if (isStartOfMsg(ser_read_buf, extract_loc)) {
157  msg_start = extract_loc;
158  } else if (isEndOfMsg(ser_read_buf, extract_loc)) { // process msg
159  // Find the properties of the image by iterating over the complete image
160  get_msg_properties(ser_read_buf, &msgProperties, msg_start);
161 
162  // Copy array to circular buffer and remove all bytes that are indications of start and stop lines
163  uint16_t i = msg_start, j = 0, k = 0, index = 0;
164  for (k = 0; k < msgProperties.height; k++) {
165  i = add(i, 8); // step over EOL and SOL or SOM and SOL
166  for (j = 0; j < msgProperties.width; j++) {
167  msg_buf[index++] = ser_read_buf[i];
168  i = add(i, 1);
169  }
170  } // continue search for new line
171  stereocam_data.len = msgProperties.width * msgProperties.height;
172  stereocam_data.fresh = 1;
173  extract_loc = add(extract_loc, 4); // step over EOM string
174  return 1;
175  }
176  extract_loc = add(extract_loc, 1);
177  }
178  return 0;
179 }
