52 /* Big Endian to Float */
53 static inline float bef(volatile uint8_t *c)
54 {
55  float f;
56  int8_t *p;
57  p = ((int8_t *)&f) + 3;
58  *p-- = *c++;
59  *p-- = *c++;
60  *p-- = *c++;
61  *p = *c;
62  return f;
63 }

65 static inline bool_t gx3_verify_chk(volatile uint8_t *buff_add)
66 {
67  uint16_t i, chk_calc;
68  chk_calc = 0;
69  for (i = 0; i < GX3_MSG_LEN - 2; i++) {
70  chk_calc += (uint8_t) * buff_add++;
71  }
72  return (chk_calc == ((((uint16_t) * buff_add) << 8) + (uint8_t) * (buff_add + 1)));
73 }

293 /* GX3 Packet Collection */
294 void gx3_packet_parse(uint8_t c)
295 {
296  switch (ahrs_gx3.packet.status) {
297  case GX3PacketWaiting:
298  ahrs_gx3.packet.msg_idx = 0;
299  if (c == GX3_HEADER) {
300  ahrs_gx3.packet.status++;
301  ahrs_gx3.packet.msg_buf[ahrs_gx3.packet.msg_idx] = c;
302  ahrs_gx3.packet.msg_idx++;
303  } else {
304  ahrs_gx3.packet.hdr_error++;
305  }
306  break;
307  case GX3PacketReading:
308  ahrs_gx3.packet.msg_buf[ahrs_gx3.packet.msg_idx] = c;
309  ahrs_gx3.packet.msg_idx++;
310  if (ahrs_gx3.packet.msg_idx == GX3_MSG_LEN) {
311  if (gx3_verify_chk(ahrs_gx3.packet.msg_buf)) {
312  ahrs_gx3.packet.msg_available = TRUE;
313  } else {
314  ahrs_gx3.packet.msg_available = FALSE;
315  ahrs_gx3.packet.chksm_error++;
316  }
317  ahrs_gx3.packet.status = 0;
318  }
319  break;
320  default:
321  ahrs_gx3.packet.status = 0;
322  ahrs_gx3.packet.msg_idx = 0;
323  break;
324  }
325 }
