118 void jevois_mavlink_event(void)
119 {
120  mavlink_message_t msg;
121  mavlink_status_t status;
122 
123 
124  while (MAVLinkChAvailable()) {
125  uint8_t c = MAVLinkGetch();
126  if (mavlink_parse_char(MAVLINK_COMM_1, c, &msg, &status)) {
127  //printf("msg.msgid is %d\n",msg.msgid);
128  switch (msg.msgid) {
129  case MAVLINK_MSG_ID_HEARTBEAT: {
130  mavlink_heartbeat_t heartbeat;
131  mavlink_msg_heartbeat_decode(&msg, &heartbeat);
132  // do something with heartbeat variable
133  }
134  break;
135 
136  case MAVLINK_MSG_ID_ATTITUDE: {
137 
138  // read attitude command from Jevois
139  mavlink_msg_attitude_decode(&msg, &jevois_mavlink_attitude);
140 
141  DEBUG_PRINT("[jevois mavlink] phi_cmd = %f\n", DegOfRad(jevois_mavlink_attitude.roll));
142  DEBUG_PRINT("[jevois mavlink] theta_cmd = %f\n", DegOfRad(jevois_mavlink_attitude.pitch));
143  DEBUG_PRINT("[jevois mavlink] psi_cmd = %f\n", DegOfRad(jevois_mavlink_attitude.yaw));
144  }
145  break;
146 
147  case MAVLINK_MSG_ID_ALTITUDE: {
148  mavlink_msg_altitude_decode(&msg, &jevois_mavlink_altitude);
149 
150  DEBUG_PRINT("[jevois mavlink] desired altitude is %f", jevois_mavlink_altitude.altitude_relative);
151  }
152  break;
153 
154  case MAVLINK_MSG_ID_DEBUG: {
155  mavlink_msg_debug_decode(&msg, &jevois_mavlink_debug);
156 
157  DEBUG_PRINT("[jevois mavlink] debug value is %f", jevois_mavlink_debug.value);
158  DEBUG_PRINT("[jevois mavlink] debug ind is %d", jevois_mavlink_debug.ind);
159  }
160  break;
161 
162  case MAVLINK_MSG_ID_MANUAL_SETPOINT: {
163  mavlink_msg_manual_setpoint_decode(&msg, &jevois_mavlink_manual_setpoint);
164 
165  DEBUG_PRINT("[jevois mavlink] phi_cmd = %f\n", DegOfRad(jevois_mavlink_manual_setpoint.roll));
166  DEBUG_PRINT("[jevois mavlink] theta_cmd = %f\n", DegOfRad(jevois_mavlink_manual_setpoint.pitch));
167  DEBUG_PRINT("[jevois mavlink] psi_cmd = %f\n", DegOfRad(jevois_mavlink_manual_setpoint.yaw));
168  DEBUG_PRINT("[jevois mavlink] alt_cmd = %f\n", jevois_mavlink_manual_setpoint.thrust);
169  }
170  break;
171 
172 
173  case MAVLINK_MSG_ID_LOCAL_POSITION_NED: {
174  mavlink_msg_local_position_ned_decode(&msg, &jevois_mavlink_local_position);
175  }
176  break;
177  }
178  }
179  }
180 }
