56 // Called on each vision analysis result after receiving the struct
57 void run_avoid_navigation_onvision(void)
58 {
59  // Send ALL vision data to the ground
60  DOWNLINK_SEND_PAYLOAD(DefaultChannel, DefaultDevice, 5, avoid_navigation_data.stereo_bin);
61 
62  switch (avoid_navigation_data.mode) {
63  case 0: // Go to Goal and stop at obstacles
64  //count 4 subsequent obstacles
65  if (avoid_navigation_data.stereo_bin[0] > 1) {
66  counter = counter + 1;
67  if (counter > 1) {
68  counter = 0;
69  //Obstacle detected, go to turn until clear mode
70  obstacle_detected = TRUE;
71  avoid_navigation_data.mode = 1;
72  }
73  } else {
74  counter = 0;
75  }
76  break;
77  case 1: // Turn until clear
78  //count 20 subsequent free frames
79  if (avoid_navigation_data.stereo_bin[0] < 1) {
80  counter = counter + 1;
81  if (counter > 12) {
82  counter = 0;
83  //Stop and put waypoint 2.5 m ahead
84  struct EnuCoor_i new_coor;
85  struct EnuCoor_i *pos = stateGetPositionEnu_i();
86  float sin_heading = sinf(ANGLE_FLOAT_OF_BFP(nav_heading));
87  float cos_heading = cosf(ANGLE_FLOAT_OF_BFP(nav_heading));
88  new_coor.x = pos->x + POS_BFP_OF_REAL(sin_heading * 3.5);
89  new_coor.y = pos->y + POS_BFP_OF_REAL(cos_heading * 3.5);
90  new_coor.z = pos->z;
91  waypoint_set_xy_i(WP_W1, new_coor.x, new_coor.y);
92  obstacle_detected = FALSE;
93  avoid_navigation_data.mode = 0;
94  }
95  } else {
96  counter = 0;
97  }
98  break;
99  case 2:
100  break;
101  default: // do nothing
102  break;
103  }
104  avoid_navigation_data.stereo_bin[2] = avoid_navigation_data.stereo_bin[0] > 20;
105  avoid_navigation_data.stereo_bin[3] = avoid_navigation_data.mode;
106  avoid_navigation_data.stereo_bin[4] = counter;
107 
108 #if STEREO_LED
109  if (obstacle_detected) {
110  LED_ON(STEREO_LED);
111  } else {
112  LED_OFF(STEREO_LED);
113  }
114 #endif
115 }
