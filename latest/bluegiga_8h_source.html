76 // BLUEGIGA is using pprz_transport
77 // FIXME it should not appear here, this will be fixed with the rx improvements some day...
78 // BLUEGIGA needs a specific read_buffer function
79 #include "subsystems/datalink/pprz_transport.h"
80 #include "led.h"
81 static inline void bluegiga_read_buffer(struct bluegiga_periph *p, struct pprz_transport *t)
82 {
83  do {
84  uint8_t c = 0;
85  do {
86  parse_pprz(t, p->rx_buf[(p->rx_extract_idx + c++) % BLUEGIGA_BUFFER_SIZE]);
87  } while (((p->rx_extract_idx + c) % BLUEGIGA_BUFFER_SIZE != p->rx_insert_idx)
88  && !(t->trans_rx.msg_received));
89  // reached end of circular read buffer or message received
90  // if received, decode and advance
91  if (t->trans_rx.msg_received) {
92 #ifdef MODEM_LED
93  LED_TOGGLE(MODEM_LED);
94 #endif
95  pprz_parse_payload(t);
96  t->trans_rx.msg_received = FALSE;
97  }
98  bluegiga_increment_buf(&p->rx_extract_idx, c);
99  } while (bluegiga_ch_available(p)); // continue till all messages read
100 }
