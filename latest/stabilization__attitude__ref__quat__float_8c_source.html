113 void stabilization_attitude_ref_init(void)
114 {
115 
116  FLOAT_EULERS_ZERO(stab_att_sp_euler);
117  float_quat_identity(&stab_att_sp_quat);
118  FLOAT_EULERS_ZERO(stab_att_ref_euler);
119  float_quat_identity(&stab_att_ref_quat);
120  FLOAT_RATES_ZERO(stab_att_ref_rate);
121  FLOAT_RATES_ZERO(stab_att_ref_accel);
122 
123  for (int i = 0; i < STABILIZATION_ATTITUDE_GAIN_NB; i++) {
124  RATES_ASSIGN(stab_att_ref_model[i].omega, omega_p[i], omega_q[i], omega_r[i]);
125  RATES_ASSIGN(stab_att_ref_model[i].zeta, zeta_p[i], zeta_q[i], zeta_r[i]);
126  RATES_ASSIGN(two_omega_squared[i], 2 * omega_p[i]*omega_p[i], 2 * omega_q[i]*omega_q[i], 2 * omega_r[i]*omega_r[i]);
127  }

148 #ifndef STABILIZATION_ATTITUDE_REF_QUAT_INFINITESIMAL_STEP
149 #define STABILIZATION_ATTITUDE_REF_QUAT_INFINITESIMAL_STEP TRUE

152 void stabilization_attitude_ref_update(void)
153 {
154 
155  /* integrate reference attitude */
156 #if STABILIZATION_ATTITUDE_REF_QUAT_INFINITESIMAL_STEP
157  struct FloatQuat qdot;
158  float_quat_derivative(&qdot, &stab_att_ref_rate, &stab_att_ref_quat);
159  QUAT_SMUL(qdot, qdot, DT_UPDATE);
160  QUAT_ADD(stab_att_ref_quat, qdot);
161 #else // use finite step (involves trig)
162  struct FloatQuat delta_q;
163  float_quat_differential(&delta_q, &stab_att_ref_rate, DT_UPDATE);
164  /* compose new ref_quat by quaternion multiplication of delta rotation and current ref_quat */
165  struct FloatQuat new_ref_quat;
166  float_quat_comp(&new_ref_quat, &stab_att_ref_quat, &delta_q);
167  QUAT_COPY(stab_att_ref_quat, new_ref_quat);
168 #endif
169  float_quat_normalize(&stab_att_ref_quat);
170 
171  /* integrate reference rotational speeds */
172  struct FloatRates delta_rate;
173  RATES_SMUL(delta_rate, stab_att_ref_accel, DT_UPDATE);
174  RATES_ADD(stab_att_ref_rate, delta_rate);
175 
176  /* compute reference angular accelerations */
177  struct FloatQuat err;
178  /* compute reference attitude error */
179  float_quat_inv_comp(&err, &stab_att_sp_quat, &stab_att_ref_quat);
180  /* wrap it in the shortest direction */
181  float_quat_wrap_shortest(&err);
182  /* propagate the 2nd order linear model: xdotdot = -2*zeta*omega*xdot - omega^2*x */
183  /* since error quaternion contains the half-angles we get 2*omega^2*err */
184  stab_att_ref_accel.p = -2.*stab_att_ref_model[ref_idx].zeta.p * stab_att_ref_model[ref_idx].omega.p *
185  stab_att_ref_rate.p
186  - two_omega_squared[ref_idx].p * err.qx;
187  stab_att_ref_accel.q = -2.*stab_att_ref_model[ref_idx].zeta.q * stab_att_ref_model[ref_idx].omega.q *
188  stab_att_ref_rate.q
189  - two_omega_squared[ref_idx].q * err.qy;
190  stab_att_ref_accel.r = -2.*stab_att_ref_model[ref_idx].zeta.r * stab_att_ref_model[ref_idx].omega.r *
191  stab_att_ref_rate.r
192  - two_omega_squared[ref_idx].r * err.qz;
193 
194  /* saturate acceleration */
195  const struct FloatRates MIN_ACCEL = { -REF_ACCEL_MAX_P, -REF_ACCEL_MAX_Q, -REF_ACCEL_MAX_R };
196  const struct FloatRates MAX_ACCEL = { REF_ACCEL_MAX_P, REF_ACCEL_MAX_Q, REF_ACCEL_MAX_R };
197  RATES_BOUND_BOX(stab_att_ref_accel, MIN_ACCEL, MAX_ACCEL);
198 
199  /* saturate angular speed and trim accel accordingly */
200  SATURATE_SPEED_TRIM_ACCEL();
201 
202  /* compute ref_euler */
203  float_eulers_of_quat(&stab_att_ref_euler, &stab_att_ref_quat);
204 }
