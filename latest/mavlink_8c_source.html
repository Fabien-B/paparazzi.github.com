117 static int16_t settings_idx_from_param_id(char *param_id)
118 {
119  int i, j;
120  int16_t settings_idx = -1;
121 
122  // Go trough all the settings to search the ID
123  for (i = 0; i < NB_SETTING; i++) {
124  for (j = 0; j < 16; j++) {
125  if (mavlink_param_names[i][j] != param_id[j]) {
126  break;
127  }
128 
129  if (mavlink_param_names[i][j] == '\0') {
130  settings_idx = i;
131  return settings_idx;
132  }
133  }
134 
135  if (mavlink_param_names[i][j] == '\0') {
136  break;
137  }
138  }
139  return settings_idx;
140 }

145 void mavlink_event(void)
146 {
147  mavlink_message_t msg;
148  mavlink_status_t status;
149 
150  // Check uplink
151  while (MAVLinkChAvailable()) {
152  char test = MAVLinkGetch();
153 
154  // When we receive a message
155  if (mavlink_parse_char(MAVLINK_COMM_0, test, &msg, &status)) {
156  switch (msg.msgid) {
157  case MAVLINK_MSG_ID_HEARTBEAT:
158  break;
159 
160  /* When requesting data streams say we can't send them */
161  case MAVLINK_MSG_ID_REQUEST_DATA_STREAM: {
162  mavlink_request_data_stream_t cmd;
163  mavlink_msg_request_data_stream_decode(&msg, &cmd);
164 
165  mavlink_msg_data_stream_send(MAVLINK_COMM_0, cmd.req_stream_id, 0, 0);
166  MAVLinkSendMessage();
167  break;
168  }
169 
170  /* Override channels with RC */
171  case MAVLINK_MSG_ID_RC_CHANNELS_OVERRIDE: {
172  mavlink_rc_channels_override_t cmd;
173  mavlink_msg_rc_channels_override_decode(&msg, &cmd);
174 #if defined RADIO_CONTROL && defined RADIO_CONTROL_TYPE_DATALINK
175  uint8_t thrust = (cmd.chan3_raw - 950) * 127 / 1100;
176  int8_t roll = -(cmd.chan1_raw - 1500) * 255 / 1100 / 2;
177  int8_t pitch = -(cmd.chan2_raw - 1500) * 255 / 1100 / 2;
178  int8_t yaw = -(cmd.chan4_raw - 1500) * 255 / 1100;
179  parse_rc_4ch_datalink(0, thrust, roll, pitch, yaw);
180  //printf("RECEIVED: RC Channel Override for: %d/%d: throttle: %d; roll: %d; pitch: %d; yaw: %d;\r\n",
181  // cmd.target_system, cmd.target_component, thrust, roll, pitch, yaw);
182 #endif
183  break;
184  }
185 
186  /* When a request is made of the parameters list */
187  case MAVLINK_MSG_ID_PARAM_REQUEST_LIST: {
188  mavlink_params_idx = 0;
189  break;
190  }
191 
192  /* When a request os made for a specific parameter */
193  case MAVLINK_MSG_ID_PARAM_REQUEST_READ: {
194  mavlink_param_request_read_t cmd;
195  mavlink_msg_param_request_read_decode(&msg, &cmd);
196 
197  // First check param_index and search for the ID if needed
198  if (cmd.param_index == -1) {
199  cmd.param_index = settings_idx_from_param_id(cmd.param_id);
200  }
201 
202  mavlink_msg_param_value_send(MAVLINK_COMM_0,
203  mavlink_param_names[cmd.param_index],
204  settings_get_value(cmd.param_index),
205  MAV_PARAM_TYPE_REAL32,
206  NB_SETTING,
207  cmd.param_index);
208  MAVLinkSendMessage();
209 
210  break;
211  }
212 
213  case MAVLINK_MSG_ID_PARAM_SET: {
214  mavlink_param_set_t set;
215  mavlink_msg_param_set_decode(&msg, &set);
216 
217  // Check if this message is for this system
218  if ((uint8_t) set.target_system == AC_ID) {
219  int16_t idx = settings_idx_from_param_id(set.param_id);
220 
221  // setting found
222  if (idx >= 0) {
223  // Only write if new value is NOT "not-a-number"
224  // AND is NOT infinity
225  if (set.param_type == MAV_PARAM_TYPE_REAL32 &&
226  !isnan(set.param_value) && !isinf(set.param_value)) {
227  DlSetting(idx, set.param_value);
228  // Report back new value
229  mavlink_msg_param_value_send(MAVLINK_COMM_0,
230  mavlink_param_names[idx],
231  settings_get_value(idx),
232  MAV_PARAM_TYPE_REAL32,
233  NB_SETTING,
234  idx);
235  MAVLinkSendMessage();
236  }
237  }
238  }
239  }
240  break;
241 
242  /* request for mission list, answer with number of waypoints */
243  case MAVLINK_MSG_ID_MISSION_REQUEST_LIST: {
244  mavlink_mission_request_list_t req;
245  mavlink_msg_mission_request_list_decode(&msg, &req);
246 
247  if (req.target_system == mavlink_system.sysid) {
248  mavlink_msg_mission_count_send(MAVLINK_COMM_0,
249  msg.sysid,
250  msg.compid,
251  NB_WAYPOINT);
252  MAVLinkSendMessage();
253  }
254  }
255  break;
256 
257  /* request for mission item, answer with waypoint */
258  case MAVLINK_MSG_ID_MISSION_REQUEST: {
259  mavlink_mission_request_t req;
260  mavlink_msg_mission_request_decode(&msg, &req);
261 
262  if (req.target_system == mavlink_system.sysid) {
263  if (req.seq < NB_WAYPOINT) {
264  mavlink_msg_mission_item_send(MAVLINK_COMM_0,
265  msg.sysid,
266  msg.compid,
267  req.seq,
268  MAV_FRAME_LOCAL_ENU,
269  MAV_CMD_NAV_WAYPOINT,
270  0, // current
271  0, // autocontinue
272  0, 0, 0, 0, // params
273  WaypointX(req.seq),
274  WaypointY(req.seq),
275  WaypointAlt(req.seq));
276  MAVLinkSendMessage();
277  }
278  }
279  }
280  break;
281 
282  default:
283  //Do nothing
284  //printf("Received message with id: %d\r\n", msg.msgid);
285  break;
286  }
287  }
288 
289  }

295 static inline void mavlink_send_heartbeat(void)
296 {
297  uint8_t mav_state = MAV_STATE_CALIBRATING;
298  uint8_t mav_mode = 0;
299 #ifdef AP
300  uint8_t mav_type = MAV_TYPE_FIXED_WING;
301  switch (pprz_mode) {
302  case PPRZ_MODE_MANUAL:
303  mav_mode |= MAV_MODE_FLAG_MANUAL_INPUT_ENABLED;
304  break;
305  case PPRZ_MODE_AUTO1:
306  mav_mode |= MAV_MODE_FLAG_STABILIZE_ENABLED|MAV_MODE_FLAG_MANUAL_INPUT_ENABLED;
307  break;
308  case PPRZ_MODE_AUTO2:
309  mav_mode |= MAV_MODE_FLAG_GUIDED_ENABLED|MAV_MODE_FLAG_AUTO_ENABLED;
310  break;
311  case PPRZ_MODE_HOME:
312  mav_mode |= MAV_MODE_FLAG_AUTO_ENABLED;
313  break;
314  default:
315  break;
316  }
317 #else
318  uint8_t mav_type = MAV_TYPE_QUADROTOR;
319  switch (autopilot_mode) {
320  case AP_MODE_HOME:
321  mav_mode |= MAV_MODE_FLAG_AUTO_ENABLED;
322  break;
323  case AP_MODE_RATE_DIRECT:
324  case AP_MODE_RATE_RC_CLIMB:
325  case AP_MODE_RATE_Z_HOLD:
326  case AP_MODE_RC_DIRECT:
327  mav_mode |= MAV_MODE_FLAG_MANUAL_INPUT_ENABLED;
328  break;
329  case AP_MODE_ATTITUDE_DIRECT:
330  case AP_MODE_ATTITUDE_CLIMB:
331  case AP_MODE_ATTITUDE_Z_HOLD:
332  case AP_MODE_ATTITUDE_RC_CLIMB:
333  case AP_MODE_HOVER_DIRECT:
334  case AP_MODE_HOVER_CLIMB:
335  case AP_MODE_HOVER_Z_HOLD:
336  case AP_MODE_CARE_FREE_DIRECT:
337  mav_mode |= MAV_MODE_FLAG_STABILIZE_ENABLED|MAV_MODE_FLAG_MANUAL_INPUT_ENABLED;
338  break;
339  case AP_MODE_NAV:
340  mav_mode |= MAV_MODE_FLAG_GUIDED_ENABLED|MAV_MODE_FLAG_AUTO_ENABLED;
341  break;
342  default:
343  break;
344  }
345 #endif
346  if (stateIsAttitudeValid()) {
347  if (kill_throttle) {
348  mav_state = MAV_STATE_STANDBY;
349  }
350  else {
351  mav_state = MAV_STATE_ACTIVE;
352  mav_mode |= MAV_MODE_FLAG_SAFETY_ARMED;
353  }
354  }
355  mavlink_msg_heartbeat_send(MAVLINK_COMM_0,
356  mav_type,
357  MAV_AUTOPILOT_PPZ,
358  mav_mode,
359  0, // custom_mode
360  mav_state);
361  MAVLinkSendMessage();
362 }
