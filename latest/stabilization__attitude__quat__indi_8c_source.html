195 static void attitude_run_indi(int32_t indi_commands[], struct Int32Quat *att_err, bool_t in_flight __attribute__((unused)))
196 {
197  //Calculate required angular acceleration
198  struct FloatRates *body_rate = stateGetBodyRates_f();
199  indi.angular_accel_ref.p = reference_acceleration.err_p * QUAT1_FLOAT_OF_BFP(att_err->qx)
200  - reference_acceleration.rate_p * body_rate->p;
201  indi.angular_accel_ref.q = reference_acceleration.err_q * QUAT1_FLOAT_OF_BFP(att_err->qy)
202  - reference_acceleration.rate_q * body_rate->q;
203  indi.angular_accel_ref.r = reference_acceleration.err_r * QUAT1_FLOAT_OF_BFP(att_err->qz)
204  - reference_acceleration.rate_r * body_rate->r;
205 
206  //Incremented in angular acceleration requires increment in control input
207  //G1 is the actuator effectiveness. In the yaw axis, we need something additional: G2.
208  //It takes care of the angular acceleration caused by the change in rotation rate of the propellers
209  //(they have significant inertia, see the paper mentioned in the header for more explanation)
210  indi.du.p = 1.0 / g1.p * (indi.angular_accel_ref.p - indi.filtered_rate_deriv.p);
211  indi.du.q = 1.0 / g1.q * (indi.angular_accel_ref.q - indi.filtered_rate_deriv.q);
212  indi.du.r = 1.0 / (g1.r + g2) * (indi.angular_accel_ref.r - indi.filtered_rate_deriv.r + g2 * indi.du.r);
213 
214  //add the increment to the total control input
215  indi.u_in.p = indi.u.p + indi.du.p;
216  indi.u_in.q = indi.u.q + indi.du.q;
217  indi.u_in.r = indi.u.r + indi.du.r;
218 
219  //bound the total control input
220  Bound(indi.u_in.p, -4500, 4500);
221  Bound(indi.u_in.q, -4500, 4500);
222  Bound(indi.u_in.r, -4500, 4500);
223 
224  //Propagate input filters
225  //first order actuator dynamics
226  indi.u_act_dyn.p = indi.u_act_dyn.p + STABILIZATION_INDI_ACT_DYN_P * (indi.u_in.p - indi.u_act_dyn.p);
227  indi.u_act_dyn.q = indi.u_act_dyn.q + STABILIZATION_INDI_ACT_DYN_Q * (indi.u_in.q - indi.u_act_dyn.q);
228  indi.u_act_dyn.r = indi.u_act_dyn.r + STABILIZATION_INDI_ACT_DYN_R * (indi.u_in.r - indi.u_act_dyn.r);
229 
230  //sensor filter
231  stabilization_indi_second_order_filter(&indi.u_act_dyn, &indi.udotdot, &indi.udot, &indi.u,
232  STABILIZATION_INDI_FILT_OMEGA, STABILIZATION_INDI_FILT_ZETA, STABILIZATION_INDI_FILT_OMEGA_R);
233 
234  //Don't increment if thrust is off
235  if (stabilization_cmd[COMMAND_THRUST] < 300) {
236  FLOAT_RATES_ZERO(indi.u);
237  FLOAT_RATES_ZERO(indi.du);
238  FLOAT_RATES_ZERO(indi.u_act_dyn);
239  FLOAT_RATES_ZERO(indi.u_in);
240  FLOAT_RATES_ZERO(indi.udot);
241  FLOAT_RATES_ZERO(indi.udotdot);
242  } else {
243  lms_estimation();
244  }
245 
246  /* INDI feedback */
247  indi_commands[COMMAND_ROLL] = indi.u_in.p;
248  indi_commands[COMMAND_PITCH] = indi.u_in.q;
249  indi_commands[COMMAND_YAW] = indi.u_in.r;
250 }

252 void stabilization_attitude_run(bool_t enable_integrator)
253 {
254 
255  /* Propagate the second order filter on the gyroscopes */
256  struct FloatRates *body_rates = stateGetBodyRates_f();
257  stabilization_indi_second_order_filter(body_rates, &indi.filtered_rate_2deriv, &indi.filtered_rate_deriv,
258  &indi.filtered_rate, STABILIZATION_INDI_FILT_OMEGA, STABILIZATION_INDI_FILT_ZETA, STABILIZATION_INDI_FILT_OMEGA_R);
259 
260  /* attitude error */
261  struct Int32Quat att_err;
262  struct Int32Quat *att_quat = stateGetNedToBodyQuat_i();
263 // INT32_QUAT_INV_COMP(att_err, *att_quat, stab_att_sp_quat);
264  int32_quat_inv_comp(&att_err, att_quat, &stab_att_sp_quat);
265  /* wrap it in the shortest direction */
266  int32_quat_wrap_shortest(&att_err);
267  int32_quat_normalize(&att_err);
268 
269  /* compute the INDI command */
270  attitude_run_indi(stabilization_att_indi_cmd, &att_err, enable_integrator);
271 
272  stabilization_cmd[COMMAND_ROLL] = stabilization_att_indi_cmd[COMMAND_ROLL];
273  stabilization_cmd[COMMAND_PITCH] = stabilization_att_indi_cmd[COMMAND_PITCH];
274  stabilization_cmd[COMMAND_YAW] = stabilization_att_indi_cmd[COMMAND_YAW];
275 
276  /* bound the result */
277  BoundAbs(stabilization_cmd[COMMAND_ROLL], MAX_PPRZ);
278  BoundAbs(stabilization_cmd[COMMAND_PITCH], MAX_PPRZ);
279  BoundAbs(stabilization_cmd[COMMAND_YAW], MAX_PPRZ);
280 }

294 // This is a simple second order low pass filter
295 void stabilization_indi_second_order_filter(struct FloatRates *input, struct FloatRates *filter_ddx,
296  struct FloatRates *filter_dx, struct FloatRates *filter_x, float omega, float zeta, float omega_r)
297 {
298  float_rates_integrate_fi(filter_x, filter_dx, 1.0 / PERIODIC_FREQUENCY);
299  float_rates_integrate_fi(filter_dx, filter_ddx, 1.0 / PERIODIC_FREQUENCY);
300  float omega2 = omega * omega;
301  float omega2_r = omega_r * omega_r;
302 
303  filter_ddx->p = -filter_dx->p * 2 * zeta * omega + (input->p - filter_x->p) * omega2; \
304  filter_ddx->q = -filter_dx->q * 2 * zeta * omega + (input->q - filter_x->q) * omega2; \
305  filter_ddx->r = -filter_dx->r * 2 * zeta * omega_r + (input->r - filter_x->r) * omega2_r;
306 }

308 // This is a Least Mean Squares adaptive filter
309 // It estiamtes the actuator effectiveness online by comparing the expected angular acceleration based on the inputs with the measured angular acceleration
310 void lms_estimation(void)
311 {
312 
313  // Only pass really low frequencies so you don't adapt to noise
314  float omega = 10.0;
315  float zeta = 0.8;
316  stabilization_indi_second_order_filter(&indi.u_act_dyn, &udotdot_estimation, &udot_estimation, &indi_u_estimation,
317  omega, zeta, omega);
318  struct FloatRates *body_rates = stateGetBodyRates_f();
319  stabilization_indi_second_order_filter(body_rates, &filtered_rate_2deriv_estimation, &filtered_rate_deriv_estimation,
320  &filtered_rate_estimation, omega, zeta, omega);
321 
322  // The inputs are scaled in order to avoid overflows
323  float du = udot_estimation.p * INDI_EST_SCALE;
324  g_est.p = g_est.p - (g_est.p * du - filtered_rate_2deriv_estimation.p) * du * mu;
325  du = udot_estimation.q * INDI_EST_SCALE;
326  g_est.q = g_est.q - (g_est.q * du - filtered_rate_2deriv_estimation.q) * du * mu;
327  du = udot_estimation.r * INDI_EST_SCALE;
328  float ddu = udotdot_estimation.r * INDI_EST_SCALE / PERIODIC_FREQUENCY;
329  float error = (g_est.r * du + g2_est * ddu - filtered_rate_2deriv_estimation.r);
330  g_est.r = g_est.r - error * du * mu / 3;
331  g2_est = g2_est - error * 1000 * ddu * mu / 3;
332 
333  //the g values should be larger than zero, otherwise there is positive feedback, the command will go to max and there is nothing to learn anymore...
334  if (g_est.p < 0.01) { g_est.p = 0.01; }
335  if (g_est.q < 0.01) { g_est.q = 0.01; }
336  if (g_est.r < 0.01) { g_est.r = 0.01; }
337  if (g2_est < 0.01) { g2_est = 0.01; }
338 
339  if (use_adaptive_indi) {
340  //Commit the estimated G values and apply the scaling
341  g1.p = g_est.p * INDI_EST_SCALE;
342  g1.q = g_est.q * INDI_EST_SCALE;
343  g1.r = g_est.r * INDI_EST_SCALE;
344  g2 = g2_est * INDI_EST_SCALE;
345  }
