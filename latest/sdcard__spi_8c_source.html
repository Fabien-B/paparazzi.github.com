93 void sdcard_spi_periodic(struct SDCard *sdcard)
94 {
95 
96  /* Do nothing if spi transaction is in progress */
97  if (sdcard->spi_t.status == SPITransPending || sdcard->spi_t.status == SPITransRunning) {
98  return;
99  }
100 
101  switch (sdcard->status) {
102 
103  /* Send dummy clock to set SD card in SPI mode */
104  case SDCard_BeforeDummyClock:
105  sdcard->spi_t.select = SPINoSelect;
106  sdcard->spi_t.output_length = 10;
107  sdcard->spi_t.input_length = 0;
108  for (uint8_t i = 0; i < 10; i++) {
109  sdcard->output_buf[i] = 0xFF;
110  }
111  sdcard->spi_t.after_cb = &sdcard_spi_spicallback;
112  spi_submit(sdcard->spi_p, &sdcard->spi_t);
113  sdcard->status = SDCard_SendingDummyClock;
114  break;
115 
116  /* Sending ACMD41 */
117  case SDCard_SendingACMD41v2:
118  sdcard_spi_send_app_cmd(sdcard, 41, 0x40000000);
119  sdcard->timeout_counter++;
120  break;
121 
122  /* While busy, keep checking if it is still busy */
123  case SDCard_Busy:
124  sdcard_spi_request_bytes(sdcard, 1);
125  break;
126 
127  /* While waiting for data token, keep polling */
128  case SDCard_WaitingForDataToken:
129  sdcard_spi_request_bytes(sdcard, 1);
130  sdcard->timeout_counter++;
131  break;
132 
133  /* While multiwrite busy, keep checking if it is still busy */
134  case SDCard_MultiWriteBusy:
135  sdcard_spi_request_bytes(sdcard, 1);
136  break;
137 
138  default:
139  break;
140  }

150 void sdcard_spi_spicallback(struct spi_transaction *t)
151 {
152  (void) t; // ignore unused warning
153 
154  switch (sdcard1.status) {
155 
156  /* Ready with dummy clock, proceed with CMD0 */
157  case SDCard_SendingDummyClock:
158  sdcard1.spi_t.select = SPISelectUnselect;
159  sdcard_spi_send_cmd(&sdcard1, 0, 0x00000000);
160  sdcard1.status = SDCard_SendingCMD0;
161  break;
162 
163  /* Ready sending CMD0, start polling result */
164  case SDCard_SendingCMD0:
165  sdcard1.response_counter = 0;
166  sdcard_spi_request_bytes(&sdcard1, 1);
167  sdcard1.status = SDCard_ReadingCMD0Resp;
168  break;
169 
170  /* Continue polling bytes until there is a response or not */
171  case SDCard_ReadingCMD0Resp:
172  if (t->input_buf[0] == 0x01) {
173  sdcard_spi_send_cmd(&sdcard1, 8, 0x000001AA);
174  sdcard1.status = SDCard_SendingCMD8;
175  } else if (sdcard1.response_counter >= 9) {
176  sdcard1.status = SDCard_Error;
177  } else {
178  sdcard_spi_request_bytes(&sdcard1, 1);
179  }
180  break;
181 
182  /* Ready sending CMD8, start polling result */
183  case SDCard_SendingCMD8:
184  sdcard1.response_counter = 0;
185  sdcard_spi_request_bytes(&sdcard1, 1);
186  sdcard1.status = SDCard_ReadingCMD8Resp;
187  break;
188 
189  /* Continue polling bytes until there is a response or not */
190  case SDCard_ReadingCMD8Resp:
191  if (t->input_buf[0] == 0x01) {
192  sdcard_spi_request_bytes(&sdcard1, 4);
193  sdcard1.status = SDCard_ReadingCMD8Parameter;
194  } else if (sdcard1.response_counter >= 9) {
195  sdcard1.status = SDCard_Error;
196  } else {
197  sdcard_spi_request_bytes(&sdcard1, 1);
198  }
199  break;
200 
201  /* Process parameter from CMD8 response (after 0x01 was received) */
202  case SDCard_ReadingCMD8Parameter:
203  if (sdcard1.input_buf[0] == 0x00 && sdcard1.input_buf[1] == 0x00 && sdcard1.input_buf[2] == 0x01
204  && sdcard1.input_buf[3] == 0xAA) {
205  sdcard1.status = SDCard_SendingACMD41v2;
206  sdcard1.timeout_counter = 0;
207  } else {
208  sdcard1.status = SDCard_Error;
209  }
210  break;
211 
212  /* Ready sending the ACMDv2 command, start polling bytes for response */
213  case SDCard_SendingACMD41v2:
214  sdcard1.response_counter = 0;
215  sdcard_spi_request_bytes(&sdcard1, 1);
216  sdcard1.status = SDCard_ReadingACMD41v2Resp;
217  break;
218 
219  /* Grabbing bytes in response to the ACMD41v2 command */
220  case SDCard_ReadingACMD41v2Resp:
221  if (t->input_buf[0] == 0x01) {
222  if (sdcard1.timeout_counter < 500 - 1) { /* Wait 500 cycles until timeout */
223  sdcard1.status = SDCard_SendingACMD41v2;
224  } else {
225  sdcard1.status = SDCard_Error;
226  }
227  } else if (t->input_buf[0] == 0x00) {
228  sdcard_spi_send_cmd(&sdcard1, 58, 0x00000000);
229  sdcard1.status = SDCard_SendingCMD58;
230  } else if (sdcard1.response_counter >= 9) {
231  sdcard1.status = SDCard_Error;
232  } else {
233  sdcard_spi_request_bytes(&sdcard1, 1);
234  }
235  break;
236 
237  /* Ready sending CMD58, request the first byte of R3 response */
238  case SDCard_SendingCMD58:
239  sdcard1.response_counter = 0;
240  sdcard_spi_request_bytes(&sdcard1, 1);
241  sdcard1.status = SDCard_ReadingCMD58Resp;
242  break;
243 
244  /* Continue polling bytes until there is a response to CMD58 */
245  case SDCard_ReadingCMD58Resp:
246  if (sdcard1.response_counter >= 9) {
247  sdcard1.status = SDCard_Error;
248  } else if (sdcard1.input_buf[0] == 0x00) {
249  sdcard_spi_request_bytes(&sdcard1, 4);
250  sdcard1.status = SDCard_ReadingCMD58Parameter;
251  } else {
252  sdcard_spi_request_bytes(&sdcard1, 1);
253  }
254  break;
255 
256  /* Parameter of CMD58 ready, processing it */
257  case SDCard_ReadingCMD58Parameter:
258  if (sdcard1.input_buf[0] & 0x80) { // bit 31 set, CCS bit is valid
259  if (sdcard1.input_buf[0] & 0x40) { // bit 30 set
260  sdcard1.card_type = SDCardType_SdV2block;
261  sdcard1.status = SDCard_Idle;
262  } else { // bit 30 not set
263  sdcard1.card_type = SDCardType_SdV1;
264  sdcard_spi_send_cmd(&sdcard1, 16, SD_BLOCK_SIZE);
265  sdcard1.status = SDCard_SendingCMD16;
266  }
267  } else { // bit 31 not set, CCS bit is unvalid
268  sdcard1.status = SDCard_Error;
269  }
270  break;
271 
272  /* Ready sending CMD16, request first byte */
273  case SDCard_SendingCMD16:
274  sdcard1.response_counter = 0;
275  sdcard_spi_request_bytes(&sdcard1, 1);
276  sdcard1.status = SDCard_ReadingCMD16Resp;
277  break;
278 
279  /* Continue polling bytes until there is a response to CMD16 */
280  case SDCard_ReadingCMD16Resp:
281  if (sdcard1.input_buf[0] == 0x00) {
282  sdcard1.status = SDCard_Idle;
283  } else if (sdcard1.response_counter >= 9) {
284  sdcard1.status = SDCard_Error;
285  } else {
286  sdcard_spi_request_bytes(&sdcard1, 1);
287  }
288  break;
289 
290  /* Ready sending CMD24, request first response byte */
291  case SDCard_SendingCMD24:
292  sdcard1.response_counter = 0;
293  sdcard_spi_request_bytes(&sdcard1, 1);
294  sdcard1.status = SDCard_ReadingCMD24Resp;
295  break;
296 
297  /* Request additional bytes until response or timeout */
298  case SDCard_ReadingCMD24Resp:
299  if (sdcard1.input_buf[0] == 0x00) {
300  sdcard_spi_request_bytes(&sdcard1, 1);
301  sdcard1.status = SDCard_BeforeSendingDataBlock;
302  } else if (sdcard1.response_counter >= 9) {
303  sdcard1.status = SDCard_Error;
304  } else {
305  sdcard_spi_request_bytes(&sdcard1, 1);
306  }
307  break;
308 
309  /* Send the data block */
310  case SDCard_BeforeSendingDataBlock:
311  sdcard1.spi_t.input_length = SD_BLOCK_SIZE + 4;
312  sdcard1.spi_t.output_length = SD_BLOCK_SIZE + 4;
313  sdcard1.spi_t.output_buf = &sdcard1.output_buf[5];
314  sdcard1.spi_t.output_buf[0] = 0xFE; // data token
315  sdcard1.spi_t.output_buf[SD_BLOCK_SIZE + 1] = 0xFF; // Fake CRC byte 1
316  sdcard1.spi_t.output_buf[SD_BLOCK_SIZE + 2] = 0xFF; // Fake CRC byte 2
317  sdcard1.spi_t.output_buf[SD_BLOCK_SIZE + 3] = 0xFF; // to request data response
318  sdcard1.spi_t.after_cb = &sdcard_spi_spicallback;
319  if (spi_submit(sdcard1.spi_p, &sdcard1.spi_t)) {
320  sdcard1.status = SDCard_SendingDataBlock;
321  } else {
322  sdcard1.status = SDCard_Error;
323  }
324  break;
325 
326  /* Finished sending the data block */
327  case SDCard_SendingDataBlock:
328  if ((sdcard1.input_buf[SD_BLOCK_SIZE + 3] & 0x0F) == 0x05) {
329  sdcard1.status = SDCard_Busy;
330  } else {
331  sdcard1.status = SDCard_Error;
332  }
333  sdcard1.spi_t.output_buf = sdcard1.output_buf;
334  break;
335 
336  case SDCard_Busy:
337  if (sdcard1.input_buf[0] != 0x00) {
338  sdcard1.status = SDCard_Idle;
339  }
340  break;
341 
342  /* Ready sending CMD17, request first response byte */
343  case SDCard_SendingCMD17:
344  sdcard1.response_counter = 0;
345  sdcard_spi_request_bytes(&sdcard1, 1);
346  sdcard1.status = SDCard_ReadingCMD17Resp;
347  break;
348 
349  /* Read response to CMD17 until response or timeout/error */
350  case SDCard_ReadingCMD17Resp:
351  if (sdcard1.input_buf[0] == 0x00) {
352  sdcard_spi_request_bytes(&sdcard1, 1);
353  sdcard1.timeout_counter = 0;
354  sdcard1.status = SDCard_WaitingForDataToken;
355  } else if (sdcard1.response_counter >= 9) {
356  sdcard1.status = SDCard_Error;
357  } else {
358  sdcard_spi_request_bytes(&sdcard1, 1);
359  }
360  break;
361 
362  /* Processing byte to see if it is a data token */
363  case SDCard_WaitingForDataToken:
364  if (sdcard1.input_buf[0] == 0xFE) { // Data token received
365  sdcard1.status = SDCard_Idle;
366  sdcard1.spi_t.input_length = SD_BLOCK_SIZE + 2; // 2 Fake CRC bytes
367  sdcard1.spi_t.output_length = SD_BLOCK_SIZE + 2;
368  sdcard1.spi_t.cdiv = SPIDiv8;
369  for (uint16_t i = 0; i < (SD_BLOCK_SIZE + 2); i++) {
370  sdcard1.output_buf[i] = 0xFF;
371  }
372  sdcard1.spi_t.after_cb = &sdcard_spi_spicallback;
373  spi_submit(sdcard1.spi_p, &sdcard1.spi_t);
374  sdcard1.status = SDCard_ReadingDataBlock;
375  } else if (sdcard1.timeout_counter > 498) {
376  sdcard1.status = SDCard_Error;
377  }
378  break;
379 
380  /* Data block received in buffer, process data */
381  case SDCard_ReadingDataBlock:
382  sdcard1.status = SDCard_Idle;
383  if (sdcard1.read_callback != NULL) {
384  sdcard1.read_callback();
385  }
386  break;
387 
388  /* Ready sending CMD25, request first response byte */
389  case SDCard_SendingCMD25:
390  sdcard1.response_counter = 0;
391  sdcard_spi_request_bytes(&sdcard1, 1);
392  sdcard1.status = SDCard_ReadingCMD25Resp;
393  break;
394 
395  /* Request additional bytes until response or timeout */
396  case SDCard_ReadingCMD25Resp:
397  if (sdcard1.input_buf[0] == 0x00) {
398  sdcard_spi_request_bytes(&sdcard1, 1);
399  sdcard1.status = SDCard_MultiWriteIdle;
400  } else if (sdcard1.response_counter >= 9) {
401  sdcard1.status = SDCard_Error;
402  } else {
403  sdcard_spi_request_bytes(&sdcard1, 1);
404  }
405  break;
406 
407  /* Check the response after a block was written */
408  case SDCard_MultiWriteWriting:
409  if ((sdcard1.input_buf[SD_BLOCK_SIZE + 3] & 0x0F) == 0x05 /* Data accepted */) {
410  sdcard1.status = SDCard_MultiWriteBusy;
411  } else {
412  sdcard1.status = SDCard_Error;
413  }
414  break;
415 
416  /* Check if sd card is still busy after a packet during multiwrite */
417  case SDCard_MultiWriteBusy:
418  if (sdcard1.input_buf[0] != 0x00) {
419  sdcard1.status = SDCard_MultiWriteIdle;
420  }
421  break;
422 
423  /* Finished the stop command transaction, waiting for internal processes of the SD card */
424  case SDCard_MultiWriteStopping:
425  sdcard1.status = SDCard_Busy;
426  break;
427 
428  /* Should not reach this */
429  default:
430  break;
431  }
432 }

443 void sdcard_spi_send_cmd(struct SDCard *sdcard, uint8_t cmd, uint32_t arg)
444 {
445  (void) cmd; (void) arg;
446  sdcard->spi_t.input_length = 6;
447  sdcard->spi_t.output_length = 6;
448  sdcard->output_buf[0] = 0x40 | cmd;
449  sdcard->output_buf[1] = arg >> 24;
450  sdcard->output_buf[2] = arg >> 16;
451  sdcard->output_buf[3] = arg >> 8;
452  sdcard->output_buf[4] = arg;
453  switch (cmd) {
454  case 0:
455  sdcard->output_buf[5] = 0x95; /* CRC byte for CMD0 */
456  break;
457  case 8:
458  sdcard->output_buf[5] = 0x87; /* CRC byte for CMD8 */
459  break;
460  default:
461  sdcard->output_buf[5] = 0x01; /* Fake CRC byte, will be ignored by the SDCard */
462  }
463  sdcard->spi_t.after_cb = &sdcard_spi_spicallback;
464 
465  spi_submit(sdcard->spi_p, &sdcard->spi_t);
466 }

478 void sdcard_spi_send_app_cmd(struct SDCard *sdcard, uint8_t cmd, uint32_t arg)
479 {
480  sdcard->spi_t.output_length = 21;
481  sdcard->spi_t.input_length = 21;
482 
483  sdcard->output_buf[0] = 0x77; /* Begin CMD55 */
484  sdcard->output_buf[1] = 0x00;
485  sdcard->output_buf[2] = 0x00;
486  sdcard->output_buf[3] = 0x00;
487  sdcard->output_buf[4] = 0x00;
488  sdcard->output_buf[5] = 0x01; /* End CMD55 */
489  sdcard->output_buf[6] = 0xFF; /* Somewhere during the following bytes, response to CMD55 is received. */
490  sdcard->output_buf[7] = 0xFF;
491  sdcard->output_buf[8] = 0xFF;
492  sdcard->output_buf[9] = 0xFF;
493  sdcard->output_buf[10] = 0xFF;
494  sdcard->output_buf[11] = 0xFF;
495  sdcard->output_buf[12] = 0xFF;
496  sdcard->output_buf[13] = 0xFF;
497  sdcard->output_buf[14] = 0xFF;
498 
499  sdcard->output_buf[15] = 0x40 + cmd; /* Begin of the ACMD */
500  sdcard->output_buf[16] = arg >> 24;
501  sdcard->output_buf[17] = arg >> 16;
502  sdcard->output_buf[18] = arg >> 8;
503  sdcard->output_buf[19] = arg;
504  sdcard->output_buf[20] = 0x01; /* End of the ACMD */
505 
506  /* Set function to be called after transfer is finished */
