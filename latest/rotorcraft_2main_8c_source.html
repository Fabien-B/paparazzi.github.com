195 STATIC_INLINE void handle_periodic_tasks(void)
196 {
197  if (sys_time_check_and_ack_timer(main_periodic_tid)) {
198  main_periodic();
199  }
200  if (sys_time_check_and_ack_timer(modules_tid)) {
201  modules_periodic_task();
202  }
203  if (sys_time_check_and_ack_timer(radio_control_tid)) {
204  radio_control_periodic_task();
205  }
206  if (sys_time_check_and_ack_timer(failsafe_tid)) {
207  failsafe_check();
208  }
209  if (sys_time_check_and_ack_timer(electrical_tid)) {
210  electrical_periodic();
211  }
212  if (sys_time_check_and_ack_timer(telemetry_tid)) {
213  telemetry_periodic();
214  }
215 #if USE_BARO_BOARD
216  if (sys_time_check_and_ack_timer(baro_tid)) {
217  baro_periodic();
218  }
219 #endif
220 }

268 STATIC_INLINE void failsafe_check(void)
269 {
270  if (radio_control.status == RC_REALLY_LOST &&
271  autopilot_mode != AP_MODE_KILL &&
272  autopilot_mode != AP_MODE_HOME &&
273  autopilot_mode != AP_MODE_FAILSAFE &&
274  autopilot_mode != AP_MODE_NAV) {
275  autopilot_set_mode(RC_LOST_MODE);
276  }
277 
278 #if FAILSAFE_ON_BAT_CRITICAL
279  if (autopilot_mode != AP_MODE_KILL &&
280  electrical.bat_critical) {
281  autopilot_set_mode(AP_MODE_FAILSAFE);
282  }
283 #endif
284 
285 #if USE_GPS
286  gps_periodic_check();
287  if (autopilot_mode == AP_MODE_NAV &&
288  autopilot_motors_on &&
289 #if NO_GPS_LOST_WITH_RC_VALID
290  radio_control.status != RC_OK &&
291 #endif
292  GpsIsLost()) {
293  autopilot_set_mode(AP_MODE_FAILSAFE);
294  }
295 
296  if (autopilot_mode == AP_MODE_HOME &&
297  autopilot_motors_on && GpsIsLost()) {
298  autopilot_set_mode(AP_MODE_FAILSAFE);
299  }
300 #endif
301 
302  autopilot_check_in_flight(autopilot_motors_on);
303 }

346 static inline void on_accel_event(void)
347 {
348 #if USE_AUTO_AHRS_FREQ || !defined(AHRS_CORRECT_FREQUENCY)
349  PRINT_CONFIG_MSG("Calculating dt for AHRS accel update.")
350  // timestamp in usec when last callback was received
351  static uint32_t last_ts = 0;
352  // current timestamp
353  uint32_t now_ts = get_sys_time_usec();
354  // dt between this and last callback
355  float dt = (float)(now_ts - last_ts) / 1e6;
356  last_ts = now_ts;
357 #else
358  PRINT_CONFIG_MSG("Using fixed AHRS_CORRECT_FREQUENCY for AHRS accel update.")
359  PRINT_CONFIG_VAR(AHRS_CORRECT_FREQUENCY)
360  const float dt = 1. / (AHRS_CORRECT_FREQUENCY);
361 #endif
362 
363  imu_scale_accel(&imu);
364 
365  if (ahrs.status != AHRS_UNINIT) {
366  ahrs_update_accel(dt);
367  }
368 }

370 static inline void on_gyro_event(void)
371 {
372 #if USE_AUTO_AHRS_FREQ || !defined(AHRS_PROPAGATE_FREQUENCY)
373  PRINT_CONFIG_MSG("Calculating dt for AHRS/INS propagation.")
374  // timestamp in usec when last callback was received
375  static uint32_t last_ts = 0;
376  // current timestamp
377  uint32_t now_ts = get_sys_time_usec();
378  // dt between this and last callback in seconds
379  float dt = (float)(now_ts - last_ts) / 1e6;
380  last_ts = now_ts;
381 #else
382  PRINT_CONFIG_MSG("Using fixed AHRS_PROPAGATE_FREQUENCY for AHRS/INS propagation.")
383  PRINT_CONFIG_VAR(AHRS_PROPAGATE_FREQUENCY)
384  const float dt = 1. / (AHRS_PROPAGATE_FREQUENCY);
385 #endif
386 
387  imu_scale_gyro(&imu);
388 
389  if (ahrs.status == AHRS_UNINIT) {
390  ahrs_aligner_run();
391  if (ahrs_aligner.status == AHRS_ALIGNER_LOCKED) {
392  ahrs_align();
393  }
394  } else {
395  ahrs_propagate(dt);
396 #ifdef SITL
397  if (nps_bypass_ahrs) { sim_overwrite_ahrs(); }
398 #endif
399  ins_propagate(dt);
400  }
401 #ifdef USE_VEHICLE_INTERFACE
402  vi_notify_imu_available();
403 #endif
404 }
