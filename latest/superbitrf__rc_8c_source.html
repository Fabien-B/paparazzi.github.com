38 static void superbitrf_rc_normalize(int16_t *in, int16_t *out, uint8_t count)
39 {
40  uint8_t i;
41  for (i = 0; i < count; i++) {
42  if (i == RADIO_THROTTLE) {
43  out[i] = (in[i] + MAX_PPRZ) / 2;
44  Bound(out[i], 0, MAX_PPRZ);
45  } else {
46  out[i] = in[i];
47  Bound(out[i], MIN_PPRZ, MAX_PPRZ);
48  }
49  }
50 }
