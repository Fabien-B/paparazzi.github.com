96 void mf_ptu_periodic(void)
97 {
98  // Read ADC
99  pressure_adc = pressure_buf.sum / pressure_buf.av_nb_sample;
100  temp_adc = temp_buf.sum / temp_buf.av_nb_sample;
101  // Read PWM
102  humid_period = USEC_OF_PWM_INPUT_TICKS(pwm_input_period_tics[PWM_INPUT_CHANNEL_HUMIDITY]);
103 
104  // Log data
105 #if LOG_PTU
106  if (pprzLogFile != -1) {
107  if (!log_ptu_started) {
108  sdLogWriteLog(pprzLogFile,
109  "P(adc) T(adc) H(usec) GPS_fix TOW(ms) Week Lat(1e7rad) Lon(1e7rad) HMSL(mm) gpseed(cm/s) course(1e7rad) climb(cm/s)\n");
110  log_ptu_started = TRUE;
111  } else {
112  sdLogWriteLog(pprzLogFile, "%d %d %d %d %d %d %d %d %d %d %d %d\n",
113  pressure_adc, temp_adc, humid_period,
114  gps.fix, gps.tow, gps.week,
115  gps.lla_pos.lat, gps.lla_pos.lon, gps.hmsl,
116  gps.gspeed, gps.course, -gps.ned_vel.z);
117  }
118  }
119 #endif
120 
121  // Send data
122 #if SEND_PTU
123 #define PTU_DATA_SIZE 3
124  float ptu_data[PTU_DATA_SIZE];
125  ptu_data[0] = (float)(PTU_PRESSURE_SCALE * ((int16_t)pressure_adc - PTU_PRESSURE_OFFSET));
126  ptu_data[1] = (float)(PTU_TEMPERATURE_SCALE * ((int16_t)temp_adc - PTU_TEMPERATURE_OFFSET));
127  ptu_data[2] = (float)(PTU_HUMIDTY_SCALE * ((int16_t)humid_period - PTU_HUMIDTY_OFFSET));
128  DOWNLINK_SEND_PAYLOAD_FLOAT(DefaultChannel, DefaultDevice, PTU_DATA_SIZE, ptu_data);
129 #endif
130 }
