103 bool_t viewvideo_function(struct image_t *img)
104 {
105  // Resize image if needed
106  struct image_t img_small;
107  image_create(&img_small,
108  img->w / viewvideo.downsize_factor,
109  img->h / viewvideo.downsize_factor,
110  IMAGE_YUV422);
111 
112  // Create the JPEG encoded image
113  struct image_t img_jpeg;
114  image_create(&img_jpeg, img_small.w, img_small.h, IMAGE_JPEG);
115 
116 #if VIEWVIDEO_USE_NETCAT
117  char nc_cmd[64];
118  sprintf(nc_cmd, "nc %s %d 2>/dev/null", STRINGIFY(VIEWVIDEO_HOST), VIEWVIDEO_PORT_OUT);
119 #else
120 #endif
121 
122  if (viewvideo.is_streaming) {
123 
124  // Only resize when needed
125  if (viewvideo.downsize_factor != 1) {
126  image_yuv422_downsample(img, &img_small, viewvideo.downsize_factor);
127  jpeg_encode_image(&img_small, &img_jpeg, VIEWVIDEO_QUALITY_FACTOR, VIEWVIDEO_USE_NETCAT);
128  } else {
129  jpeg_encode_image(img, &img_jpeg, VIEWVIDEO_QUALITY_FACTOR, VIEWVIDEO_USE_NETCAT);
130  }
131 
132 #if VIEWVIDEO_USE_NETCAT
133  // Open process to send using netcat (in a fork because sometimes kills itself???)
134  pid_t pid = fork();
135 
136  if (pid < 0) {
137  printf("[viewvideo] Could not create netcat fork.\n");
138  } else if (pid == 0) {
139  // We are the child and want to send the image
140  FILE *netcat = popen(nc_cmd, "w");
141  if (netcat != NULL) {
142  fwrite(jpegbuf, sizeof(uint8_t), size, netcat);
143  pclose(netcat); // Ignore output, because it is too much when not connected
144  } else {
145  printf("[viewvideo] Failed to open netcat process.\n");
146  }
147 
148  // Exit the program since we don't want to continue after transmitting
149  exit(0);
150  } else {
151  // We want to wait until the child is finished
152  wait(NULL);
153  }
154 #else
155  if (viewvideo.use_rtp) {
156 
157  // Send image with RTP
158  rtp_frame_send(
159  &video_sock, // UDP socket
160  &img_jpeg,
161  0, // Format 422
162  VIEWVIDEO_QUALITY_FACTOR, // Jpeg-Quality
163  0, // DRI Header
164  VIEWVIDEO_RTP_TIME_INC // 90kHz time increment
165  );
166  // Extra note: when the time increment is set to 0,
167  // it is automaticaly calculated by the send_rtp_frame function
168  // based on gettimeofday value. This seems to introduce some lag or jitter.
169  // An other way is to compute the time increment and set the correct value.
170  // It seems that a lower value is also working (when the frame is received
171  // the timestamp is always "late" so the frame is displayed immediately).
172  // Here, we set the time increment to the lowest possible value
173  // (1 = 1/90000 s) which is probably stupid but is actually working.
174  }
175 #endif
176 
177  }
178 
179  // Free all buffers
180  image_free(&img_jpeg);
181  image_free(&img_small);
182  return TRUE;
183 }

188 void viewvideo_init(void)
189 {
190  char save_name[512];
191 // struct UdpSocket video_sock;
192  udp_socket_create(&video_sock, STRINGIFY(VIEWVIDEO_HOST), VIEWVIDEO_PORT_OUT, -1, VIEWVIDEO_BROADCAST);
193 
194  cv_add(viewvideo_function);
195 
196  viewvideo.is_streaming = TRUE;
197 
198 #if VIEWVIDEO_USE_NETCAT
199  // Create an Netcat receiver file for the streaming
200  sprintf(save_name, "%s/netcat-recv.sh", STRINGIFY(VIEWVIDEO_SHOT_PATH));
201  FILE *fp = fopen(save_name, "w");
202  if (fp != NULL) {
203  fprintf(fp, "i=0\n");
204  fprintf(fp, "while true\n");
205  fprintf(fp, "do\n");
206  fprintf(fp, "\tn=$(printf \"%%04d\" $i)\n");
207  fprintf(fp, "\tnc -l 0.0.0.0 %d > img_${n}.jpg\n", (int)(VIEWVIDEO_PORT_OUT));
208  fprintf(fp, "\ti=$((i+1))\n");
209  fprintf(fp, "done\n");
210  fclose(fp);
211  }
212 #else
213  // Create an SDP file for the streaming
214  sprintf(save_name, "%s/stream.sdp", STRINGIFY(VIEWVIDEO_SHOT_PATH));
215  FILE *fp = fopen(save_name, "w");
216  if (fp != NULL) {
217  fprintf(fp, "v=0\n");
218  fprintf(fp, "m=video %d RTP/AVP 26\n", (int)(VIEWVIDEO_PORT_OUT));
219  fprintf(fp, "c=IN IP4 0.0.0.0\n");
220  fclose(fp);
221  }
222 #endif
223 }
