75 bool_t InitializeSpiral(uint8_t CenterWP, uint8_t EdgeWP, float StartRad, float IncRad, float Segments, float ZKoord)
76 {
77  Center = CenterWP; // center of the helix
78  Edge = EdgeWP; // edge point on the maximaum radius
79  SRad = StartRad; // start radius of the helix
80  Segmente = Segments;
81  ZPoint = ZKoord;
82  nav_radius_min = MIN_CIRCLE_RADIUS;
83  if (SRad < nav_radius_min) SRad = nav_radius_min;
84  IRad = IncRad; // multiplier for increasing the spiral
85 
86  EdgeCurrentX = WaypointX(Edge) - WaypointX(Center);
87  EdgeCurrentY = WaypointY(Edge) - WaypointY(Center);
88 
89  Spiralradius = sqrt(EdgeCurrentX*EdgeCurrentX+EdgeCurrentY*EdgeCurrentY);
90 
91  TransCurrentX = stateGetPositionEnu_f()->x - WaypointX(Center);
92  TransCurrentY = stateGetPositionEnu_f()->y - WaypointY(Center);
93  TransCurrentZ = stateGetPositionUtm_f()->alt - ZPoint;
94  DistanceFromCenter = sqrt(TransCurrentX*TransCurrentX+TransCurrentY*TransCurrentY);
95 
96  // SpiralTheta = atan2(TransCurrentY,TransCurrentX);
97  // Fly2X = Spiralradius*cos(SpiralTheta+M_PI)+WaypointX(Center);
98  // Fly2Y = Spiralradius*sin(SpiralTheta+M_PI)+WaypointY(Center);
99 
100  // Alphalimit denotes angle, where the radius will be increased
101  Alphalimit = 2*M_PI / Segments;
102  //current position
103  FlyFromX = stateGetPositionEnu_f()->x;
104  FlyFromY = stateGetPositionEnu_f()->y;
105 
106  if(DistanceFromCenter > Spiralradius)
107  CSpiralStatus = Outside;
108  return FALSE;
109 }

111 bool_t SpiralNav(void)
112 {
113  TransCurrentX = stateGetPositionEnu_f()->x - WaypointX(Center);
114  TransCurrentY = stateGetPositionEnu_f()->y - WaypointY(Center);
115  DistanceFromCenter = sqrt(TransCurrentX*TransCurrentX+TransCurrentY*TransCurrentY);
116 
117  float DistanceStartEstim;
118  float CircleAlpha;
119 
120  switch(CSpiralStatus)
121  {
122  case Outside:
123  //flys until center of the helix is reached an start helix
124  nav_route_xy(FlyFromX,FlyFromY,WaypointX(Center), WaypointY(Center));
125  // center reached?
126  if (nav_approaching_xy(WaypointX(Center), WaypointY(Center), FlyFromX, FlyFromY, 0)) {
127  // nadir image
128 #ifdef DIGITAL_CAM
129  dc_send_command(DC_SHOOT);
130 #endif
131  CSpiralStatus = StartCircle;
132  }
133  break;
134  case StartCircle:
135  // Starts helix
136  // storage of current coordinates
137  // calculation needed, State switch to Circle
138  nav_circle_XY(WaypointX(Center), WaypointY(Center), SRad);
139  if(DistanceFromCenter >= SRad){
140  LastCircleX = stateGetPositionEnu_f()->x;
141  LastCircleY = stateGetPositionEnu_f()->y;
142  CSpiralStatus = Circle;
143  // Start helix
144 #ifdef DIGITAL_CAM
145  dc_Circle(360/Segmente);
146 #endif
147  }
148  break;
149  case Circle: {
150  nav_circle_XY(WaypointX(Center), WaypointY(Center), SRad);
151  // Trigonometrische Berechnung des bereits geflogenen Winkels alpha
152  // equation:
153  // alpha = 2 * asin ( |Starting position angular - current positon| / (2* SRad)
154  // if alphamax already reached, increase radius.
155 
156  //DistanceStartEstim = |Starting position angular - current positon|
157  DistanceStartEstim = sqrt (((LastCircleX-stateGetPositionEnu_f()->x)*(LastCircleX-stateGetPositionEnu_f()->x))
158  + ((LastCircleY-stateGetPositionEnu_f()->y)*(LastCircleY-stateGetPositionEnu_f()->y)));
159  CircleAlpha = (2.0 * asin (DistanceStartEstim / (2 * SRad)));
160  if (CircleAlpha >= Alphalimit) {
161  LastCircleX = stateGetPositionEnu_f()->x;
162  LastCircleY = stateGetPositionEnu_f()->y;
163  CSpiralStatus = IncSpiral;
164  }
165  break;
166  }
167  case IncSpiral:
168  // increasing circle radius as long as it is smaller than max helix radius
169  if(SRad + IRad < Spiralradius)
170  {
171  SRad = SRad + IRad;
172 #ifdef DIGITAL_CAM
173  if (dc_cam_tracing) {
174  // calculating Cam angle for camera alignment
175  TransCurrentZ = stateGetPositionUtm_f()->alt - ZPoint;
176  dc_cam_angle = atan(SRad/TransCurrentZ) * 180 / M_PI;
177  }
178 #endif
179  }
180  else {
181  SRad = Spiralradius;
182 #ifdef DIGITAL_CAM
183  // Stopps DC
184  dc_stop();
185 #endif
186  }
187  CSpiralStatus = Circle;
188  break;
189  default:
190  break;
191  }
192  return TRUE;
193 }
