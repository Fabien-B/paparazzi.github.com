94 void imu_apogee_event( void )
95 {
96  // If the itg3200 I2C transaction has succeeded: convert the data
97  mpu60x0_i2c_event(&imu_apogee.mpu);
98  if (imu_apogee.mpu.data_available) {
99  RATES_ASSIGN(imu.gyro_unscaled, imu_apogee.mpu.data_rates.rates.p, -imu_apogee.mpu.data_rates.rates.q, -imu_apogee.mpu.data_rates.rates.r);
100  VECT3_ASSIGN(imu.accel_unscaled, imu_apogee.mpu.data_accel.vect.x, -imu_apogee.mpu.data_accel.vect.y, -imu_apogee.mpu.data_accel.vect.z);
101  imu_apogee.mpu.data_available = FALSE;
102  imu_apogee.gyr_valid = TRUE;
103  imu_apogee.acc_valid = TRUE;
104  }
