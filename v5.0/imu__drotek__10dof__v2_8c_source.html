70 void imu_impl_init(void)
71 {
72  /* MPU-60X0 */
73  mpu60x0_i2c_init(&imu_drotek2.mpu, &(DROTEK_2_I2C_DEV), DROTEK_2_MPU_I2C_ADDR);
74  // change the default configuration
75  imu_drotek2.mpu.config.smplrt_div = DROTEK_2_SMPLRT_DIV;
76  imu_drotek2.mpu.config.dlpf_cfg = DROTEK_2_LOWPASS_FILTER;
77  imu_drotek2.mpu.config.gyro_range = DROTEK_2_GYRO_RANGE;
78  imu_drotek2.mpu.config.accel_range = DROTEK_2_ACCEL_RANGE;
79 
80  /* HMC5883 magnetometer */
81  hmc58xx_init(&imu_drotek2.hmc, &(DROTEK_2_I2C_DEV), DROTEK_2_HMC_I2C_ADDR);
82 
83  /* mag is declared as slave to call the configure function,
84  * regardless if it is an actual MPU slave or passthrough
85  */
86  imu_drotek2.mpu.config.nb_slaves = 1;
87  /* set callback function to configure mag */
88  imu_drotek2.mpu.config.slaves[0].configure = &imu_drotek2_configure_mag_slave;
89 
90  // use hmc mag via passthrough
91  imu_drotek2.mpu.config.i2c_bypass = TRUE;
92 
93  imu_drotek2.gyro_valid = FALSE;
94  imu_drotek2.accel_valid = FALSE;
95  imu_drotek2.mag_valid = FALSE;
96 }

98 void imu_periodic(void)
99 {
100  // Start reading the latest gyroscope data
101  mpu60x0_i2c_periodic(&imu_drotek2.mpu);
102 
103  // Read HMC58XX at ~50Hz (main loop for rotorcraft: 512Hz)
104  if (imu_drotek2.mpu.config.initialized) {
105  RunOnceEvery(10, hmc58xx_read(&imu_drotek2.hmc));
106  }

109 void imu_drotek2_event(void)
110 {
111  // If the MPU6050 I2C transaction has succeeded: convert the data
112  mpu60x0_i2c_event(&imu_drotek2.mpu);
113 
114  if (imu_drotek2.mpu.data_available) {
115 #if IMU_DROTEK_2_ORIENTATION_IC_UP
116  /* change orientation, so if ICs face up, z-axis is down */
117  imu.gyro_unscaled.p = imu_drotek2.mpu.data_rates.rates.p;
118  imu.gyro_unscaled.q = -imu_drotek2.mpu.data_rates.rates.q;
119  imu.gyro_unscaled.r = -imu_drotek2.mpu.data_rates.rates.r;
120  imu.accel_unscaled.x = imu_drotek2.mpu.data_accel.vect.x;
121  imu.accel_unscaled.y = -imu_drotek2.mpu.data_accel.vect.y;
122  imu.accel_unscaled.z = -imu_drotek2.mpu.data_accel.vect.z;
123 #else
124  /* default orientation as should be printed on the pcb, z-down, ICs down */
125  RATES_COPY(imu.gyro_unscaled, imu_drotek2.mpu.data_rates.rates);
126  VECT3_COPY(imu.accel_unscaled, imu_drotek2.mpu.data_accel.vect);
127 #endif
128 
129  imu_drotek2.mpu.data_available = FALSE;
130  imu_drotek2.gyro_valid = TRUE;
131  imu_drotek2.accel_valid = TRUE;
132  }
133 
134  /* HMC58XX event task */
135  hmc58xx_event(&imu_drotek2.hmc);
136  if (imu_drotek2.hmc.data_available) {
137 #if IMU_DROTEK_2_ORIENTATION_IC_UP
138  imu.mag_unscaled.x = imu_drotek2.hmc.data.vect.x;
139  imu.mag_unscaled.y = -imu_drotek2.hmc.data.vect.y;
140  imu.mag_unscaled.z = -imu_drotek2.hmc.data.vect.z;
141 #else
142  VECT3_COPY(imu.mag_unscaled, imu_drotek2.hmc.data.vect);
143 #endif
144  imu_drotek2.hmc.data_available = FALSE;
145  imu_drotek2.mag_valid = TRUE;
146  }
147 }
