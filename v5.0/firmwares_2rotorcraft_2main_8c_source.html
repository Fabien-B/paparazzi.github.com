177 STATIC_INLINE void handle_periodic_tasks( void ) {
178  if (sys_time_check_and_ack_timer(main_periodic_tid))
179  main_periodic();
180  if (sys_time_check_and_ack_timer(modules_tid))
181  modules_periodic_task();
182  if (sys_time_check_and_ack_timer(radio_control_tid))
183  radio_control_periodic_task();
184  if (sys_time_check_and_ack_timer(failsafe_tid))
185  failsafe_check();
186  if (sys_time_check_and_ack_timer(electrical_tid))
187  electrical_periodic();
188  if (sys_time_check_and_ack_timer(baro_tid))
189  baro_periodic();
190  if (sys_time_check_and_ack_timer(telemetry_tid))
191  telemetry_periodic();
192 }

215 STATIC_INLINE void failsafe_check( void ) {
216  if (radio_control.status != RC_OK &&
217  autopilot_mode != AP_MODE_KILL &&
218  autopilot_mode != AP_MODE_NAV)
219  {
220  autopilot_set_mode(AP_MODE_FAILSAFE);
221  }
222 
223 #if USE_GPS
224  if (autopilot_mode == AP_MODE_NAV &&
225  autopilot_motors_on &&
226 #if NO_GPS_LOST_WITH_RC_VALID
227  radio_control.status != RC_OK &&
228 #endif
229  GpsIsLost())
230  {
231  autopilot_set_mode(AP_MODE_FAILSAFE);
232  }
233 #endif
234 }
