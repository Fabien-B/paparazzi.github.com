101 void ins_propagate() {
102  /* untilt accels and speeds */
103  FLOAT_RMAT_VECT3_TRANSP_MUL(ins_ltp_accel, (*stateGetNedToBodyRMat_f()), ahrs_impl.accel);
104  FLOAT_RMAT_VECT3_TRANSP_MUL(ins_ltp_speed, (*stateGetNedToBodyRMat_f()), ahrs_impl.speed);
105 
106  //Add g to the accelerations
107  ins_ltp_accel.z += 9.81;
108 
109  //Save the accelerations and speeds
110  stateSetAccelNed_f(&ins_ltp_accel);
111  stateSetSpeedNed_f(&ins_ltp_speed);
112 
113  //Don't set the height if we use the one from the gps
114 #if !USE_GPS_HEIGHT
115  //Set the height and save the position
116  ins_ltp_pos.z = -(ahrs_impl.altitude * INT32_POS_OF_CM_NUM) / INT32_POS_OF_CM_DEN;
117  stateSetPositionNed_i(&ins_ltp_pos);
