115 void stabilization_rate_read_rc(void)
116 {
117  if (ROLL_RATE_DEADBAND_EXCEEDED()) {
118  stabilization_rate_sp.p = (float) radio_control.values[RADIO_ROLL] * STABILIZATION_INDI_MAX_RATE / MAX_PPRZ;
119  } else {
120  stabilization_rate_sp.p = 0;
121  }
122 
123  if (PITCH_RATE_DEADBAND_EXCEEDED()) {
124  stabilization_rate_sp.q = (float) radio_control.values[RADIO_PITCH] * STABILIZATION_INDI_MAX_RATE / MAX_PPRZ;
125  } else {
126  stabilization_rate_sp.q = 0;
127  }
128 
129  if (YAW_RATE_DEADBAND_EXCEEDED() && !THROTTLE_STICK_DOWN()) {
130  stabilization_rate_sp.r = (float) radio_control.values[RADIO_YAW] * STABILIZATION_INDI_MAX_RATE / MAX_PPRZ;
131  } else {
132  stabilization_rate_sp.r = 0;
133  }
134 }
