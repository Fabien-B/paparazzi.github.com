76 static inline void autopilot_arming_set(bool motors_on)
77 {
78  if (motors_on) {
79  autopilot_check_motor_status = STATUS_MOTORS_ON;
80  } else if (kill_switch_is_on() && (autopilot_check_motor_status == STATUS_MOTORS_ON)) {

84  autopilot_check_motor_status = STATUS_MOTORS_AUTOMATICALLY_OFF;

96 static inline bool autopilot_arming_check_valid(bool yaw_must_be_centered)
97 {
98  if (!THROTTLE_STICK_DOWN()) {
99  autopilot.arming_status = AP_ARMING_STATUS_THROTTLE_NOT_DOWN;
100  } else if (!PITCH_STICK_CENTERED()) {
101  autopilot.arming_status = AP_ARMING_STATUS_PITCH_NOT_CENTERED;
102  } else if (!ROLL_STICK_CENTERED()) {
103  autopilot.arming_status = AP_ARMING_STATUS_ROLL_NOT_CENTERED;
104  } else {
105  if (yaw_must_be_centered && !YAW_STICK_CENTERED()) {
106  autopilot.arming_status = AP_ARMING_STATUS_YAW_NOT_CENTERED;
107  } else if (!yaw_must_be_centered && YAW_STICK_CENTERED()) {
108  autopilot.arming_status = AP_ARMING_STATUS_YAW_CENTERED;
109  } else {
110  return true; // all checks valid
111  }
112  }
113  return false; // one of the checks failed
114 }

122 static inline void autopilot_arming_check_motors_on(void)
123 {
124  /* only allow switching motor if not in KILL mode */
125  if (autopilot_get_mode() != AP_MODE_KILL) {
126 
127  switch (autopilot_check_motor_status) {
128  case STATUS_INITIALISE_RC: // Wait until RC is initialised (it being centered is a good pointer to this)
129  if (autopilot_arming_check_valid(YAW_MUST_BE_CENTERED)) {
130  autopilot_check_motor_status = STATUS_MOTORS_OFF;
131  }
132  break;
133  case STATUS_MOTORS_RC_KILLED: // Motors were killed by kill mode
134  // If the vehicle was killed accidentally, allow rapid re-arm
135  if ( (get_sys_time_float() - motor_kill_time) < MOTOR_RE_ARM_TIME) {
136  autopilot_check_motor_status = STATUS_MOTORS_ON;
137  } else {
138  autopilot_check_motor_status = STATUS_MOTORS_AUTOMATICALLY_OFF;
139  autopilot.motors_on = false;
140  }
141  break;
142  case STATUS_MOTORS_AUTOMATICALLY_OFF: // Motors were disarmed externally
143  //(possibly due to crash)
144  //wait extra delay before enabling the normal arming state machine
145  autopilot.motors_on = false;
146  autopilot_motors_on_counter = 0;
147  if (autopilot_arming_check_valid(YAW_MUST_BE_CENTERED)) {
148  autopilot_check_motor_status = STATUS_MOTORS_AUTOMATICALLY_OFF_SAFETY_WAIT;
149  }
150  break;
151  case STATUS_MOTORS_AUTOMATICALLY_OFF_SAFETY_WAIT:
152  autopilot_motors_on_counter++;
153  if (autopilot_motors_on_counter >= MOTOR_ARMING_DELAY) {
154  autopilot_check_motor_status = STATUS_MOTORS_OFF;
155  } else {
156  autopilot.arming_status = AP_ARMING_STATUS_WAITING;
157  }
158  break;
159  case STATUS_MOTORS_OFF:
160  autopilot.motors_on = false;
161  autopilot_motors_on_counter = 0;
162  autopilot.arming_status = AP_ARMING_STATUS_WAITING;
163  if (autopilot_arming_check_valid(YAW_MUST_BE_PUSHED)) { // stick pushed
164  autopilot_check_motor_status = STATUS_M_OFF_STICK_PUSHED;
165  }
166  break;
167  case STATUS_M_OFF_STICK_PUSHED:
168  autopilot.motors_on = false;
169  autopilot_motors_on_counter++;
170  if (autopilot_motors_on_counter >= MOTOR_ARMING_DELAY) {
171  autopilot_check_motor_status = STATUS_START_MOTORS;
172  } else if (!autopilot_arming_check_valid(YAW_MUST_BE_PUSHED)) { // stick released too soon
173  autopilot_check_motor_status = STATUS_MOTORS_OFF;
174  } else {
175  autopilot.arming_status = AP_ARMING_STATUS_WAITING;
176  }
177  break;
178  case STATUS_START_MOTORS:
179  autopilot.motors_on = true;
180  autopilot_motors_on_counter = MOTOR_ARMING_DELAY;
181  autopilot_set_in_flight(false); // stop fc from starting control (integration and yaw) till arm process is complete
182  if (YAW_STICK_CENTERED()) { // wait until stick released
183  autopilot_check_motor_status = STATUS_MOTORS_ON;
184  }
185  break;
186  case STATUS_MOTORS_ON:
187  autopilot.arming_status = AP_ARMING_STATUS_ARMED;
188  autopilot.motors_on = true;
189  autopilot_motors_on_counter = MOTOR_ARMING_DELAY;
190  if (THROTTLE_STICK_DOWN() && YAW_STICK_PUSHED()) { // stick pushed
191  autopilot_check_motor_status = STATUS_M_ON_STICK_PUSHED;
192  }
193  break;
194  case STATUS_M_ON_STICK_PUSHED:
195  autopilot.motors_on = true;
196  autopilot_motors_on_counter--;
197  if (autopilot_motors_on_counter == 0) {
198  autopilot_check_motor_status = STATUS_STOP_MOTORS;
199  } else if (!(THROTTLE_STICK_DOWN() && YAW_STICK_PUSHED())) { // stick released too soon
200  autopilot_check_motor_status = STATUS_MOTORS_ON;
201  } else {
202  autopilot.arming_status = AP_ARMING_STATUS_DISARMING;
203  }
204  break;
205  case STATUS_STOP_MOTORS:
206  autopilot.motors_on = false;
207  autopilot_motors_on_counter = 0;
208  if (autopilot_arming_check_valid(YAW_MUST_BE_CENTERED)) { // wait till release disarm stick before allowing to re-arm
209  autopilot_check_motor_status = STATUS_MOTORS_OFF;
210  }
211  break;
212  default:
213  break;
214  }
215  } else {
216  autopilot.arming_status = AP_ARMING_STATUS_KILLED;
217  if (kill_switch_is_on()) {
218  autopilot.motors_on = false;
219  }
220  }
221 }
