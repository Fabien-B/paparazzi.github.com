125 void gps_skytraq_parse(uint8_t c) {
126  if (gps_skytraq.status < GOT_PAYLOAD)
127  gps_skytraq.checksum ^= c;
128  switch (gps_skytraq.status) {
129  case UNINIT:
130  if (c == SKYTRAQ_SYNC1)
131  gps_skytraq.status = GOT_SYNC1;
132  break;
133  case GOT_SYNC1:
134  if (c != SKYTRAQ_SYNC2) {
135  gps_skytraq.error_last = GPS_SKYTRAQ_ERR_OUT_OF_SYNC;
136  goto error;
137  }
138  gps_skytraq.status = GOT_SYNC2;
139  break;
140  case GOT_SYNC2:
141  gps_skytraq.len = c<<8;
142  gps_skytraq.status = GOT_LEN1;
143  break;
144  case GOT_LEN1:
145  gps_skytraq.len += c;
146  gps_skytraq.status = GOT_LEN2;
147  if (gps_skytraq.len > GPS_SKYTRAQ_MAX_PAYLOAD) {
148  gps_skytraq.error_last = GPS_SKYTRAQ_ERR_MSG_TOO_LONG;
149  goto error;
150  }
151  break;
152  case GOT_LEN2:
153  gps_skytraq.msg_id = c;
154  gps_skytraq.msg_idx = 0;
155  gps_skytraq.checksum = c;
156  gps_skytraq.status = GOT_ID;
157  break;
158  case GOT_ID:
159  gps_skytraq.msg_buf[gps_skytraq.msg_idx] = c;
160  gps_skytraq.msg_idx++;
161  if (gps_skytraq.msg_idx >= gps_skytraq.len-1) {
162  gps_skytraq.status = GOT_PAYLOAD;
163  }
164  break;
165  case GOT_PAYLOAD:
166  if (c != gps_skytraq.checksum) {
167  gps_skytraq.error_last = GPS_SKYTRAQ_ERR_CHECKSUM;
168  goto error;
169  }
170  gps_skytraq.status = GOT_CHECKSUM;
171  break;
172  case GOT_CHECKSUM:
173  if (c != SKYTRAQ_SYNC3) {
174  gps_skytraq.error_last = GPS_SKYTRAQ_ERR_OUT_OF_SYNC;
175  goto error;
176  }
177  gps_skytraq.status = GOT_SYNC3;
178  break;
179  case GOT_SYNC3:
180  gps_skytraq.msg_available = TRUE;
181  goto restart;
182  default:
183  gps_skytraq.error_last = GPS_SKYTRAQ_ERR_UNEXPECTED;
184  goto error;
185  }
186  return;
187  error:
188  gps_skytraq.error_cnt++;
189  restart:
190  gps_skytraq.status = UNINIT;
191  return;
192 }
