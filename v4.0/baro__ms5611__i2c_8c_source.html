62 static int8_t baro_ms5611_crc(uint16_t* prom) {
63  int32_t i, j;
64  uint32_t res = 0;
65  uint8_t crc = prom[7] & 0xF;
66  prom[7] &= 0xFF00;
67  for (i = 0; i < 16; i++) {
68  if (i & 1) res ^= ((prom[i>>1]) & 0x00FF);
69  else res ^= (prom[i>>1]>>8);
70  for (j = 8; j > 0; j--) {
71  if (res & 0x8000) res ^= 0x1800;
72  res <<= 1;
73  }
74  }
75  prom[7] |= crc;
76  if (crc == ((res >> 12) & 0xF)) return 0;
77  else return -1;
78 }

85 void baro_ms5611_periodic( void ) {
86  if (cpu_time_sec > 1) {
87  if (ms5611_status >= MS5611_IDLE) {
88  /* start D1 conversion */
89  ms5611_status = MS5611_CONV_D1;
90  ms5611_trans.buf[0] = MS5611_START_CONV_D1;
91  I2CTransmit(MS5611_I2C_DEV, ms5611_trans, MS5611_SLAVE_ADDR, 1);
92  RunOnceEvery((4*30), DOWNLINK_SEND_MS5611_COEFF(DefaultChannel, DefaultDevice,
93  &ms5611_c[0], &ms5611_c[1], &ms5611_c[2], &ms5611_c[3],
94  &ms5611_c[4], &ms5611_c[5], &ms5611_c[6], &ms5611_c[7]));
95  }
96  else if (ms5611_status == MS5611_UNINIT) {
97  /* reset sensor */
98  ms5611_status = MS5611_RESET;
99  ms5611_trans.buf[0] = MS5611_SOFT_RESET;
100  I2CTransmit(MS5611_I2C_DEV, ms5611_trans, MS5611_SLAVE_ADDR, 1);
101  }
102  else if (ms5611_status == MS5611_RESET_OK) {
103  /* start getting prom data */
104  ms5611_status = MS5611_PROM;
105  ms5611_trans.buf[0] = MS5611_PROM_READ | (prom_cnt << 1);
106  I2CTransceive(MS5611_I2C_DEV, ms5611_trans, MS5611_SLAVE_ADDR, 1, 2);
107  }
108  }
109 }

133 void baro_ms5611_event( void ) {
134  if (ms5611_trans.status == I2CTransSuccess) {
135  switch (ms5611_status) {
136 
137  case MS5611_RESET:
138  ms5611_status = MS5611_RESET_OK;
139  ms5611_trans.status = I2CTransDone;
140  break;
141 
142  case MS5611_PROM:
143  /* read prom data */
144  ms5611_c[prom_cnt++] = (ms5611_trans.buf[0] << 8) | ms5611_trans.buf[1];
145  if (prom_cnt < PROM_NB) {
146  /* get next prom data */
147  ms5611_trans.buf[0] = MS5611_PROM_READ | (prom_cnt << 1);
148  I2CTransceive(MS5611_I2C_DEV, ms5611_trans, MS5611_SLAVE_ADDR, 1, 2);
149  }
150  else {
151  /* done reading prom */
152  ms5611_trans.status = I2CTransDone;
153  /* check prom crc */
154  if (baro_ms5611_crc(ms5611_c) == 0) {
155  DOWNLINK_SEND_MS5611_COEFF(DefaultChannel, DefaultDevice,
156  &ms5611_c[0], &ms5611_c[1], &ms5611_c[2], &ms5611_c[3],
157  &ms5611_c[4], &ms5611_c[5], &ms5611_c[6], &ms5611_c[7]);
158  ms5611_status = MS5611_IDLE;
159  }
160  else {
161  /* checksum error, try again */
162  baro_ms5611_init();
163  }
164  }
165  break;
166 
167  case MS5611_CONV_D1:
168  ms5611_status = MS5611_CONV_D1_OK;
169  ms5611_trans.status = I2CTransDone;
170  break;
171 
172  case MS5611_ADC_D1:
173  /* read D1 (pressure) */
174  ms5611_d1 = (ms5611_trans.buf[0] << 16) |
175  (ms5611_trans.buf[1] << 8) |
176  ms5611_trans.buf[2];
177  /* start D2 conversion */
178  ms5611_status = MS5611_CONV_D2;
179  ms5611_trans.buf[0] = MS5611_START_CONV_D2;
180  I2CTransmit(MS5611_I2C_DEV, ms5611_trans, MS5611_SLAVE_ADDR, 1);
181  break;
182 
183  case MS5611_CONV_D2:
184  ms5611_status = MS5611_CONV_D2_OK;
185  ms5611_trans.status = I2CTransDone;
186  break;
187 
188  case MS5611_ADC_D2: {
189  int64_t dt, baroms, tempms, off, sens, t2, off2, sens2;
190  /* read D2 (temperature) */
191  ms5611_d2 = (ms5611_trans.buf[0] << 16) |
192  (ms5611_trans.buf[1] << 8) |
193  ms5611_trans.buf[2];
194  ms5611_status = MS5611_IDLE;
195  ms5611_trans.status = I2CTransDone;
196 
197  /* difference between actual and ref temperature */
198  dt = ms5611_d2 - (int64_t)ms5611_c[5] * (1<<8);
199  /* actual temperature */
200  tempms = 2000 + ((int64_t)dt * ms5611_c[6]) / (1<<23);
201  /* offset at actual temperature */
202  off = ((int64_t)ms5611_c[2] * (1<<16)) + ((int64_t)ms5611_c[4] * dt) / (1<<7);
203  /* sensitivity at actual temperature */
204  sens = ((int64_t)ms5611_c[1] * (1<<15)) + ((int64_t)ms5611_c[3] * dt) / (1<<8);
205  /* second order temperature compensation */
206  if (tempms < 2000) {
207  t2 = (dt*dt) / (1<<31);
208  off2 = 5 * ((int64_t)(tempms-2000)*(tempms-2000)) / (1<<1);
209  sens2 = 5 * ((int64_t)(tempms-2000)*(tempms-2000)) / (1<<2);
210  if (tempms < -1500) {
211  off2 = off2 + 7 * (int64_t)(tempms+1500)*(tempms+1500);
212  sens2 = sens2 + 11 * ((int64_t)(tempms+1500)*(tempms+1500)) / (1<<1);
213  }
214  tempms = tempms - t2;
215  off = off - off2;
216  sens = sens - sens2;
217  }
218  /* temperature compensated pressure */
219  baroms = (((int64_t)ms5611_d1 * sens) / (1<<21) - off) / (1<<15);
220 #ifdef SENSOR_SYNC_SEND
221  ftempms = tempms / 100.;
222  fbaroms = baroms / 100.;
223  DOWNLINK_SEND_BARO_MS5611(DefaultChannel, DefaultDevice,
224  &ms5611_d1, &ms5611_d2, &fbaroms, &ftempms);
225 #endif
226  break;
227  }
228 
229  default:
230  ms5611_trans.status = I2CTransDone;
231  break;
232  }
233  }
