347  if (pprz_mode != PPRZ_MODE_HOME && pprz_mode != PPRZ_MODE_GPS_OUT_OF_ORDER && launch) {
348  if (too_far_from_home) {
349  pprz_mode = PPRZ_MODE_HOME;

353  pprz_mode = RC_LOST_MODE;

365  }
366  mode_changed |= mcu1_status_update();

368  PERIODIC_SEND_PPRZ_MODE(DefaultChannel, DefaultDevice);
