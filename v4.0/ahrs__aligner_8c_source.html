58 void ahrs_aligner_run(void) {
59 
60  RATES_ADD(gyro_sum, imu.gyro);
61  VECT3_ADD(accel_sum, imu.accel);
62  VECT3_ADD(mag_sum, imu.mag);
63 
64  ref_sensor_samples[samples_idx] = imu.accel.z;
65  samples_idx++;
66 
67 #ifdef AHRS_ALIGNER_LED
68  RunOnceEvery(50, {LED_TOGGLE(AHRS_ALIGNER_LED);});
69 #endif
70 
71  if (samples_idx >= SAMPLES_NB) {
72  int32_t avg_ref_sensor = accel_sum.z;
73  if ( avg_ref_sensor >= 0)
74  avg_ref_sensor += SAMPLES_NB / 2;
75  else
76  avg_ref_sensor -= SAMPLES_NB / 2;
77  avg_ref_sensor /= SAMPLES_NB;
78 
79  ahrs_aligner.noise = 0;
80  int i;
81  for (i=0; i<SAMPLES_NB; i++) {
82  int32_t diff = ref_sensor_samples[i] - avg_ref_sensor;
83  ahrs_aligner.noise += abs(diff);
84  }
85 
86  RATES_SDIV(ahrs_aligner.lp_gyro, gyro_sum, SAMPLES_NB);
87  VECT3_SDIV(ahrs_aligner.lp_accel, accel_sum, SAMPLES_NB);
88  VECT3_SDIV(ahrs_aligner.lp_mag, mag_sum, SAMPLES_NB);
89 
90  INT_RATES_ZERO(gyro_sum);
91  INT_VECT3_ZERO(accel_sum);
92  INT_VECT3_ZERO(mag_sum);
93  samples_idx = 0;
94 
95  if (ahrs_aligner.noise < LOW_NOISE_THRESHOLD)
96  ahrs_aligner.low_noise_cnt++;
97  else
98  if ( ahrs_aligner.low_noise_cnt > 0)
99  ahrs_aligner.low_noise_cnt--;
100 
101  if (ahrs_aligner.low_noise_cnt > LOW_NOISE_TIME) {
102  ahrs_aligner.status = AHRS_ALIGNER_LOCKED;
103 #ifdef AHRS_ALIGNER_LED
104  LED_ON(AHRS_ALIGNER_LED);
105 #endif
106  }
107  }
