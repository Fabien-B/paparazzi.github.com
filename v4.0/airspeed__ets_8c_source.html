116 void airspeed_ets_read_event( void ) {
117  int n;
118  float airspeed_tmp = 0.0;
119 
120  // Get raw airspeed from buffer
121  airspeed_ets_raw = ((uint16_t)(airspeed_ets_i2c_trans.buf[1]) << 8) | (uint16_t)(airspeed_ets_i2c_trans.buf[0]);
122  // Check if this is valid airspeed
123  if (airspeed_ets_raw == 0)
124  airspeed_ets_valid = FALSE;
125  else
126  airspeed_ets_valid = TRUE;
127 
128  // Continue only if a new airspeed value was received
129  if (airspeed_ets_valid) {
130  // Calculate offset average if not done already
131  if (!airspeed_ets_offset_init) {
132  --airspeed_ets_cnt;
133  // Check if averaging completed
134  if (airspeed_ets_cnt == 0) {
135  // Calculate average
136  airspeed_ets_offset = (uint16_t)(airspeed_ets_offset_tmp / AIRSPEED_ETS_OFFSET_NBSAMPLES_AVRG);
137  // Limit offset
138  if (airspeed_ets_offset < AIRSPEED_ETS_OFFSET_MIN)
139  airspeed_ets_offset = AIRSPEED_ETS_OFFSET_MIN;
140  if (airspeed_ets_offset > AIRSPEED_ETS_OFFSET_MAX)
141  airspeed_ets_offset = AIRSPEED_ETS_OFFSET_MAX;
142  airspeed_ets_offset_init = TRUE;
143  }
144  // Check if averaging needs to continue
145  else if (airspeed_ets_cnt <= AIRSPEED_ETS_OFFSET_NBSAMPLES_AVRG)
146  airspeed_ets_offset_tmp += airspeed_ets_raw;
147  }
148  // Convert raw to m/s
149 #ifdef AIRSPEED_ETS_REVERSE
150  if (airspeed_ets_offset_init && airspeed_ets_raw < airspeed_ets_offset)
151  airspeed_tmp = AIRSPEED_ETS_SCALE * sqrtf( (float)(airspeed_ets_offset-airspeed_ets_raw) ) - AIRSPEED_ETS_OFFSET;
152 #else
153  if (airspeed_ets_offset_init && airspeed_ets_raw > airspeed_ets_offset)
154  airspeed_tmp = AIRSPEED_ETS_SCALE * sqrtf( (float)(airspeed_ets_raw-airspeed_ets_offset) ) - AIRSPEED_ETS_OFFSET;
155 #endif
156  else
157  airspeed_tmp = 0.0;
158  // Airspeed should always be positive
159  if (airspeed_tmp < 0.0)
160  airspeed_tmp = 0.0;
161  // Moving average
162  airspeed_ets_buffer[airspeed_ets_buffer_idx++] = airspeed_tmp;
163  if (airspeed_ets_buffer_idx >= AIRSPEED_ETS_NBSAMPLES_AVRG)
164  airspeed_ets_buffer_idx = 0;
165  airspeed_ets = 0.0;
166  for (n = 0; n < AIRSPEED_ETS_NBSAMPLES_AVRG; ++n)
167  airspeed_ets += airspeed_ets_buffer[n];
168  airspeed_ets = airspeed_ets / (float)AIRSPEED_ETS_NBSAMPLES_AVRG;
169 #if USE_AIRSPEED
170  EstimatorSetAirspeed(airspeed_ets);
171 #endif
172 #ifdef SENSOR_SYNC_SEND
173  DOWNLINK_SEND_AIRSPEED_ETS(DefaultChannel, DefaultDevice, &airspeed_ets_raw, &airspeed_ets_offset, &airspeed_ets);
174 #endif
175  } else {
176  airspeed_ets = 0.0;
177  }
178 
179  // Transaction has been read
180  airspeed_ets_i2c_trans.status = I2CTransDone;
181 }
